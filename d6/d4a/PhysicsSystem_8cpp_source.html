109void PhysicsSystem::start() {
110 std::lock_guard lock(g_threadLifecycleMutex);
111 auto i = PhysicsSystem::get();
112 if (!i.has_value()) {
113 return;
114 }
115 auto* physics = i.value();
116
117 // Clear intentional stop so supervisor may restart the thread if needed
118 g_intentionallyStopped.store(false, std::memory_order_release);
119
120 // If thread is already running, nothing to do
121 if (g_threadAlive.load(std::memory_order_acquire)) {
122 return;
123 }
124
125 // If there is an old thread object, join it to clean up resources
126 if (physics->thread.joinable()) {
127 try {
128 physics->thread.join();
129 } catch (...) { }
130 }
131
132 physics->thread = std::jthread([physics](std::stop_token token) { // NOLINT
133 g_threadAlive.store(true, std::memory_order_release);
134 try {
135 // TracyFiberEnter("Physics Thread");
136 while (!token.stop_requested()) {
137 using namespace std::chrono;
138 time_point begin = steady_clock::now();
139 PROFILE_ZONE_N("physics::simulation");
140
141 // Apply queued add/remove requests on the physics thread before
142 // building this frame's cache to keep pointer lifetimes safe
143 physics->FlushPendingMutations();
144
145 // Loop the physics simulation a set amount of times per frame
146 physics->CachePhysicsObjects();
147 for (int i = 0; i < physics->m.tickCount; i++) {
148 Time::GetInstance()->PhysTick();
149 physics->Tick();
150
151 // Interrupt the loop if we're running out of budget
152 duration elapsed = steady_clock::now() - begin;
153 if (elapsed >= physics->m.targetFrametime) {
154 break;
155 }
156 }
157
158 // Handle constant frame time
159 duration elapsed = steady_clock::now() - begin;
160 if (elapsed < physics->m.targetFrametime) {
161 PROFILE_ZONE_NC("physics::wait", 0x404040);
162 std::this_thread::sleep_for(physics->m.targetFrametime - elapsed);
163 }
164 }
165 } catch (const std::exception& e) { TOAST_ERROR("Physics thread crashed with exception: {}", e.what()); } catch (...) {
166 TOAST_ERROR("Physics thread crashed with unknown exception");
167 }
168
169 // When we stop the physics thread, restore rigidbody velocities
170 {
171 std::lock_guard sim_lock(physics->m.simulationMutex);
172 for (auto* rb : physics->m.rigidbodies) {
173 RbResetVelocity(rb);
174 }
175
176 for (auto* rb : physics->m.boxes) {
177 BoxResetVelocity(rb);
178 }
179 }
180
181 // Mark thread as no longer alive
182 g_threadAlive.store(false, std::memory_order_release);
183 // TracyFiberLeave;
184 });
185}

187void PhysicsSystem::stop() {
188 PhysicsSystem* physicsPtr = nullptr;
189 bool threadJoinable = false;
190 {
191 std::lock_guard lock(g_threadLifecycleMutex);
192 auto i = PhysicsSystem::get();
193 if (!i.has_value()) {
194 return;
195 }
196 physicsPtr = *i;
197 // Mark as intentionally stopped so main-thread supervisor doesn't respawn the thread
198 g_intentionallyStopped.store(true, std::memory_order_release);
199 threadJoinable = physicsPtr->thread.joinable();
200 if (threadJoinable) {
201 physicsPtr->thread.request_stop();
202 }
203 }
204
205 // Join outside of the lifecycle mutex to avoid deadlocks
206 if (threadJoinable) {
207 try {
208 physicsPtr->thread.join();
209 } catch (...) { }
210 }
211 g_threadAlive.store(false, std::memory_order_release);
212
213 if (physicsPtr != nullptr) {
214 physicsPtr->FlushPendingMutations();
215 }
216}

220void PhysicsSystem::Tick() {
221 PROFILE_ZONE;
222
223 std::list<std::function<void()>> localCallbacks;
224
225 {
226 std::lock_guard sim_lock(m.simulationMutex);
227
228 // Store previous positions for interpolation before physics step
229 for (auto* rb : m.cachedRigidbodies) {
230 rb->StorePreviousPosition();
231 }
232
233 // Propagate the PhysTick down the object tree as first
234 toast::World::Instance()->PhysTick();
235
236 // Handle Rigidbody physics
237 for (auto* rb : m.cachedRigidbodies) {
238 RigidbodyPhysics(rb, localCallbacks);
239 }
240
241 // Handle Box physics
242 for (auto* rb : m.cachedBoxRigidbodies) {
243 BoxPhysics(rb);
244 }
245 }
246
247 // Record the time of this physics step for interpolation
248 m.lastPhysicsTime.store(std::chrono::steady_clock::now(), std::memory_order_release);
249
250 // Append local callbacks to the global list after releasing the simulation lock
251 if (!localCallbacks.empty()) {
252 std::lock_guard lock(m.callbackMutex);
253 m.callbackList.splice(m.callbackList.end(), localCallbacks);
254 }
255}

264void PhysicsSystem::FlushPendingMutations() {
265 std::deque<PendingMutation> localMutations;
266 {
267 std::lock_guard queue_lock(m.mutationQueueMutex);
268 if (m.pendingMutations.empty()) {
269 return;
270 }
271 localMutations.swap(m.pendingMutations);
272 }
273
274 std::vector<std::shared_ptr<std::promise<void>>> completions;
275 {
276 std::lock_guard sim_lock(m.simulationMutex);
277 for (auto& mutation : localMutations) {
278 ApplyPendingMutation(mutation);
279 if (mutation.completion != nullptr) {
280 completions.emplace_back(std::move(mutation.completion));
281 }
282 }
283 }
284
285 for (const auto& completion : completions) {
286 completion->set_value();
287 }
288}

290void PhysicsSystem::ApplyPendingMutation(const PendingMutation& mutation) {
291 switch (mutation.type) {
292 case MutationType::AddRigidbody: {
293 auto* rb = static_cast<Rigidbody*>(mutation.object);
294 if (std::ranges::find(m.rigidbodies, rb) == m.rigidbodies.end()) {
295 m.rigidbodies.emplace_back(rb);
296 }
297 break;
298 }
299
300 case MutationType::RemoveRigidbody: {
301 auto* rb = static_cast<Rigidbody*>(mutation.object);
302 m.rigidbodies.remove(rb);
303 m.colliding.remove(rb);
304 break;
305 }
306
307 case MutationType::AddCollider: {
308 auto* collider = static_cast<ConvexCollider*>(mutation.object);
309 if (std::ranges::find(m.colliders, collider) == m.colliders.end()) {
310 m.colliders.emplace_back(collider);
311 }
312 break;
313 }
314
315 case MutationType::RemoveCollider: {
316 auto* collider = static_cast<ConvexCollider*>(mutation.object);
317 m.colliders.remove(collider);
318 break;
319 }
320
321 case MutationType::AddTrigger: {
322 auto* trigger = static_cast<Trigger*>(mutation.object);
323 if (std::ranges::find(m.triggers, trigger) == m.triggers.end()) {
324 m.triggers.emplace_back(trigger);
325 }
326 break;
327 }
328
329 case MutationType::RemoveTrigger: {
330 auto* trigger = static_cast<Trigger*>(mutation.object);
331 m.triggers.remove(trigger);
332 break;
333 }
334
335 case MutationType::AddBox: {
336 auto* box = static_cast<BoxRigidbody*>(mutation.object);
337 if (std::ranges::find(m.boxes, box) == m.boxes.end()) {
338 m.boxes.emplace_back(box);
339 }
340 break;
341 }
342
343 case MutationType::RemoveBox: {
344 auto* box = static_cast<BoxRigidbody*>(mutation.object);
345 m.boxes.remove(box);
346 break;
347 }
348 }
349}

617void PhysicsSystem::UpdateVisualInterpolation() {
618 auto i = PhysicsSystem::get();
619 if (!i.has_value()) {
620 return;
621 }
622 auto* physics = *i;
623 std::lock_guard sim_lock(physics->m.simulationMutex);
624
625 // Calculate time elapsed since last physics step
626 auto now = std::chrono::steady_clock::now();
627 auto last_physics = physics->m.lastPhysicsTime.load(std::memory_order_acquire);
628 std::chrono::duration<double> elapsed = now - last_physics;
629
630 // Calculate interpolation alpha
631 double fixed_dt = physics->m.targetFrametime.count();
632 double alpha = glm::clamp(elapsed.count() / fixed_dt, 0.0, 1.0);
633
634 // Update global interpolation alpha
635 Rigidbody::UpdateInterpolationAlpha(alpha);
636
637 // Update all rigidbody visual transforms
638 for (auto* rb : physics->m.rigidbodies) {
639 rb->UpdateVisualTransform();
640 }
641}

651void PhysicsSystem::RigidbodyPhysics(Rigidbody* rb, std::list<std::function<void()>>& localCallbacks) {
652 if (not rb->enabled()) {
653 return;
654 }
655
656 PROFILE_ZONE;
657 const auto& name = rb->parent()->name();
658 PROFILE_TEXT(name.c_str(), name.size());
659
660 RbKinematics(rb);
661
662 // Collision loops - operate only on cached (visible/active) physics lists to improve performance
663
664 // Rigidbody vs Rigidbody (only iterate cached rigidbodies and only those after this one to avoid double checks)
665 {
666 auto cached_it = std::ranges::find(m.cachedRigidbodies, rb);
667 if (cached_it != m.cachedRigidbodies.end()) {
668 for (auto it = std::next(cached_it); it != m.cachedRigidbodies.end(); ++it) {
669 if (not(*it)->enabled()) {
670 continue;
671 }
672
673 auto manifold = RbRbCollision(rb, *it);
674 if (manifold.has_value()) {
675 RbRbResolution(rb, *it, manifold.value());
676
677 if (rb->enterCallback && rb->CanCallBack(*it)) {
678 auto* other = *it;
679 localCallbacks.emplace_back([rb, other]() {
680 rb->enterCallback(other);
681 });
682 }
683
684 if ((*it)->enterCallback && (*it)->CanCallBack(rb)) {
685 auto* other = *it;
686 localCallbacks.emplace_back([rb, other]() {
687 other->enterCallback(rb);
688 });
689 }
690
691 if (std::ranges::find(m.colliding, rb) != m.colliding.end()) {
692 m.colliding.emplace_back(rb);
693 }
694
695 if (std::ranges::find(m.colliding, *it) != m.colliding.end()) {
696 m.colliding.emplace_back(*it);
697 }
698 } else {
699 auto find = std::ranges::find(m.colliding, rb);
700 if (find != m.colliding.end()) {
701 auto* other = *it;
702 localCallbacks.emplace_back([rb, other]() {
703 rb->exitCallback(other);
704 });
705 m.colliding.erase(find);
706 }
707
708 find = std::ranges::find(m.colliding, *it);
709 if (find != m.colliding.end()) {
710 auto* other = *it;
711 localCallbacks.emplace_back([rb, other]() {
712 other->exitCallback(rb);
713 });
714 m.colliding.erase(find);
715 }
716 }
717 }
718 }
719 }
720
721 // Rigidbody vs Box (only cached boxes)
722 for (auto* b : m.cachedBoxRigidbodies) {
723 auto manifold = RbBoxCollision(rb, b);
724 if (manifold.has_value()) {
725 RbBoxResolution(rb, b, manifold.value());
726 }
727 }
728
729 // Rigidbody vs Convex Colliders (only cached convex colliders)
730 for (auto* c : m.cachedConvexColliders) {
731 if (not c->parent->enabled()) {
732 continue;
733 }
734
735 auto manifold = RbMeshCollision(rb, c);
736 if (manifold.has_value()) {
737 RbMeshResolution(rb, c, manifold.value());
738 }
739 }
740
741 for (auto* t : m.triggers) {
742 if (not t->enabled()) {
743 continue;
744 }
745
746 RbTriggerCollision(rb, t, localCallbacks);
747 }
748
749 // Final position integration
750 RbIntegration(rb);
751}

753void PhysicsSystem::BoxPhysics(BoxRigidbody* rb) {
754 if (not rb->enabled()) {
755 return;
756 }
757
758 PROFILE_ZONE;
759 // PROFILE_TEXT(rb->parent()->name(), rb->parent()->name().size());
760
761 BoxKinematics(rb);
762
763 // Collision loops - operate only on cached (visible/active) physics lists to improve performance
764
765 // Box vs Convex Colliders (only cached convex colliders)
766 for (auto* c : m.cachedConvexColliders) {
767 if (not c->parent->enabled()) {
768 continue;
769 }
770

794void PhysicsSystem::CachePhysicsObjects() {
795 std::lock_guard sim_lock(m.simulationMutex);
796
797 // detects if inside frustum or out, cached lists contains in screen physics
798 auto rb_view = m.rigidbodies | std::views::filter([](auto* rb) -> bool {
799 if (rb->m_skipBoundsCheck) {
800 return true;
801 }
802 auto pstition = rb->GetPosition();
803 return OclussionVolume::isSphereOnPlanes(glm::vec3(pstition.x, pstition.y, 0.f), rb->radius);
804 });
805 m.cachedRigidbodies.assign(rb_view.begin(), rb_view.end());
806
807 auto box_view = m.boxes | std::views::filter([](auto* rb) -> bool {
808 return OclussionVolume::isAABBOnPlanes(rb->GetAABB());
809 });
810 m.cachedBoxRigidbodies.assign(box_view.begin(), box_view.end());
811
812 // Cache visible convex colliders for this simulation frame.
813 auto convex_view = m.colliders | std::views::filter([](auto* rb) -> bool {
814 return OclussionVolume::isAABBOnPlanes(rb->getAABB());
815 });
816 m.cachedConvexColliders.assign(convex_view.begin(), convex_view.end());

819std::optional<RayResult> PhysicsSystem::RayCollision(Line* ray, ColliderFlags flags) {
820 if (not get().has_value()) {
821 TOAST_WARN("Raycast skipped because physics system doesn't exist");
822 return std::nullopt;
823 }
824 auto* physics = get().value();
825 std::lock_guard sim_lock(physics->m.simulationMutex);
826
827 std::optional<RayResult> result = std::nullopt;
828 std::optional<dvec2> col_hit;
829 std::optional<dvec2> rb_hit;
830
831 for (auto* c : physics->m.colliders) {
832 if (!c->parent->enabled()) {
833 continue;
834 }
835
836 if ((static_cast<unsigned int>(flags) & static_cast<unsigned int>(c->flags)) == 0u) {
837 continue;
838 }
839 auto collision = ConvexRayCollision(ray, c);
840 if (not collision.has_value()) {
841 continue;
842 }
843
844 if (not col_hit.has_value() || length2(collision->first - ray->p1) < length2(*col_hit - ray->p1)) {
845 col_hit = collision->first;
846 const float d = static_cast<float>(distance(*col_hit, ray->p1));
847
848 // same as below
849 if (result && result.value().distance < d) {
850 continue;
851 }
852 result = { .type = RayResult::Collider, .point = *col_hit, .normal = collision->second, .distance = d, .other = c->parent };
853 }
854 }
855
856 for (auto* r : physics->m.rigidbodies) {
857 if (!r->enabled()) {
858 continue;
859 }
860
861 if ((static_cast<unsigned int>(flags) & static_cast<unsigned int>(r->flags)) == 0u) {
862 continue;
863 }
864 std::optional<dvec2> collision = RbRayCollision(ray, r);
865 if (not collision.has_value()) {
866 continue;
867 }
868
869 if (not rb_hit.has_value() || length(*collision - ray->p1) < length2(*rb_hit - ray->p1)) {
870 rb_hit = collision.value();
871 const float d = static_cast<float>(distance(*rb_hit, ray->p1));
872
873 // Do not modify result if theres already one with less distance
874 if (result && result.value().distance < d) {
875 continue;
876 }
877 result = { .type = RayResult::Rigidbody, .point = *rb_hit, .normal = ray->tangent, .distance = d, .other = r->parent() };
878 }
879 }
880 if (result != std::nullopt) {
881 // renderer::DebugLine(ray->p1, result->point, vec4(0.0f, 0.0f, 1.0f, 1.0f));
882 }
883
884 return result;
885}

887toast::Object* PhysicsSystem::PointCollision(glm::vec2 point, ColliderFlags flags) {
888 if (not get().has_value()) {
889 return nullptr;
890 }
891 auto* physics = get().value();
892 std::lock_guard sim_lock(physics->m.simulationMutex);
893
894 const dvec2 pt { point };
895
896 // Colliders
897 for (auto* c : physics->m.colliders) {
898 if (!c->parent->enabled()) {
899 continue;
900 }
901 if ((static_cast<unsigned int>(flags) & static_cast<unsigned int>(c->flags)) == 0u) {
902 continue;
903 }
904 if (c->edges.empty()) {
905 continue;
906 }
907
908 bool inside = true;
909 for (const auto& edge : c->edges) {
910 dvec2 to_point = pt - edge.p1;
911 // If dot with the outward normal is positive, point is outside this edge
912 if (glm::dot(to_point, edge.normal) > 0.0) {
913 inside = false;
914 break;
915 }
916 }
917 if (inside) {
918 return c->parent->parent();
919 }
920 }
921
922 // Check rigidbodies
923 for (auto* rb : physics->m.rigidbodies) {
924 if (!rb->enabled()) {
925 continue;
926 }
927 if ((static_cast<unsigned int>(flags) & static_cast<unsigned int>(rb->flags)) == 0u) {
928 continue;
929 }
930 if (glm::length2(pt - rb->GetPosition()) <= rb->radius * rb->radius) {
931 return rb->parent();
932 }
933 }
934
935 return nullptr;
936}
