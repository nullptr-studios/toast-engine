46void PhysicsSystem::start() {
47 auto i = PhysicsSystem::get();
48 if (!i.has_value()) {
49 return;
50 }
51 auto* physics = i.value();
52 if (physics->thread.joinable()) {
53 return;
54 }
55
56 physics->thread = std::jthread([physics](std::stop_token token) { // NOLINT
57 while (!token.stop_requested()) {
58 using namespace std::chrono;
59 time_point begin = steady_clock::now();
60
61 // Loop the physics simulation a set amount of times per frame
62 for (int i = 0; i < physics->m.tickCount; i++) {
63 PROFILE_ZONE_N("physics::simulation");
64 Time::GetInstance()->PhysTick();
65 physics->Tick();
66
67 // Interrupt the loop if we're running out of budget
68 duration elapsed = steady_clock::now() - begin;
69 if (elapsed >= physics->m.targetFrametime) {
70 break;
71 }
72 }
73
74 // Handle constant frame time
75 duration elapsed = steady_clock::now() - begin;
76 if (elapsed < physics->m.targetFrametime) {
77 PROFILE_ZONE_NC("physics::wait", 0x404040);
78 std::this_thread::sleep_for(physics->m.targetFrametime - elapsed);
79 }
80 }
81
82 // When we stop the physics thread, restore rigidbody velocities
83 for (auto* rb : physics->m.rigidbodies) {
84 RbResetVelocity(rb);
85 }
86 });
87}

213void PhysicsSystem::RigidbodyPhysics(Rigidbody* rb) {
214 PROFILE_ZONE;
215 const auto& name = rb->parent()->name();
216 PROFILE_TEXT(name.c_str(), name.size());
217
218 RbKinematics(rb);
219
220 // Collision loops
221
222 for (auto it = ++std::ranges::find(m.rigidbodies, rb); it != m.rigidbodies.end(); ++it) {
223 auto manifold = RbRbCollision(rb, *it);
224 if (manifold.has_value()) {
225 RbRbResolution(rb, *it, manifold.value());
226 }
227 }
228
229 for (auto* c : m.colliders) {
230 auto manifold = RbMeshCollision(rb, c);
231 if (manifold.has_value()) {
232 RbMeshResolution(rb, c, manifold.value());
233 }
234 }
235
236 // Final position integration
237 RbIntegration(rb);
238}
