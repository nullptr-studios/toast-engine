54void PhysicsSystem::start() {
55 auto i = PhysicsSystem::get();
56 if (!i.has_value()) {
57 return;
58 }
59 auto* physics = i.value();
60 if (physics->thread.joinable()) {
61 return;
62 }
63
64 physics->thread = std::jthread([physics](std::stop_token token) { // NOLINT
65 while (!token.stop_requested()) {
66 using namespace std::chrono;
67 time_point begin = steady_clock::now();
68
69 // Loop the physics simulation a set amount of times per frame
70 for (int i = 0; i < physics->m.tickCount; i++) {
71 PROFILE_ZONE_N("physics::simulation");
72 Time::GetInstance()->PhysTick();
73 physics->Tick();
74
75 // Interrupt the loop if we're running out of budget
76 duration elapsed = steady_clock::now() - begin;
77 if (elapsed >= physics->m.targetFrametime) {
78 break;
79 }
80 }
81
82 // Handle constant frame time
83 duration elapsed = steady_clock::now() - begin;
84 if (elapsed < physics->m.targetFrametime) {
85 PROFILE_ZONE_NC("physics::wait", 0x404040);
86 std::this_thread::sleep_for(physics->m.targetFrametime - elapsed);
87 }
88 }
89
90 // When we stop the physics thread, restore rigidbody velocities
91 for (auto* rb : physics->m.rigidbodies) {
92 RbResetVelocity(rb);
93 }
94
95 for (auto* rb : physics->m.boxes) {
96 BoxResetVelocity(rb);
97 }
98 });
99}

120void PhysicsSystem::Tick() {
121 PROFILE_ZONE;
122
123 // Store previous positions for interpolation before physics step
124 for (auto* rb : m.rigidbodies) {
125 rb->StorePreviousPosition();
126 }
127
128 // Propagate the PhysTick down the object tree as first
129 toast::World::Instance()->PhysTick();
130
131 // Handle Rigidbody physics
132 for (auto* rb : m.rigidbodies) {
133 RigidbodyPhysics(rb);
134 }
135
136 // Handle Box physics
137 for (auto* rb : m.boxes) {
138 BoxPhysics(rb);
139 }
140
141 // Record the time of this physics step for interpolation
142 m.lastPhysicsTime.store(std::chrono::steady_clock::now(), std::memory_order_release);
143}

283 auto i = PhysicsSystem::get();
284 if (!i.has_value()) {
285 return;
286 }
287 auto* physics = *i;
288
289 // Calculate time elapsed since last physics step
290 auto now = std::chrono::steady_clock::now();
291 auto last_physics = physics->m.lastPhysicsTime.load(std::memory_order_acquire);
292 std::chrono::duration<double> elapsed = now - last_physics;
293
294 // Calculate interpolation alpha

296 double alpha = glm::clamp(elapsed.count() / fixed_dt, 0.0, 1.0);
297
298 // Update global interpolation alpha
299 Rigidbody::UpdateInterpolationAlpha(alpha);
300
301 // Update all rigidbody visual transforms

368std::optional<RayResult> PhysicsSystem::RayCollision(Line* ray, ColliderFlags flags) {
369 if (not get().has_value()) {
370 TOAST_WARN("Raycast skipped because physics system doesn't exist");
371 return std::nullopt;
372 }
373 auto* physics = get().value();
374
375 std::optional<RayResult> result = std::nullopt;
376 std::optional<dvec2> col_hit;
377 std::optional<dvec2> rb_hit;
378
379 for (auto* c : physics->m.colliders) {
380 if ((static_cast<unsigned int>(flags) & static_cast<unsigned int>(c->flags)) == 0u) {
381 continue;
382 }
383 auto collision = ConvexRayCollision(ray, c);
384 if (not collision.has_value()) {
385 continue;
386 }
387
388 if (not col_hit.has_value() || length2(collision->first - ray->p1) < length2(*col_hit - ray->p1)) {
389 col_hit = collision->first;
390 const float d = static_cast<float>(distance(*col_hit, ray->p1));
391
392 // same as below
393 if (result && result.value().distance < d) {
394 continue;
395 }
396 result = { .type = RayResult::Collider, .point = *col_hit, .normal = collision->second, .distance = d, .other = c->parent };
397 }
398 }
399
400 for (auto* r : physics->m.rigidbodies) {
401 if ((static_cast<unsigned int>(flags) & static_cast<unsigned int>(r->flags)) == 0u) {
402 continue;
403 }
404 std::optional<dvec2> collision = RbRayCollision(ray, r);
405 if (not collision.has_value()) {
406 continue;
407 }
408
409 if (not rb_hit.has_value() || length(*collision - ray->p1) < length2(*rb_hit - ray->p1)) {
410 rb_hit = collision.value();
411 const float d = static_cast<float>(distance(*rb_hit, ray->p1));
412
413 // Do not modify result if theres already one with less distance
414 if (result && result.value().distance < d) {
415 continue;
416 }
417 result = { .type = RayResult::Rigidbody, .point = *rb_hit, .normal = ray->tangent, .distance = d, .other = r->parent() };
418 }
419 }
420 if (result != std::nullopt) {
421 renderer::DebugLine(ray->p1, result->point, vec4(0.0f, 0.0f, 1.0f, 1.0f));
422 }
423
424 return result;
425}
