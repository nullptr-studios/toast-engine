65 void CalculateValue() {
66 // If no keys are pressed, set value to zero and mark as finished
67 if (m.pressedKeys.empty()) {
68 value = static_cast<Value>(0);
69 // Only transition to Finished if we were active (Started or Ongoing)
70 // This prevents spamming Finished events when already inactive
71 if (state == Started || state == Ongoing) {
72 state = Finished;
73 }
74 return;
75 }
76
77 // Sum all pressed key values to get the final action value
78 // For boolean actions, this is just "is any key pressed?"
79 // For 1D/2D actions, this combines directional inputs (e.g., WASD, analog sticks)
80 value = std::accumulate(m.pressedKeys.begin(), m.pressedKeys.end(), static_cast<Value>(0.0f), [](Value a, const std::pair<int, Value>& pair) {
81 return a + pair.second;
82 });
83
84 // Prevent values from exceeding valid bounds
85 if constexpr (std::is_same_v<Value, float>) {
86 constexpr float MIN = -1.0f;
87 constexpr float MAX = 1.0f;
88 value = std::clamp(value, MIN, MAX);
89 } else if constexpr (std::is_same_v<Value, glm::vec2>) {
90 glm::vec2 min = { -1.0f, -1.0f };
91 glm::vec2 max = { 1.0f, 1.0f };
92 value = glm::clamp(value, min, max);
93 }
94
95 // Mouse position/delta/scroll are one-frame events
96 // They should always be Ongoing, never Started
97 if (HasOnlyTransientInputs()) {
98 state = Ongoing;
99 return;
100 }
101
102 // Update Action State
103 if (state == Finished || state == Null) {
104 state = Started;
105 } else if (state == Started) {
106 state = Ongoing;
107 }
108 // If already Ongoing, stay Ongoing
109 }

112 bool CheckState(std::string_view state) {
113 if (m.states.empty()) {
114 return true;
115 }
116
117 bool blocked = false;
118 bool has_positive = false;
119 bool matched_positive = false;
120
121 for (const std::string& s : m.states) {
122 if (s.empty()) {
123 continue;
124 }
125
126 if (s.starts_with('-')) {
127 // if it matches the current state, block
128 if (s.substr(1) == state) {
129 blocked = true;
130 break;
131 }
132 continue;
133 }
134
135 // record presence and match
136 has_positive = true;
137 if (s == state) {
138 matched_positive = true;
139 }
140 }
141
142 if (blocked) {
143 return false;
144 }
145 // If any positive gates exist, require a positive match, otherwise allow
146 return has_positive ? matched_positive : true;
147 }

160template<ActionValues Value>
161auto Action<Value>::create(std::string_view name, const sol::object& obj) noexcept -> std::optional<Action<Value>> {
162 // check the object is the correct type
163 if (!obj.is<sol::table>()) {
164 TOAST_ERROR("Action object is not a Lua Table");
165 return std::nullopt;
166 }
167 auto action_lua = obj.as<sol::table>();
168 if (action_lua.empty()) {
169 TOAST_ERROR("Action is empty");
170 return std::nullopt;
171 }
172
173 // creating our action
174 Action<Value> action;
175 action.name = name;
176
177 // States
178 auto states_lua = action_lua["states"].get_or_create<sol::table>();
179 action.m.states.reserve(states_lua.size());
180 for (const auto& [_, s] : states_lua) {
181 // check if it's a string beforehand
182 if (!s.is<std::string>()) {
183 continue;
184 }
185 std::string state = s.as<std::string>();
186 action.m.states.emplace_back(state);
187 }
188
189 TOAST_TRACE("Action has {} states", action.m.states.size());
190
191 // Binds
192 auto binds_lua = action_lua["binds"].get_or_create<sol::table>();
193 action.m.binds.reserve(binds_lua.size());
194 for (const auto& [_, a] : binds_lua) {
195 auto bind = Bind::create(a);
196 if (!bind.has_value()) {
197 continue;
198 }
199 action.m.binds.emplace_back(*bind);
200 }
201
202 // If creating a bind was not possible, we return nullopt
203 if (binds_lua.empty()) {
204 TOAST_ERROR("Action doesn't have binds");
205 return std::nullopt;
206 }
207 TOAST_TRACE("Action has {} binds", action.m.binds.size());
208
209 TOAST_INFO("Created Action \"{}\"", action.name);
210 return action;
211}
