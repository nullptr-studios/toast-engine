103void MeshRendererComponent::OnRender(const glm::mat4& precomputed_mat) noexcept {
104 if (!enabled()) {
105 return;
106 }
107
108 if (!OclussionVolume::isTransformedAABBOnPlanes(
109 renderer::IRendererBase::GetInstance()->GetFrustumPlanes(), m_mesh->boundingBox(), GetWorldMatrix()
110 )) {
111 return;
112 }
113
114 // guard against null pointers (material or mesh might have failed to load)
115 if (m_material == nullptr) {
116 return;
117 }
118 if (m_material->GetShader() == nullptr || m_mesh == nullptr) {
119 return;
120 }
121
122 PROFILE_ZONE;
123
124 // compute transform once
125 const glm::mat4 model = GetWorldMatrix();
126 const glm::mat4 mvp = precomputed_mat * model;
127
128 m_material->Use();
129 auto shader = m_material->GetShader();
130 if (shader) {
131 // upload world matrix for deferred / lighting passes
132 shader->Set("gWorld", model);
133
134 // set generic transform uniform
135 shader->Set("gMVP", mvp);
136 }
137
138 // draw
139 m_mesh->Draw();
