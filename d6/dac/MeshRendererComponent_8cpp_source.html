194void MeshRendererComponent::OnRender(renderer::IRenderablePass pass, const glm::mat4& precomputed_mat) noexcept {
195 if (!enabled()) {
196 return;
197 }
198
199 if (m_mesh == nullptr) {
200 return;
201 }
202
203 const bool external_only = m_useExternalTextureOnly && m_useExternalTexture && m_externalTextureId != 0;
204 if (!external_only && m_material == nullptr) {
205 return;
206 }
207
208 if (pass != renderer::IRenderablePass::DIRECTIONAL_SHADOW &&
209 !OclussionVolume::isTransformedAABBOnPlanes(m_mesh->boundingBox(), GetWorldMatrix())) {
210 return;
211 }
212
213 // guard against null pointers (material or mesh might have failed to load)
214 if (!external_only && m_material->GetShader() == nullptr) {
215 return;
216 }
217
218 if (pass == renderer::IRenderablePass::OCCLUSION) {
219 if (!m_isOccluder) {
220 return;
221 }
222 }
223
224 if (pass == renderer::IRenderablePass::DIRECTIONAL_SHADOW) {
225 if (!m_castsDirectionalShadow) {
226 return;
227 }
228 }
229
230 PROFILE_ZONE;
231
232 // compute transform once
233 const glm::mat4 model = GetWorldMatrix();
234 const glm::mat4 mvp = precomputed_mat * model;
235 if (pass == renderer::IRenderablePass::GEOMETRY) {
236 auto shader = external_only ? m_externalShader : m_material->GetShader();
237 if (shader) {
238 if (external_only || (m_useExternalTexture && m_externalTextureId != 0)) {
239 shader->Use();
240 } else {
241 m_material->Use();
242 }
243
244 shader->Set("gWorld", model);
245
246 // set generic transform uniform
247 shader->Set("gMVP", mvp);
248
249 if (external_only) {
250 // default.shader uses gColor; keep neutral tint for HUD
251 shader->Set("gColor", glm::vec4(1.0f));
252 }
253
254 if (m_useExternalTexture && m_externalTextureId != 0) {
255 glActiveTexture(GL_TEXTURE0 + m_externalTextureUnit);
256 glBindTexture(GL_TEXTURE_2D, m_externalTextureId);
257 shader->SetSampler(m_externalTextureSampler, m_externalTextureUnit);
258 }
259
260 const bool receivesDirectionalShadows = (shader.get() == m_externalShader.get()) || (shader.get() == m_defaultShader.get());
261 if (auto* renderer = renderer::IRendererBase::GetInstance(); renderer && receivesDirectionalShadows) {
262 shader->Set("gDirectionalLightMatrix", renderer->GetDirectionalShadowMatrix());
263 shader->Set("gDirectionalLightDir", renderer->GetGlobalLightDirection());
264 shader->Set("gDirectionalShadowBias", renderer->GetDirectionalShadowBias());
265 shader->Set("gDirectionalShadowStrength", renderer->GetDirectionalShadowStrength());
266 shader->Set("gDirectionalShadowsEnabled", renderer->IsDirectionalShadowsEnabled() ? 1 : 0);
267
268 glActiveTexture(GL_TEXTURE15);
269 glBindTexture(GL_TEXTURE_2D, renderer->GetDirectionalShadowMapTexture());
270 shader->SetSampler("gDirectionalShadowMap", 15);
271 glActiveTexture(GL_TEXTURE0);
272 }
273
274 ApplyCustomUniforms(shader.get());
275 }
276
277 // draw
278
279 // restore state
280 // m_texture->Unbind();
281 // m_shader->unuse();
282 } else if (pass == renderer::IRenderablePass::OCCLUSION || pass == renderer::IRenderablePass::DIRECTIONAL_SHADOW) {
283 // Ensure alpha-test shader samples a valid texture binding in depth-only passes.
284 if (m_useExternalTexture && m_externalTextureId != 0) {
285 glActiveTexture(GL_TEXTURE0);
286 glBindTexture(GL_TEXTURE_2D, m_externalTextureId);
287 } else if (!external_only && m_material) {
288 m_material->Use();
289 }
290
291 m_occlusionShader->Use();
292 m_occlusionShader->Set("gWorld", model);
293 m_occlusionShader->SetSampler("Texture", 0);
294
295 // set generic transform uniform
296 m_occlusionShader->Set("gMVP", mvp);
297 }
298
299 if (pass == renderer::IRenderablePass::DIRECTIONAL_SHADOW) {
300 // Shadow-map rendering must always write depth for valid caster silhouettes.
301 glDepthMask(GL_TRUE);
302 } else if (m_drawToDepth) {
303 glDepthMask(GL_TRUE);
304 } else {
305 glDepthMask(GL_FALSE);
306 }
307
308 if (pass != renderer::IRenderablePass::OCCLUSION && pass != renderer::IRenderablePass::DIRECTIONAL_SHADOW) {
309 glEnable(GL_BLEND);
310 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
311 } else {
312 glDisable(GL_BLEND);
313 }
314
315 m_mesh->Draw();
316}
