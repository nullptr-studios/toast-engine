70void Engine::Run(int argc, char** argv) {
71 // Before starting the engine we store the arguments 0x
72 // OPTIMIZE: Maybe we can put this on the entrypoint to optimize winmain
73 m_arguments.clear();
74 m_arguments.reserve(argc > 0 ? static_cast<size_t>(argc - 1) : 0);
75 for (int i = 1; i < argc; ++i) {
76 m_arguments.emplace_back(argv[i]);
77 }

79 Init();
80 purge_timer = 0.0f;
81
82 auto* window = m->window.get();
83 auto* world = m->gameWorld.get();
84
85 updateTimer += Time::delta();
86
87#ifdef _WIN32
88 // Request 1ms timer resolution for accurate sleep_for on Windows.
89 // Without this, Sleep/sleep_for is ~15ms, making any FPS cap below 70 wildly inaccurate.
90 timeBeginPeriod(1);
91#endif
92
93 using clock_t = std::chrono::high_resolution_clock;
94 clock_t::time_point frameStart;
95
96 while (!GetShouldClose()) {
97 updateTimer = 0.0;
98 // This is our frame 0x
99 PROFILE_ZONE_N("Frame");
100
101 // Record frame start at the very top so the limiter measures the full frame cost
102 frameStart = clock_t::now();
103
104 // avoid heavy work if minimized
105 if (window->IsMinimized()) {
106 // Still poll events
107 window->PollEventsOnly();
108 m_windowShouldClose.store(window->ShouldClose(), std::memory_order_relaxed);
109 // Back off a bit to avoid busy-waiting while minimized
110 window->WaitEventsTimeout(0.016); // ~60 FPS
111 PROFILE_FRAME;
112 continue;
113 }
114
115 // Poll OS events early in the frame to reduce input latency
116 window->PollEventsOnly();
117
118 m->time->Tick();
119
120 m->resourceManager->LoadResourcesMainThread();
121
122 m->eventSystem->PollEvents();
123
124 // Ensure any pending Begin calls are executed as early as possible in the frame
125 world->RunBeginQueue();
126
127 m->coroutineHandler->Tick();
128 m->inputSystem->Tick();
129
130 world->EarlyTick();
131
132 // Interpolate rb transforms before rendering
133 physics::PhysicsSystem::UpdateVisualInterpolation();
134
135 world->Tick();
136 world->LateTick();
137
138#ifdef TOAST_EDITOR
139 world->EditorTick();
140#endif
141
142 m->layerStack->TickLayers();
143
144 Render();
145
146 // Start the ImGui frame, only for editor
147#ifdef TOAST_EDITOR
148 m->renderer->StartImGuiFrame();
149 EditorTick();
150 m->renderer->EndImGuiFrame();
151#endif
152
153 m->audioSystem->Tick();
154
155 // Swap after all rendering and UI is done
156 window->SwapBuffers();
157
158 // Only when VSync is OFF; when VSync is on the driver already paces us.
159 const auto& cfg = m->renderer->GetRendererConfig();
160 if (!cfg.vSync && cfg.maxFPS > 0) {
161 const auto targetDuration = std::chrono::duration<double>(1.0 / static_cast<double>(cfg.maxFPS));
162 const auto targetTime = frameStart + targetDuration;
163
164 auto now = clock_t::now();
165 while (targetTime - now > std::chrono::milliseconds(2)) {
166 std::this_thread::sleep_for(std::chrono::milliseconds(1));
167 now = clock_t::now();
168 }
169 while (clock_t::now() < targetTime) {
170#ifdef _WIN32
171 _mm_pause();
172#endif
173 }
174 }
175
176 // DestroyQueue also removes scenes 0x
177 world->RunDestroyQueue();
178
179 m_windowShouldClose.store(window->ShouldClose(), std::memory_order_relaxed);
180
181 // Purge unused resources from the cache (every 120 seconds)
182 const double current_uptime = Time::uptime();
183 if (current_uptime - purge_timer >= 120.0) {
184 purge_timer = current_uptime;
185 TOAST_TRACE("Purging unused resources...");
186 resource::PurgeResources();
187 }
188
189 PROFILE_FRAME;
190 }
191 Close();
192}
