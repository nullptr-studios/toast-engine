214void HUDActor::CreateView() {
215 if (m_view) {
216 return;
217 }
218
219 auto* hud = renderer::HUD::HUDLayer::Get();
220 if (!hud) {
221 return;
222 }
223
224 // World HUD depends on HUDLayer's Ultralight update/render loop.
225 hud->Enable();
226
227 ultralight::ViewConfig cfg;
228 cfg.is_accelerated = true;
229 cfg.is_transparent = true;
230 cfg.enable_images = true;
231 cfg.enable_javascript = true;
232 cfg.initial_focus = false;
233
234 m_view = hud->CreateView(m_viewWidth, m_viewHeight, cfg, false);
235 if (!m_view) {
236 return;
237 }
238 ApplyMeshScaleFromViewSize();
239 SyncResolvedTexture();
240
241 hud->SetViewSortOrder(m_view, m_sortOrder);
242 if (!m_url.empty()) {
243 m_view->LoadURL(ultralight::String(m_url.c_str()));
244 }
245
246 if (!m_pendingScripts.empty()) {
247 for (const auto& script : m_pendingScripts) {
248 m_view->EvaluateScript(ultralight::String(script.c_str()));
249 }
250 m_pendingScripts.clear();
251 }
252}

282void HUDActor::ApplyMeshScaleFromViewSize() {
283 if (!m_scaleMeshByViewSize) {
284 return;
285 }
286 m_pixelToUnitScale = glm::clamp(m_pixelToUnitScale, kMinPixelToUnitScale, kMaxPixelToUnitScale);
287
288 // quad.obj spans 2x2 units, so use half extents when mapping view pixels to world units.
289 const float width_units = static_cast<float>(m_viewWidth) * 0.5f * m_pixelToUnitScale;
290 const float height_units = static_cast<float>(m_viewHeight) * 0.5f * m_pixelToUnitScale;
291 transform()->scale(glm::vec3(width_units, height_units, 1.0f));
292}
