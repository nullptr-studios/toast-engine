88void DebugDrawLayer::Flush() const {
89 glBindVertexArray(m_vao);
90 glBindBuffer(GL_ARRAY_BUFFER, m_vbo);
91
92 size_t byte_size = m_vertices.size() * sizeof(DebugVertex);
93
94 static size_t capacity_bytes = 1024 * sizeof(DebugVertex);
95 if (byte_size > capacity_bytes) {
96 capacity_bytes = byte_size;
97 glBufferData(GL_ARRAY_BUFFER, capacity_bytes, nullptr, GL_STREAM_DRAW);
98 }
99 if (byte_size > 0) {
100 glBufferSubData(GL_ARRAY_BUFFER, 0, byte_size, m_vertices.data());
101 }
102
103 m_shader->Use();
104 m_shader->Set("transform", m_renderer->GetViewProjectionMatrix());
105
106 // draw as lines
107 glDrawArrays(GL_LINES, 0, static_cast<GLsizei>(m_vertices.size()));
108
109 // unbind
110 glBindBuffer(GL_ARRAY_BUFFER, 0);
111 glBindVertexArray(0);
112 glUseProgram(0);
113}

148void DebugDrawLayer::DrawRect(const glm::vec3& pos, const glm::vec3& size, float rotation, const glm::vec4& color) {
149 if (!m_enabled) {

154
155 glm::vec3 a = glm::vec3(-half_size.x, -half_size.y, 0.f);
156 glm::vec3 b = glm::vec3(half_size.x, -half_size.y, 0.f);
157 glm::vec3 c = glm::vec3(half_size.x, half_size.y, 0.f);
158 glm::vec3 d = glm::vec3(-half_size.x, half_size.y, .0f);
159 a = glm::vec3(rotatemxt * glm::vec4(a, 1.0f)) + pos;
160 b = glm::vec3(rotatemxt * glm::vec4(b, 1.0f)) + pos;

162 d = glm::vec3(rotatemxt * glm::vec4(d, 1.0f)) + pos;
163
164 DrawLine(a, b, color);
165 DrawLine(b, c, color);
166 DrawLine(c, d, color);

170void DebugDrawLayer::DrawCircle(const glm::vec2& center, float radius, const glm::vec4& color, int segments) {
171 if (!m_enabled) {
172 return;
173 }
174 if (segments <= 0) {
175 segments = glm::clamp<int>(static_cast<int>(radius * 0.15f * 16.0f), 12, 128);
176 }
177
178 double step = glm::two_pi<double>() / static_cast<double>(segments);
179 glm::vec2 prev = center + glm::vec2(radius, 0.0f);
180 for (int i = 1; i <= segments; ++i) {
181 double ang = step * static_cast<double>(i);
182 glm::vec2 cur = center + glm::vec2(cosf(ang) * radius, sinf(ang) * radius);
183 DrawLine(prev, cur, color);
184 prev = cur;
185 }
186}
