22// Pooled bucket sizes
23static constexpr std::array<std::size_t, 8> K_BUCKET_SIZES { 32, 64, 128, 256, 512, 1024, 2048, 4096 };
24inline constexpr std::size_t K_BUCKET_SIZES_SIZE = K_BUCKET_SIZES.size();
25constexpr std::size_t K_LARGE_ALLOCATION_THRESHOLD = K_BUCKET_SIZES.back();
26
27constexpr std::size_t KB = 1024ull;
28constexpr std::size_t MB = KB * 1024ull;
29constexpr std::size_t GB = MB * 1024ull;

31// cap free nodes per bucket to avoid unlimited pooling
32static constexpr std::size_t K_MAX_FREE_PER_BUCKET = KB * 16; // 16K blocks per bucket max free
33
34struct BlockHeader { // stored at block start
35 uint32_t bucketIndex; // bucket or large sentinel
36 std::size_t requestedSize; // user bytes (wider)
37 BlockHeader* next; // freelist linkage when free
38};

41static std::atomic<std::size_t> g_currentBytes { 0 };
42static std::atomic<std::size_t> g_peakBytes { 0 };
43static std::atomic<std::size_t> g_largeBytes { 0 };
44static std::atomic<std::size_t> g_largeCount { 0 };
45static std::array<std::atomic<std::size_t>, K_BUCKET_SIZES_SIZE> g_bucketInUse {}; // active per bucket
46static std::array<std::atomic<BlockHeader*>, K_BUCKET_SIZES_SIZE> g_freeHead {}; // freelist head per bucket (typed)
47static std::array<std::atomic<std::size_t>, K_BUCKET_SIZES_SIZE> g_freeCount {}; // free nodes per bucket
48static std::atomic<std::size_t> g_poolReservedBytes { 0 };

52static int BucketIndexFor(std::size_t total) noexcept {
53 // binary search for first bucket >= total
54 auto it = std::lower_bound(K_BUCKET_SIZES.begin(), K_BUCKET_SIZES.end(), total);
55 if (it == K_BUCKET_SIZES.end()) {
56 return -1;
57 }
58 return static_cast<int>(std::distance(K_BUCKET_SIZES.begin(), it));
59}

61static void PushFreeNode(int idx, BlockHeader* hdr) noexcept {
62 // enforce a max cached nodes per bucket to avoid unbounded growth
63 auto cnt = g_freeCount[idx].load(std::memory_order_relaxed);
64 if (cnt >= K_MAX_FREE_PER_BUCKET) {
65 // pool full: drop the node
66 std::free(hdr);
67 return;
68 }
69 // add into freelist
70 BlockHeader* old = g_freeHead[idx].load(std::memory_order_relaxed);
71 do {
72 hdr->next = old;
73 } while (!g_freeHead[idx].compare_exchange_weak(old, hdr, std::memory_order_release, std::memory_order_relaxed));
74 g_freeCount[idx].fetch_add(1, std::memory_order_relaxed);
75 // account reserved bytes for the free pool
76 g_poolReservedBytes.fetch_add(K_BUCKET_SIZES[idx], std::memory_order_relaxed);
77}
78
79static BlockHeader* PopFreeNode(int idx) noexcept {
80 BlockHeader* old = g_freeHead[idx].load(std::memory_order_acquire);
81 while (old) {
82 BlockHeader* hdr = old;
83 BlockHeader* nxt = hdr->next;
84 if (g_freeHead[idx].compare_exchange_weak(old, nxt, std::memory_order_acq_rel, std::memory_order_acquire)) {
85 g_freeCount[idx].fetch_sub(1, std::memory_order_relaxed);
86 // taken out of free pool, decrement reserved bytes
87 g_poolReservedBytes.fetch_sub(K_BUCKET_SIZES[idx], std::memory_order_relaxed);
88 return hdr;
89 }
90 }
91 return nullptr;
92}

94void* Alloc(std::size_t size) noexcept {
95 if (size == 0) {
96 size = 1;
97 }
98 std::size_t total = size + sizeof(BlockHeader);
99 int idx = BucketIndexFor(total);
100 if (idx >= 0) {
101 BlockHeader* hdr = PopFreeNode(idx);
102 if (!hdr) {
103 void* raw = std::malloc(K_BUCKET_SIZES[idx]);
104 if (!raw) {
105 return nullptr;
106 }
107 hdr = static_cast<BlockHeader*>(raw);
108 hdr->bucketIndex = (uint32_t)idx;
109 hdr->next = nullptr;
110 // NOTE: do NOT count this allocation as poolReserved
111 }
112 hdr->requestedSize = size;
113 g_bucketInUse[idx].fetch_add(1, std::memory_order_relaxed);
114 auto cur = g_currentBytes.fetch_add(size, std::memory_order_relaxed) + size;
115 auto peak = g_peakBytes.load(std::memory_order_relaxed);
116 // stalls thread until we successfully update peak
117 while (cur > peak && !g_peakBytes.compare_exchange_weak(peak, cur, std::memory_order_relaxed)) { }
118 void* user = hdr + 1;
119#ifdef TRACY_ENABLE
120 TracyAlloc(user, size);
121#endif
122 return user;
123 }
124
125 // large allocation
126 void* raw = std::malloc(total);
127 if (!raw) {
128 return nullptr;
129 }
130 auto* hdr = static_cast<BlockHeader*>(raw);
131 hdr->bucketIndex = kLargeSentinel;
132 hdr->requestedSize = size;
133 hdr->next = nullptr;
134 g_largeBytes.fetch_add(size, std::memory_order_relaxed);
135 g_largeCount.fetch_add(1, std::memory_order_relaxed);
136 auto cur = g_currentBytes.fetch_add(size, std::memory_order_relaxed) + size;
137 auto peak = g_peakBytes.load(std::memory_order_relaxed);
138 // stalls thread until we successfully update peak
139 while (cur > peak && !g_peakBytes.compare_exchange_weak(peak, cur, std::memory_order_relaxed)) { }
140 void* user = hdr + 1;
141#ifdef TRACY_ENABLE
142 TracyAlloc(user, size);
143#endif
144 return user;
145}

167void Free(void* ptr) noexcept {
168 if (!ptr) {
169 return;
170 }
171 BlockHeader* hdr = HeaderFromUser(ptr);
172 std::size_t sz = hdr->requestedSize;
173#ifdef TRACY_ENABLE
174 TracyFree(ptr);
175#endif
176 g_currentBytes.fetch_sub(sz, std::memory_order_relaxed);
177 if (hdr->bucketIndex == kLargeSentinel) {
178 g_largeBytes.fetch_sub(sz, std::memory_order_relaxed);
179 g_largeCount.fetch_sub(1, std::memory_order_relaxed);
180 std::free(hdr);
181 return;
182 }
183 if (hdr->bucketIndex < K_BUCKET_SIZES_SIZE) {
184 int idx = (int)hdr->bucketIndex;
185 g_bucketInUse[idx].fetch_sub(1, std::memory_order_relaxed);
186 hdr->requestedSize = 0;
187 PushFreeNode(idx, hdr);
188 } else {
189 std::free(hdr);
190 }
191}

204void TrimPools() noexcept {
205 for (std::size_t i = 0; i < K_BUCKET_SIZES_SIZE; ++i) {
206 BlockHeader* node = g_freeHead[i].exchange(nullptr, std::memory_order_acq_rel);
207 std::size_t released = 0;
208 while (node) {
209 auto* hdr = node;
210 node = hdr->next;
211 released += K_BUCKET_SIZES[i];
212 std::free(hdr);
213 }
214 if (released) {
215 g_freeCount[i].store(0, std::memory_order_relaxed);
216 // adjust reserved bytes by the total released
217 g_poolReservedBytes.fetch_sub(released, std::memory_order_relaxed);
218 }
219 }
