29void Light2D::OnRender(const glm::mat4& premultiplied_matrix) const {
30 auto model = transform()->GetWorldMatrix();
31 auto mvp = premultiplied_matrix * model;
32
33 m_lightShader->Use();
34
35 // famebuffer samplers
36 m_lightShader->SetSampler("gLightAccum", 0);
37 m_lightShader->SetSampler("gNormal", 1);
38
39 m_lightShader->Set("gMVP", mvp);
40 m_lightShader->Set("gLightColor", m_color);
41 m_lightShader->Set("gLightIntensity", m_intensity);
42 m_lightShader->Set("gLightVolumetricIntensity", m_volumetricIntensity);
43
44 m_lightShader->Set("gLightAngle", glm::radians(m_angle));
45
46 m_lightShader->Set("gRadialSoftness", m_radialSoftness);
47 m_lightShader->Set("gAngularSoftness", m_angularSoftness);
48
49 m_lightShader->Set("gNormalMappingEnabled", m_normalMappingEnabled);
50
51 m_lightShader->Set(
52 "gInvScreenSize", glm::vec2(1.0f / static_cast<float>(m_lightBuffer->Width()), 1.0f / static_cast<float>(m_lightBuffer->Height()))
53 );
54
55 // Bind current light accumulation texture
56 glActiveTexture(GL_TEXTURE0 + 0);
57 glBindTexture(GL_TEXTURE_2D, m_lightBuffer->GetColorTexture(0));
58

62
63 m_lightMesh->Draw();

66}
