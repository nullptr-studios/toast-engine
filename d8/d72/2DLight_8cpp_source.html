52void Light2D::OnRender(const glm::mat4& premultiplied_matrix) const {
53 if (!enabled()) {
54 return;
55 }
56
57 // Culling
58 if (!OclussionVolume::isSphereOnPlanes(transform()->worldPosition(), m_radius)) {
59 return;
60 }
61
62 // Disable depth writes for light accumulation
63 GLboolean prevDepthMask = GL_TRUE;
64 glGetBooleanv(GL_DEPTH_WRITEMASK, &prevDepthMask);
65
66 GLint prevBlendSrc = GL_SRC_ALPHA, prevBlendDst = GL_ONE_MINUS_SRC_ALPHA;
67 glGetIntegerv(GL_BLEND_SRC_RGB, &prevBlendSrc);
68 glGetIntegerv(GL_BLEND_DST_RGB, &prevBlendDst);
69
70 glDepthMask(GL_FALSE);
71
72 glEnable(GL_BLEND);
73 glBlendFunc(GL_SRC_ALPHA, GL_ONE);
74
75 glEnable(GL_DEPTH_TEST);
76
77 auto model = transform()->GetWorldMatrix();
78 auto mvp = premultiplied_matrix * model;
79
80 m_lightShader->Use();
81
82 m_lightShader->Set("gMVP", mvp);
83
84 glActiveTexture(GL_TEXTURE0);
85 glBindTexture(GL_TEXTURE_2D, renderer::IRendererBase::GetInstance()->GetShadowMapTexture());
86 m_lightShader->Set("sdfTex", 0);
87 if (auto* geometryFb = renderer::IRendererBase::GetInstance()->GetGeometryFramebuffer()) {
88 m_lightShader->Set("screenSize", glm::ivec2(geometryFb->Width(), geometryFb->Height()));
89 }
90
91 const glm::vec2 lightPosUV = TransformToUV(transform()->worldPosition(), premultiplied_matrix);
92 const glm::vec3 lightEdgeWorld = glm::vec3(model * glm::vec4(1.0f, 0.0f, 0.0f, 1.0f));
93 const glm::vec2 lightEdgeUV = TransformToUV(lightEdgeWorld, premultiplied_matrix);
94 const float lightRadiusUV = std::max(0.0001f, glm::distance(lightPosUV, lightEdgeUV));
95 glm::vec2 lightDirUV = lightEdgeUV - lightPosUV;
96 if (glm::dot(lightDirUV, lightDirUV) < 1e-8f) {
97 lightDirUV = glm::vec2(1.0f, 0.0f);
98 } else {
99 lightDirUV = glm::normalize(lightDirUV);
100 }
101 m_lightShader->Set("lightPosUV", lightPosUV);
102 m_lightShader->Set("lightRadius", lightRadiusUV);
103 m_lightShader->Set("gLightDirection", lightDirUV);
104
105 const auto& renderConfig = renderer::IRendererBase::GetInstance()->GetRendererConfig();
106 m_lightShader->Set("gShadowSteps", static_cast<int>(renderConfig.shadowRaymarchSteps));
107
108 m_lightShader->Set("gLightShadowRadius", m_lightShadowRadius);
109 m_lightShader->Set("doShadows", m_castShadow);
110
111 m_lightShader->Set("gLightColor", m_color);
112 m_lightShader->Set("gLightIntensity", m_intensity);
113 m_lightShader->Set("gLightVolumetricIntensity", m_volumetricIntensity);
114 m_lightShader->Set("gLightAngle", glm::radians(m_angle));
115 m_lightShader->Set("gRadialSoftness", m_radialSoftness);
116 m_lightShader->Set("gAngularSoftness", m_angularSoftness);
117
118 m_lightMesh->Draw();
119
120 glBlendFunc(static_cast<GLenum>(prevBlendSrc), static_cast<GLenum>(prevBlendDst));
121 glDisable(GL_CULL_FACE);
122 glEnable(GL_BLEND);
123 glEnable(GL_DEPTH_TEST);
124 glDepthMask(prevDepthMask ? GL_TRUE : GL_FALSE);
125 glDepthFunc(GL_LEQUAL);
126 glClearDepth(1.0);
127}
