53void ResourceManager::LoadResourcesMainThread() {
54 if (m_uploadResources.empty()) {

58 PROFILE_ZONE;
59 // safe swap under lock to drain the queue
60 std::vector<std::weak_ptr<IResource>> local;

62 {
63 std::lock_guard<std::mutex> lg(m_uploadMtx);
64 local.swap(m_uploadResources);

75void ResourceManager::PurgeResources() {
76 PROFILE_ZONE;
77 std::vector<std::shared_ptr<IResource>> toDestroy;
78 {
79 std::lock_guard lk(m_mtx);
80 for (auto it = m_cachedResources.begin(); it != m_cachedResources.end();) {
81 if (it->second.use_count() == 1) {
82 // move owning shared_ptr out so we can destroy it outside the lock
83 toDestroy.push_back(std::move(it->second));
84 it = m_cachedResources.erase(it);
85 } else {
86 ++it;
87 }
88 }
89 }
90
91 // Now allow resources to be destroyed
92 for (auto& r : toDestroy) {
93 r.reset();
94 }

107bool ResourceManager::OpenFile(const std::string& path, std::vector<uint8_t>& data) const {
108 PROFILE_ZONE;
109 // if using pkg, read from pack file
110 if (m_pkg) {
111 return g_packFile.ReadFile(path, data);
112 }
113
114 std::string p;
115 // else read from filesystem
116 if (path.find("assets/") == std::string::npos) {
117 p = "assets/" + path;
118 } else {
119 p = path;
120 }
121
122 std::ifstream ifs(p, std::ios::binary);
123 if (!ifs) {
124 return false;
125 }
126 ifs.seekg(0, std::ios::end);
127 size_t size = ifs.tellg();
128 ifs.seekg(0, std::ios::beg);
129 data.resize(size);
130 ifs.read(reinterpret_cast<char*>(data.data()), static_cast<std::streamsize>(size));
131 return true;
132}

134bool ResourceManager::SaveFile(const std::string& path, const std::string& content) {
135 TOAST_INFO("Saving File {}", path);
136 std::string pat {};
137 if (path.find("assets/") == std::string::npos) {
138 pat = "assets/" + path;
139 } else {
140 pat = path;
141 }
142
143 namespace fs = std::filesystem;
144
145 // Ensure parent directories exist
146 try {
147 if (const fs::path p(pat); p.has_parent_path()) {
148 fs::create_directories(p.parent_path());
149 }
150 } catch (const fs::filesystem_error&) {
151 // failed to create parent directories
152 return false;
153 }
154
155 // Open file for writing in binary mode
156 std::ofstream ofs(pat, std::ios::binary | std::ios::out | std::ios::trunc);
157 if (!ofs.is_open()) {
158 return false;
159 }
160
161 ofs.write(content.data(), static_cast<std::streamsize>(content.size()));
162 return ofs.good();
163}

165editor::ResourceSlot::Entry ResourceManager::CreateResourceSlotEntry(const std::filesystem::path& path) {
166 if (!g_fileIcon) {
167 g_fileIcon = resource::ResourceManager::GetInstance()->LoadResource<Texture>("editor/icons/genericFile.png");
168 g_jsonIcon = resource::ResourceManager::GetInstance()->LoadResource<Texture>("editor/icons/jsonFile.png");

172 editor::ResourceSlot::Entry e;
173 e.isDirectory = false;
174 // if (ec) rel = de.path().filename();
175 e.relativePath = path;
176 e.name = path.filename().string();
177 e.extension = path.has_extension() ? path.extension().string() : std::string();
178 std::ranges::transform(e.extension, e.extension.begin(), [](unsigned char c) {
179 return static_cast<char>(std::tolower(c));
180 });
181
182 // icon
183 if (e.extension == ".png" || e.extension == ".jpg") {
184 std::string normalized = e.relativePath.string();
185 std::replace(normalized.begin(), normalized.end(), '\\', '/');
186 e.icon = resource::ResourceManager::GetInstance()->LoadResource<Texture>(normalized);
187 } else if (e.extension == ".json") {
188 e.icon = g_jsonIcon;
189 } else if (e.extension == ".obj") {
190 e.icon = g_objIcon;
191 } else {
192 e.icon = g_fileIcon;
193 }
194 if (!e.icon) {
195 // generic
196 e.icon = g_fileIcon;
197 }
198
199 return e;
200}
