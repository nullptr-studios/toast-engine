18GLuint PostProcessManager::ExecuteStack(std::vector<IPostProcess*>* stack, Framebuffer* inputFBO) {
19 Framebuffer* srcFBO = inputFBO;
20 Framebuffer* dstFBO = m_fboA;
21 bool tonemappingExecuted = false;
22
23 for (auto* effect : *stack) {
24 if (!effect || !effect->IsEnabled()) {
25 continue;
26 }
27 if (effect->GetTypeId() == "Tonemaping") {
28 tonemappingExecuted = true;
29 }
30 effect->Execute(srcFBO, dstFBO);
31
32 srcFBO = dstFBO;
33 dstFBO = (dstFBO == m_fboA) ? m_fboB : m_fboA;
34 }
35
36 // Tonemapping is mandatory for HDR->SDR correctness.
37 if (!tonemappingExecuted && m_mandatoryTonemapping) {
38 m_mandatoryTonemapping->Execute(srcFBO, dstFBO);
39 srcFBO = dstFBO;
40 }
41
42 return srcFBO->GetColorTexture(0);
43}

58void PostProcessManager::PostProcessPass(Framebuffer* inputFBO, Framebuffer* outputFBO) {
59 if (!m_fboA || !m_fboB) {
60 TOAST_WARN("PostProcessManager buffers not initialized, falling back to direct blit");
61 inputFBO->BlitTo(outputFBO, GL_COLOR_BUFFER_BIT, GL_LINEAR);
62 ClearOverride();
63 return;
64 }
65
66 GLuint finalTex = 0;
67 if (m_overrideStack) {
68 finalTex = ExecuteStack(m_overrideStack, inputFBO);
69 } else {
70 std::vector<IPostProcess*> temp;
71 temp.reserve(m_globalStack.size());
72 for (auto& e : m_globalStack) {
73 temp.push_back(e.get());
74 }
75 finalTex = ExecuteStack(&temp, inputFBO);
76 }
77
78 // blit the final texture into output FB
79 Framebuffer* srcFBO = nullptr;
80
81 if (finalTex == inputFBO->GetColorTexture(0)) {
82 srcFBO = inputFBO;
83 } else if (finalTex == m_texA) {
84 srcFBO = m_fboA;
85 } else if (finalTex == m_texB) {
86 srcFBO = m_fboB;
87 } else {
88 TOAST_WARN("PostProcessManager: final texture not recognized, falling back to inputFBO blit");
89 inputFBO->BlitTo(outputFBO, GL_COLOR_BUFFER_BIT, GL_LINEAR);
90 ClearOverride();
91 return;
92 }
93
94 glDisable(GL_DEPTH_TEST);
95
96 outputFBO->bind();
97
98 glDrawBuffer(GL_COLOR_ATTACHMENT0);
99
100 glViewport(0, 0, outputFBO->Width(), outputFBO->Height());
101 glScissor(0, 0, outputFBO->Width(), outputFBO->Height());
102
103 // glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
104
105 srcFBO->BlitTo(outputFBO, GL_COLOR_BUFFER_BIT, GL_LINEAR);
106
107 glEnable(GL_DEPTH_TEST);
108
109 Framebuffer::unbind();
110 ClearOverride();
111}

195bool PostProcessManager::LoadStackInto(const json_t& j, std::vector<std::unique_ptr<IPostProcess>>& dst) {
196 if (!j.is_array()) {
197 return false;
198 }
199
200 std::vector<std::unique_ptr<IPostProcess>> parsed;
201 parsed.reserve(j.size());
202
203 for (const auto& item : j) {
204 if (!item.is_object() || !item.contains("type")) {
205 continue;
206 }
207
208 const auto type = item.at("type").get<std::string>();
209 auto pass = renderer::post::CreateByType(type);
210 if (!pass) {
211 TOAST_WARN("PostProcessManager: could not create pass '{}' while loading stack", type);
212 continue;
213 }
214 pass->LoadState(item);
215 parsed.push_back(std::move(pass));
216 }
217
218 dst = std::move(parsed);
219 return true;
220}
