56void AtlasRendererComponent::BuildQuadFromRegion(
57 spine::AtlasRegion* region, const glm::mat4& transform, uint32_t color, std::vector<renderer::SpineVertex>& vertices,
58 std::vector<uint16_t>& indices
59) {
60 if (!region) {
61 return;
62 }
63
64 // Get UV coordinates from the region
65 float u = region->u;
66 float v = region->v;
67 float u2 = region->u2;
68 float v2 = region->v2;
69
70 // Flip V coordinates on Y axis
71 v = 1.0f - v;
72 v2 = 1.0f - v2;
73
74 // Get dimensions
75 float width = static_cast<float>(region->width) / 50.0f; // 1 scale unit is 50 pixls
76 float height = static_cast<float>(region->height) / 50.0f;
77
78 // Current vertex offset for indices
79 uint16_t baseIndex = static_cast<uint16_t>(vertices.size());
80
81 // 4 vertices for the quad
82 std::array<renderer::SpineVertex, 4> quadVerts;
83
84 // Handle rotated regions
85 if (region->degrees == 90) {
86 std::swap(width, height);
87
88 float halfW = width * 0.5f;
89 float halfH = height * 0.5f;
90
91 // Bottom-left vertex
92 quadVerts[0].position = glm::vec3(halfH, -halfW, 0.0f);
93 quadVerts[0].texCoord = glm::vec2(u2, v2);
94 quadVerts[0].colorABGR = color;
95
96 // Bottom-right vertex
97 quadVerts[1].position = glm::vec3(halfH, halfW, 0.0f);
98 quadVerts[1].texCoord = glm::vec2(u, v2);
99 quadVerts[1].colorABGR = color;
100
101 // Top-right vertex
102 quadVerts[2].position = glm::vec3(-halfH, halfW, 0.0f);
103 quadVerts[2].texCoord = glm::vec2(u, v);
104 quadVerts[2].colorABGR = color;
105
106 // Top-left vertex
107 quadVerts[3].position = glm::vec3(-halfH, -halfW, 0.0f);
108 quadVerts[3].texCoord = glm::vec2(u2, v);
109 quadVerts[3].colorABGR = color;
110 } else {
111 // Unrotated region
112 float halfW = width * 0.5f;
113 float halfH = height * 0.5f;
114
115 // Bottom-left
116 quadVerts[0].position = glm::vec3(-halfW, -halfH, 0.0f);
117 quadVerts[0].texCoord = glm::vec2(u, v2);
118 quadVerts[0].colorABGR = color;
119
120 // Bottom-right
121 quadVerts[1].position = glm::vec3(halfW, -halfH, 0.0f);
122 quadVerts[1].texCoord = glm::vec2(u2, v2);
123 quadVerts[1].colorABGR = color;
124
125 // Top-right
126 quadVerts[2].position = glm::vec3(halfW, halfH, 0.0f);
127 quadVerts[2].texCoord = glm::vec2(u2, v);
128 quadVerts[2].colorABGR = color;
129
130 // Top-left
131 quadVerts[3].position = glm::vec3(-halfW, halfH, 0.0f);
132 quadVerts[3].texCoord = glm::vec2(u, v);
133 quadVerts[3].colorABGR = color;
134 }
135
136 // Transform vertices by the sprite's transform
137 for (auto& vert : quadVerts) {
138 glm::vec4 transformedPos = transform * glm::vec4(vert.position, 1.0f);
139 vert.position = glm::vec3(transformedPos);
140 vertices.push_back(vert);
141 }
142
143 // 6 indices for 2 triangles
144 indices.push_back(baseIndex + 0);
145 indices.push_back(baseIndex + 1);
146 indices.push_back(baseIndex + 2);
147 indices.push_back(baseIndex + 0);
148 indices.push_back(baseIndex + 2);
149 indices.push_back(baseIndex + 3);
150}

152void AtlasRendererComponent::RefreshSprites() {
153 m_spriteCache.clear();
154
155 // Collect all AtlasSpriteComponent children of this component
156 for (auto& [id, childPtr] : children.GetAll()) {
157 if (auto* sprite = dynamic_cast<toast::AtlasSpriteComponent*>(childPtr.get())) {
158 // Update the sprite's region if needed
159 if (sprite->GetRegion() == nullptr && !sprite->GetRegionName().empty() && m_atlas) {
160 spine::AtlasRegion* region = FindRegion(sprite->GetRegionName());
161 sprite->SetRegion(region);
162 }
163 m_spriteCache.push_back(sprite);
164 }
165 }
166
167 m_spriteCacheDirty = false;
168}

177std::string AtlasRendererComponent::GenerateSpriteName(const std::string& regionName) {
178 // Count existing sprites with this region name
179 int count = 0;
180 for (const auto& [id, childPtr] : children.GetAll()) {
181 if (auto* sprite = dynamic_cast<toast::AtlasSpriteComponent*>(childPtr.get())) {
182 if (sprite->GetRegionName() == regionName) {
183 count++;
184 }
185 }
186 }
187
188 // Generate name
189 return regionName + "_" + std::to_string(count);

192void AtlasRendererComponent::OnRender(const glm::mat4& precomputed_mat) noexcept {
193 if (!enabled()) {
194 return;
195 }
196
197 if (!m_atlas || !m_atlas->GetAtlasData()) {
198 return;
199 }
200
201 // Refresh sprite cache
202 if (m_spriteCacheDirty) {
203 RefreshSprites();
204 }
205
206 // Early exit if no sprites
207 if (m_spriteCache.empty()) {
208 return;
209 }
210
211 PROFILE_ZONE;
212
213 const glm::mat4 parentWorld = GetWorldMatrix();
214
215 // Clear buffers for this frame
216 m_tempVerts.clear();
217 m_tempIndices.clear();
218
219 // Build geometry for all sprites
220 for (auto* sprite : m_spriteCache) {
221 if (!sprite->enabled() || !sprite->GetRegion()) {
222 continue;
223 }
224
225 // Get sprite's world transform (relative to parent)
226 glm::mat4 spriteTransform = parentWorld * sprite->GetMatrix();
227
228 // Build quad for this sprite
229 BuildQuadFromRegion(sprite->GetRegion(), spriteTransform, sprite->GetColorABGR(), m_tempVerts, m_tempIndices);
230 }
231
232 // Early exit if no visible sprites
233 if (m_tempIndices.empty()) {
234 return;
235 }
236
237 // Update mesh
238 m_dynamicMesh.UpdateDynamicSpine(m_tempVerts.data(), m_tempVerts.size(), m_tempIndices.data(), m_tempIndices.size());
239
240 // Compute bounding box
241 m_dynamicMesh.ComputeSpineBoundingBox(m_tempVerts.data(), m_tempVerts.size());
242
243 // Frustum culling
244 const auto& frustumPlanes = renderer::IRendererBase::GetInstance()->GetFrustumPlanes();
245 if (!OclussionVolume::isTransformedAABBOnPlanes(frustumPlanes, m_dynamicMesh.dynamicBoundingBox(), glm::mat4(1.0f))) {
246 return;
247 }
248
249 const glm::mat4 mvp = precomputed_mat;
250
251 m_shader->Use();
252 m_shader->Set("transform", mvp);
253
254 spine::Vector<spine::AtlasPage*>& pages = m_atlas->GetAtlasData()->getPages();
255 if (pages.size() > 0) {
256 std::shared_ptr<Texture>* tex_ptr = static_cast<std::shared_ptr<Texture>*>(pages[0]->texture);
257 if (tex_ptr) {
258 tex_ptr->get()->Bind(0);
259 }
260 }
261
262 // Draw all sprites in one call
263 m_dynamicMesh.DrawDynamicSpine(m_tempIndices.size());
264}
