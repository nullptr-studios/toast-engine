105 if (!m_skeleton || !m_skeletonData) {
106 return;
107 }
108
109 auto* data = m_skeletonData->GetSkeletonData();
110 if (!data) {

112 }
113
114 // clear all attachments
115 for (int i = 0; i < m_skeleton->getSlots().size(); ++i) {
116 m_skeleton->getSlots()[i]->setAttachment(nullptr);

118
119 // Find the attachment entry
120 for (int s = 0; s < data->getSkins().size(); ++s) {
121 auto* const skin = data->getSkins()[s];
122 auto i = skin->getAttachments();
123 while (i.hasNext()) {

131 slot->setAttachment(entry._attachment);
132 m_skeleton->updateWorldTransform(spine::Physics_None);

136 }
137 }
138 }
139
140 for (int i = 0; i < m_skeleton->getSlots().size(); ++i) {
141 auto* slot = m_skeleton->getSlots()[i];

154void AtlasRendererComponent::OnRender(const glm::mat4& precomputed_mat) noexcept {
155 if (!enabled()) {
156 return;
157 }
158
159 if (!m_skeleton) {
160 return;
161 }
162
163 const glm::mat4 model = GetWorldMatrix();
164
165 // Frustum culling using the dynamic AABB
166 const auto& frustumPlanes = renderer::IRendererBase::GetInstance()->GetFrustumPlanes();
167 if (!OclussionVolume::isTransformedAABBOnPlanes(frustumPlanes, m_dynamicMesh.dynamicBoundingBox(), model)) {
168 return; // Outside frustum, skip rendering
169 }
170
171 PROFILE_ZONE;
172
173 spine::RenderCommand* command = SpineSkeletonRenderer::getRenderer().render(*m_skeleton);
174
175 const float z_step_cmd = 0.01f; // inter-command z offset step
176 const float z_step_vertex = 1e-4f; // per-triangle/vertex z offset step
177 float z_offset = 0.0f;
178
179 const glm::mat4 mvp = precomputed_mat * model;
180
181 m_shader->Use();
182 m_shader->Set("transform", mvp);
183
184 // Reuse temporary buffers
185 m_tempVerts.clear();
186 m_tempIndices.clear();
187
188 // cache last bound texture to avoid redundant binds
189 m_lastBoundTexture = 0;
190
191 auto flush_batch = [&]() {
192 if (m_tempIndices.empty()) {
193 return;
194 }
195 // Update GPU buffer
196 m_dynamicMesh.UpdateDynamicSpine(m_tempVerts.data(), m_tempVerts.size(), m_tempIndices.data(), m_tempIndices.size());
197 // Draw mesh
198 m_dynamicMesh.DrawDynamicSpine(m_tempIndices.size());
199 // clear for next batch
200 m_tempVerts.clear();
201 m_tempIndices.clear();
202 };
203
204 while (command) {
205 if (m_tempVerts.capacity() < static_cast<size_t>(command->numVertices)) {
206 m_tempVerts.reserve(static_cast<size_t>(command->numVertices));
207 }
208 if (m_tempIndices.capacity() < static_cast<size_t>(command->numIndices)) {
209 m_tempIndices.reserve(static_cast<size_t>(command->numIndices));
210 }
211
212 // Determine command texture
213 std::shared_ptr<Texture>* tex_ptr = static_cast<std::shared_ptr<Texture>*>(command->texture);
214 unsigned int tex_id = tex_ptr->get()->id();
215
216 if (tex_id != m_lastBoundTexture && !m_tempIndices.empty()) {
217 flush_batch();
218 }
219
220 // bind texture if needed
221 if (tex_id != m_lastBoundTexture) {
222 if (tex_id != 0) {
223 tex_ptr->get()->Bind(0);
224 }
225 m_lastBoundTexture = tex_id;
226 }
227
228 // append vertices
229 size_t start_vert = m_tempVerts.size();
230 m_tempVerts.resize(start_vert + command->numVertices);
231 for (int i = 0; i < command->numVertices; ++i) {
232 auto& v = m_tempVerts[start_vert + i];
233 v.position = glm::vec3(command->positions[(i * 2) + 0], command->positions[(i * 2) + 1], 0.0f);
234 v.texCoord = glm::vec2(command->uvs[(i * 2) + 0], command->uvs[(i * 2) + 1]);
235 v.colorABGR = command->colors[i];
236 }
237
238 // append indices
239 size_t start_idx = m_tempIndices.size();
240 m_tempIndices.resize(start_idx + command->numIndices);
241 for (int i = 0; i < command->numIndices; ++i) {
242 m_tempIndices[start_idx + i] = static_cast<uint16_t>(command->indices[i] + start_vert);
243 }
244
245 // Per-triangle Z layering
246 for (size_t i = 0; i + 2 < command->numIndices; i += 3) {
247 uint16_t ia = static_cast<uint16_t>(command->indices[i + 0] + start_vert);
248 uint16_t ib = static_cast<uint16_t>(command->indices[i + 1] + start_vert);
249 uint16_t ic = static_cast<uint16_t>(command->indices[i + 2] + start_vert);
250 float z = z_offset;
251 m_tempVerts[ia].position.z = z;
252 m_tempVerts[ib].position.z = z;
253 m_tempVerts[ic].position.z = z;
254 z_offset += z_step_vertex;
255 }
256
257 // Increment Z between commands
258 z_offset += z_step_cmd;
259
260 command = command->next;
261 }
262
263 // flush any remaining geometry
264 flush_batch();
265}

300void AtlasRendererComponent::UpdateMeshBounds() {
301 spine::RenderCommand* command = SpineSkeletonRenderer::getRenderer().render(*m_skeleton);
302 // collect all vertices to compute bounding box for frustum culling
303 {
304 spine::RenderCommand* cmd = command;
305 size_t totalVerts = 0;
306 while (cmd) {
307 totalVerts += cmd->numVertices;
308 cmd = cmd->next;
309 }
310
311 // Build temporary vertex positions for bounding box computation
312 m_tempVerts.reserve(totalVerts);
313 cmd = command;
314 while (cmd) {
315 for (int i = 0; i < cmd->numVertices; ++i) {
316 renderer::SpineVertex v {};
317 v.position = glm::vec3(cmd->positions[(i * 2) + 0], cmd->positions[(i * 2) + 1], 0.0f);
318 v.texCoord = glm::vec2(cmd->uvs[(i * 2) + 0], cmd->uvs[(i * 2) + 1]);
319 v.colorABGR = cmd->colors[i];
320 m_tempVerts.push_back(v);
321 }
322 cmd = cmd->next;
323 }
324
325 // Compute dynamic bounding box
326 m_dynamicMesh.ComputeSpineBoundingBox(m_tempVerts.data(), m_tempVerts.size());
327 }
328
329 // Reset buffers for actual rendering pass
330 m_tempVerts.clear();
331 m_tempIndices.clear();
332}
