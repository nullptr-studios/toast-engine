103void AtlasRendererComponent::SetOnlyAttachmentByName(const std::string& name) const {
104 if (!m_skeleton || !m_skeletonData) {
105 return;
106 }
107
108 auto* data = m_skeletonData->GetSkeletonData();
109 if (!data) {
110 return;
111 }
112
113 // clear all attachments
114 for (int i = 0; i < m_skeleton->getSlots().size(); ++i) {
115 m_skeleton->getSlots()[i]->setAttachment(nullptr);
116 }
117
118 // Find the attachment entry
119 for (int s = 0; s < data->getSkins().size(); ++s) {
120 auto* const skin = data->getSkins()[s];
121 auto i = skin->getAttachments();
122 while (i.hasNext()) {
123 auto& entry = i.next();
124 if (entry._name.buffer() && name == entry._name.buffer()) {
125 int slot_index = entry._slotIndex;
126 spine::Slot* slot = nullptr;
127 if (slot_index >= 0 && slot_index < m_skeleton->getSlots().size()) {
128 slot = m_skeleton->getSlots()[slot_index];
129 }
130 if (slot) {
131 slot->setAttachment(entry._attachment);
132 m_skeleton->updateWorldTransform(spine::Physics_None);
133 return;
134 }
135 }
136 }
137 }
138
139 for (int i = 0; i < m_skeleton->getSlots().size(); ++i) {
140 auto* slot = m_skeleton->getSlots()[i];
141 if (slot->getData().getName().buffer() && name == slot->getData().getName().buffer()) {
142 // attempt to get attachment with same name
143 if (const auto att = m_skeleton->getAttachment(slot->getData().getName().buffer(), name.c_str())) {
144 slot->setAttachment(att);
145 m_skeleton->updateWorldTransform(spine::Physics_None);
146 return;
147 }
148 }

152void AtlasRendererComponent::OnRender(const glm::mat4& precomputed_mat) noexcept {
153 if (!enabled()) {
154 return;
155 }
156
157 PROFILE_ZONE;
158
159 if (!m_skeleton) {
160 return;
161 }
162
163 const spine::RenderCommand* command = SpineSkeletonRenderer::getRenderer().render(*m_skeleton);
164
165 const float z_step = 0.01f; // z offset step
166 float z_offset = 0.0f;
167
168 const glm::mat4 model = GetWorldMatrix();
169 const glm::mat4 mvp = precomputed_mat * model;
170
171 m_shader->Use();
172 m_shader->Set("transform", mvp);
173
174 // Reuse temporary buffers
175 m_tempVerts.clear();
176 m_tempIndices.clear();
177
178 // cache last bound texture to avoid redundant binds
179 m_lastBoundTexture = 0;
180
181 auto flush_batch = [&]() {
182 if (m_tempIndices.empty()) {
183 return;
184 }
185 // Update GPU buffer
186 m_dynamicMesh.UpdateDynamicSpine(m_tempVerts.data(), m_tempVerts.size(), m_tempIndices.data(), m_tempIndices.size());
187 // Draw mesh
188 m_dynamicMesh.DrawDynamicSpine(m_tempIndices.size());
189 // clear for next batch
190 m_tempVerts.clear();
191 m_tempIndices.clear();
192 };
193
194 while (command) {
195 if (m_tempVerts.capacity() < static_cast<size_t>(command->numVertices)) {
196 m_tempVerts.reserve(static_cast<size_t>(command->numVertices));
197 }
198 if (m_tempIndices.capacity() < static_cast<size_t>(command->numIndices)) {
199 m_tempIndices.reserve(static_cast<size_t>(command->numIndices));
200 }
201
202 // Determine command texture
203 const std::shared_ptr<Texture>* tex_ptr = static_cast<std::shared_ptr<Texture>*>(command->texture);
204 const unsigned int tex_id = tex_ptr->get()->id();
205
206 if (tex_id != m_lastBoundTexture && !m_tempIndices.empty()) {
207 flush_batch();
208 }
209
210 // bind texture if needed
211 if (tex_id != m_lastBoundTexture) {
212 if (tex_id != 0) {
213 tex_ptr->get()->Bind(0);
214 }
215 m_lastBoundTexture = tex_id;
216 }
217
218 // append vertices
219 size_t start_vert = m_tempVerts.size();
220 m_tempVerts.resize(start_vert + command->numVertices);
221 for (int i = 0; i < command->numVertices; ++i) {
222 auto& v = m_tempVerts[start_vert + i];
223 v.position = glm::vec3(command->positions[(i * 2) + 0], command->positions[(i * 2) + 1], z_offset);
224 v.texCoord = glm::vec2(command->uvs[(i * 2) + 0], command->uvs[(i * 2) + 1]);
225 v.colorABGR = command->colors[i];
226 }
227
228 // append indices
229 size_t start_idx = m_tempIndices.size();
230 m_tempIndices.resize(start_idx + command->numIndices);
231 for (int i = 0; i < command->numIndices; ++i) {
232 m_tempIndices[start_idx + i] = static_cast<uint16_t>(command->indices[i] + start_vert);
233 }
234
235 // Increment Z
236 z_offset += z_step;
237
238 command = command->next;
239 }
240
241 // flush any remaining geometry
242 flush_batch();
243}
