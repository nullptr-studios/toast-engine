62void AtlasRendererComponent::BuildQuadFromRegion(
63 spine::AtlasRegion* region, const glm::mat4& transform, uint32_t color, std::vector<renderer::SpineVertex>& vertices,
64 std::vector<uint16_t>& indices
65) {
66 if (!region) {
67 return;
68 }
69
70 float width = static_cast<float>(region->width) / 50.0f;
71 float height = static_cast<float>(region->height) / 50.0f;
72
73 float halfW = width * 0.5f;
74 float halfH = height * 0.5f;
75
76 // Current vertex offset for indices
77 uint16_t baseIndex = static_cast<uint16_t>(vertices.size());
78
79 // 4 vertices for the quad
80 std::array<renderer::SpineVertex, 4> quadVerts;
81
82 // Quad positions
83 // Bottom-left, Bottom-right, Top-right, Top-left
84 quadVerts[0].position = glm::vec3(-halfW, -halfH, 0.0f);
85 quadVerts[1].position = glm::vec3(halfW, -halfH, 0.0f);
86 quadVerts[2].position = glm::vec3(halfW, halfH, 0.0f);
87 quadVerts[3].position = glm::vec3(-halfW, halfH, 0.0f);
88
89 // Set colors
90 quadVerts[0].colorABGR = color;
91 quadVerts[1].colorABGR = color;
92 quadVerts[2].colorABGR = color;
93 quadVerts[3].colorABGR = color;
94
95 // Get UV coordinates from the region
96 float u = region->u;
97 float v = region->v;
98 float u2 = region->u2;
99 float v2 = region->v2;
100
101 if (region->degrees == 90) {
102 // Sprite is rotated 90° CW in the atlas
103 // Rotate UVs to compensate
104 quadVerts[0].texCoord = glm::vec2(u2, v2); // BL
105 quadVerts[1].texCoord = glm::vec2(u2, v); // BR
106 quadVerts[2].texCoord = glm::vec2(u, v); // TR
107 quadVerts[3].texCoord = glm::vec2(u, v2); // TL
108 } else {
109 // Unrotated sprite - standard UV mapping
110 quadVerts[0].texCoord = glm::vec2(u, v2); // Bottom-left
111 quadVerts[1].texCoord = glm::vec2(u2, v2); // Bottom-right
112 quadVerts[2].texCoord = glm::vec2(u2, v); // Top-right
113 quadVerts[3].texCoord = glm::vec2(u, v); // Top-left
114 }
115
116 // Transform vertices by the sprite's transform
117 for (auto& vert : quadVerts) {
118 glm::vec4 transformedPos = transform * glm::vec4(vert.position, 1.0f);
119 vert.position = glm::vec3(transformedPos);
120 vertices.push_back(vert);
121 }
122
123 // 6 indices for 2 triangles
124 indices.push_back(baseIndex + 0);
125 indices.push_back(baseIndex + 1);
126 indices.push_back(baseIndex + 2);
127 indices.push_back(baseIndex + 0);
128 indices.push_back(baseIndex + 2);
129 indices.push_back(baseIndex + 3);
130}

132void AtlasRendererComponent::RefreshSprites() {
133 m_spriteCache.clear();
134
135 // Collect all AtlasSpriteComponent children of this component
136 for (auto& [id, childPtr] : children.GetAll()) {
137 if (auto* sprite = dynamic_cast<toast::AtlasSpriteComponent*>(childPtr.get())) {
138 // Update the sprite's region if needed
139 if (sprite->GetRegion() == nullptr && !sprite->GetRegionName().empty() && m_atlas) {
140 spine::AtlasRegion* region = FindRegion(sprite->GetRegionName());
141 sprite->SetRegion(region);
142 }
143 m_spriteCache.push_back(sprite);
144 }
145 }
146
147 m_spriteCacheDirty = false;
148}

157std::string AtlasRendererComponent::GenerateSpriteName(const std::string& regionName) {
158 // Count existing sprites with this region name
159 int count = 0;
160 for (const auto& [id, childPtr] : children.GetAll()) {
161 if (auto* sprite = dynamic_cast<toast::AtlasSpriteComponent*>(childPtr.get())) {
162 if (sprite->GetRegionName() == regionName) {
163 count++;
164 }
165 }
166 }
167
168 // Generate name
169 return regionName + "_" + std::to_string(count);
170}

172void AtlasRendererComponent::OnRender(const glm::mat4& precomputed_mat) noexcept {
173 if (!enabled()) {
174 return;
175 }
176
177 if (!m_atlas || !m_atlas->GetAtlasData()) {
178 return;
179 }
180
181 // Refresh sprite cache
182 if (m_spriteCacheDirty) {
183 RefreshSprites();
184 }
185
186 // Early exit if no sprites
187 if (m_spriteCache.empty()) {
188 return;
189 }
190
191 PROFILE_ZONE;
192
193 const glm::mat4 parentWorld = GetWorldMatrix();
194
195 // Clear buffers for this frame
196 m_tempVerts.clear();
197 m_tempIndices.clear();
198
199 // Build geometry for all sprites
200 for (auto* sprite : m_spriteCache) {
201 if (!sprite->enabled() || !sprite->GetRegion()) {
202 continue;
203 }
204
205 // Get sprite's world transform (relative to parent)
206 glm::mat4 spriteTransform = parentWorld * sprite->GetMatrix();
207
208 // Build quad for this sprite
209 BuildQuadFromRegion(sprite->GetRegion(), spriteTransform, sprite->GetColorABGR(), m_tempVerts, m_tempIndices);
210 }
211
212 // Early exit if no visible sprites
213 if (m_tempIndices.empty()) {
214 return;
215 }
216
217 // Update mesh
218 m_dynamicMesh.UpdateDynamicSpine(m_tempVerts.data(), m_tempVerts.size(), m_tempIndices.data(), m_tempIndices.size());
219
220 // Compute bounding box
221 m_dynamicMesh.ComputeSpineBoundingBox(m_tempVerts.data(), m_tempVerts.size());
222
223 // Frustum culling
224 const auto& frustumPlanes = renderer::IRendererBase::GetInstance()->GetFrustumPlanes();
225 if (!OclussionVolume::isTransformedAABBOnPlanes(frustumPlanes, m_dynamicMesh.dynamicBoundingBox(), glm::mat4(1.0f))) {
226 return;
227 }
228
229 const glm::mat4 mvp = precomputed_mat;
230
231 m_shader->Use();
232 m_shader->Set("transform", mvp);
233
234 spine::Vector<spine::AtlasPage*>& pages = m_atlas->GetAtlasData()->getPages();
235 if (pages.size() > 0) {
236 std::shared_ptr<Texture>* tex_ptr = static_cast<std::shared_ptr<Texture>*>(pages[0]->texture);
237 if (tex_ptr) {
238 tex_ptr->get()->Bind(0);
239 }
240 }
241
242 // Draw all sprites in one call
243 m_dynamicMesh.DrawDynamicSpine(m_tempIndices.size());
244}
