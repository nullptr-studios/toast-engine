65void AtlasRendererComponent::OnRender(renderer::IRenderablePass pass, const glm::mat4& precomputed_mat) noexcept {
66 if (!enabled()) {
67 return;
68 }
69
70 if (!m.atlas || !m.atlas->GetAtlasData()) {
71 return;
72 }
73
74#ifdef TOAST_EDITOR
75 m.spriteCacheDirty = true;
76#endif
77
78 // Refresh sprite cache if dirty
79 if (m.spriteCacheDirty) {
80 RefreshSprites();
81
82 m.tempVerts.clear();
83 m.tempIndices.clear();
84
85 for (auto* sprite : m.spriteCache) {
86 if (!sprite->enabled() || !sprite->GetRegion()) {
87 continue;
88 }
89
90 const glm::mat4 parent_world = GetWorldMatrix();
91
92 glm::mat4 sprite_transform = parent_world * sprite->GetMatrix();
93 BuildQuadFromRegion(sprite->GetRegion(), sprite_transform, sprite->GetColorABGR(), m.tempVerts, m.tempIndices);
94 }
95
96 // Early exit if no visible sprites
97 if (m.tempIndices.empty()) {
98 return;
99 }
100
101 // Update mesh
102 m.dynamicMesh.UpdateDynamicSpine(m.tempVerts.data(), m.tempVerts.size(), m.tempIndices.data(), m.tempIndices.size());
103
104 // Compute bounding box
105 m.dynamicMesh.ComputeSpineBoundingBox(m.tempVerts.data(), m.tempVerts.size());
106 }
107
108 // Early exit if no sprites
109 if (m.spriteCache.empty()) {
110 return;
111 }
112 PROFILE_ZONE;
113
114 // Clear buffers for this frame
115 if (pass == renderer::IRenderablePass::OCCLUSION) {
116 m.isOnScreen = OclussionVolume::isTransformedAABBOnPlanes(m.dynamicMesh.dynamicBoundingBox(), glm::mat4(1.0f));
117 if (!m.isOccluder) {
118 return; // Skip occlusion pass if not an occluder
119 }
120 }
121
122 // Frustum culling
123 if (!m.isOnScreen) {
124 return;
125 }
126
127 const glm::mat4 mvp = precomputed_mat;
128
129 if (pass == renderer::IRenderablePass::GEOMETRY) {
130 m.shader->Use();
131 m.shader->Set("transform", mvp);
132 } else if (pass == renderer::IRenderablePass::OCCLUSION) {
133 m.occlusionShader->Use();
134 m.occlusionShader->Set("gWorld", GetWorldMatrix());
135
136 // set generic transform uniform
137 m.occlusionShader->Set("gMVP", mvp);
138 }
139
140 spine::Vector<spine::AtlasPage*>& pages = m.atlas->GetAtlasData()->getPages();
141 if (pages.size() > 0) {
142 std::shared_ptr<Texture>* tex_ptr = static_cast<std::shared_ptr<Texture>*>(pages[0]->texture);
143 if (tex_ptr) {
144 tex_ptr->get()->Bind(0);
145 }
146 }
147
148 // Draw all sprites in one call
149 if (m.drawToDepth) {
150 glDepthMask(GL_TRUE);
151 } else {
152 glDepthMask(GL_FALSE);
153 }
154
155 if (pass != renderer::IRenderablePass::OCCLUSION) {
156 glEnable(GL_BLEND);
157 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
158 } else {
159 glDisable(GL_BLEND);
160 }
161 m.dynamicMesh.DrawDynamicSpine(m.tempIndices.size());
162}

217void AtlasRendererComponent::BuildQuadFromRegion(
218 spine::AtlasRegion* region, const glm::mat4& transform, uint32_t color, std::vector<renderer::SpineVertex>& vertices,
219 std::vector<uint16_t>& indices
220) {
221 if (!region) {
222 return;
223 }
224
225 float width = static_cast<float>(region->width) / 50.0f;
226 float height = static_cast<float>(region->height) / 50.0f;
227
228 float half_w = width * 0.5f;
229 float half_h = height * 0.5f;
230
231 // Current vertex offset for indices
232 uint16_t base_index = static_cast<uint16_t>(vertices.size());
233
234 // 4 vertices for the quad
235 std::array<renderer::SpineVertex, 4> quad_verts;
236
237 // Quad positions
238 // Bottom-left, Bottom-right, Top-right, Top-left
239 quad_verts[0].position = glm::vec3(-half_w, -half_h, 0.0f);
240 quad_verts[1].position = glm::vec3(half_w, -half_h, 0.0f);
241 quad_verts[2].position = glm::vec3(half_w, half_h, 0.0f);
242 quad_verts[3].position = glm::vec3(-half_w, half_h, 0.0f);
243
244 // Set colors
245 quad_verts[0].colorABGR = color;
246 quad_verts[1].colorABGR = color;
247 quad_verts[2].colorABGR = color;
248 quad_verts[3].colorABGR = color;
249
250 // Get UV coordinates from the region
251 float u = region->u;
252 float v = region->v;
253 float u2 = region->u2;
254 float v2 = region->v2;
255
256 if (region->degrees == 90) {
257 // Sprite is rotated 90° CW in the atlas
258 // Rotate UVs to compensate
259 quad_verts[0].texCoord = glm::vec2(u2, v2); // BL
260 quad_verts[1].texCoord = glm::vec2(u2, v); // BR
261 quad_verts[2].texCoord = glm::vec2(u, v); // TR
262 quad_verts[3].texCoord = glm::vec2(u, v2); // TL
263 } else {
264 // Unrotated sprite - standard UV mapping
265 quad_verts[0].texCoord = glm::vec2(u, v2); // Bottom-left
266 quad_verts[1].texCoord = glm::vec2(u2, v2); // Bottom-right
267 quad_verts[2].texCoord = glm::vec2(u2, v); // Top-right
268 quad_verts[3].texCoord = glm::vec2(u, v); // Top-left
269 }
270
271 // Transform vertices by the sprite's transform
272 for (auto& vert : quad_verts) {
273 glm::vec4 transformed_pos = transform * glm::vec4(vert.position, 1.0f);
274 vert.position = glm::vec3(transformed_pos);
275 vertices.push_back(vert);
276 }
277
278 // 6 indices for 2 triangles
279 indices.push_back(base_index + 0);
280 indices.push_back(base_index + 1);
281 indices.push_back(base_index + 2);
282 indices.push_back(base_index + 0);
283 indices.push_back(base_index + 2);
284 indices.push_back(base_index + 3);
285}

333std::string AtlasRendererComponent::GenerateSpriteName(std::string_view region_name) {
334 // Count existing sprites with this region name
335 int count = 0;
336 for (const auto& [id, childPtr] : children.GetAll()) {
337 if (auto* sprite = dynamic_cast<toast::AtlasSpriteComponent*>(childPtr.get())) {
338 if (sprite->GetRegionName() == region_name) {
339 count++;
340 }
341 }
342 }
343
344 // Generate name
345 return std::format("{}_{}", region_name, count);
