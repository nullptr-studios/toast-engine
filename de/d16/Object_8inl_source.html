46T* Object::Children::_CreateObject(std::optional<unsigned> id) {
47 // Create an uptr and assign it an id
48 std::unique_ptr<Object> obj = std::make_unique<T>();
49 unsigned obj_id = 0;
50 if (id.has_value()) {

55 obj->m_id = obj_id;
