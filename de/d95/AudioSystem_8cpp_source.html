30void audio::AudioSystem::Init() {
31 // Create FMOD Studio system first, then get low-level system from it
32 // Both systems are needed: Studio for events, Core for raw audio playback
33 ERRCHECK(FMOD::Studio::System::create(&m.studio_system));
34 ERRCHECK(m.studio_system->getCoreSystem(&m.low_level_system));

38 // live-update while on debug

40
41 ERRCHECK(m.low_level_system->getMasterChannelGroup(&m.master_group));
42 initialize_reverb();
43}

54auto audio::AudioSystem::CoreSystem::load(Data& audio_data) -> std::expected<void, AudioError> {
55 PROFILE_ZONE;
56 if (!owner->is_loaded(audio_data)) {
57 TOAST_INFO("[AudioSystem] Loading Sound from file {}", audio_data.GetFilePath());
58 FMOD::Sound* sound;
59 ERRCHECK(owner->m.low_level_system->createSound(audio_data.GetFilePath(), audio_data.Is3D() ? FMOD_3D : FMOD_2D, 0, &sound));
60 ERRCHECK(sound->setMode(audio_data.Loop() ? FMOD_LOOP_NORMAL : FMOD_LOOP_OFF));
61 ERRCHECK(sound->set3DMinMaxDistance(0.5f * owner->DISTANCE_FACTOR, 5000.0f * owner->DISTANCE_FACTOR));
62 // Cache sound for later playback to avoid reloading from disk
63 owner->m.sounds.insert({ audio_data.GetUniqueID(), sound });
64 unsigned int msLength = 0;
65 ERRCHECK(owner->m.sounds[audio_data.GetUniqueID()]->getLength(&msLength, FMOD_TIMEUNIT_MS));
66 audio_data.SetLengthMS(msLength);
67 audio_data.SetLoaded(true);
68 return {};
69 }
70
71 TOAST_WARN("[AudioSystem] Sound File was already loaded!");
72 return std::unexpected(AudioError::AlreadyLoaded);
73}

75auto audio::AudioSystem::CoreSystem::play(const Data& audio_data) -> std::expected<void, AudioError> {
76 if (!owner->is_loaded(audio_data)) {
77 TOAST_ERROR("[AudioSystem] Can't play, sound was not loaded yet from {}", audio_data.GetFilePath());
78 return std::unexpected(AudioError::NotLoaded);
79 }
80
81 TOAST_INFO("[AudioSystem] Playing Sound: {}", audio_data.GetUniqueID());
82 FMOD::Channel* channel;
83 // Start paused to configure 3D position and volume before playback begins
84 ERRCHECK(owner->m.low_level_system->playSound(owner->m.sounds[audio_data.GetUniqueID()], nullptr, true /* start paused */, &channel));
85
86 if (audio_data.Is3D()) {
87 owner->set_3d_channel_position(audio_data, channel);
88 }
89
90 channel->setVolume(audio_data.GetVolume());
91
92 // Track looping sounds so they can be stopped/updated later
93 if (audio_data.Loop()) {
94 owner->m.loops_playing.insert({ audio_data.GetUniqueID(), channel });
95 }
96
97 ERRCHECK(channel->setReverbProperties(0, audio_data.GetReverbAmount()));
98
99 // start audio playback
100 ERRCHECK(channel->setPaused(false));
101 return {};
102}

104auto audio::AudioSystem::CoreSystem::stop(const Data& audio_data) -> std::expected<void, AudioError> {
105 if (!is_playing(audio_data)) {
106 TOAST_WARN("[AudioSystem] Can't stop a looping sound that's not playing!");
107 return std::unexpected(AudioError::NotPlaying);
108 }
109
110 TOAST_INFO("[AudioSystem] Stopping sound {}", audio_data.GetUniqueID());
111 ERRCHECK(owner->m.loops_playing[audio_data.GetUniqueID()]->stop());
112 owner->m.loops_playing.erase(audio_data.GetUniqueID());
113 return {};
114}

116auto audio::AudioSystem::CoreSystem::update_volume(Data& audio_data, float new_volume, unsigned int fade_length) -> std::expected<void, AudioError> {
117 if (!is_playing(audio_data)) {
118 TOAST_WARN("[AudioSystem] Can't update sound loop volume! (It isn't playing or might not be loaded)");
119 return std::unexpected(AudioError::NotPlaying);
120 }
121
122 FMOD::Channel* channel = owner->m.loops_playing[audio_data.GetUniqueID()];
123 // Use instant volume change for short fades since FMOD's default is 64 samples anyway
124 if (fade_length <= 64) {
125 ERRCHECK(channel->setVolume(new_volume));
126 } else {
127 bool fade_up = new_volume > audio_data.GetVolume();
128 // get current audio clock time
129 unsigned long long parentclock = 0;
130 ERRCHECK(channel->getDSPClock(nullptr, &parentclock));
131
132 // Fade up needs immediate volume set + fade to 1.0, fade down just fades to target
133 float target_fade_vol = fade_up ? 1.0f : new_volume;
134
135 if (fade_up) {
136 ERRCHECK(channel->setVolume(new_volume));
137 }
138
139 ERRCHECK(channel->addFadePoint(parentclock, audio_data.GetVolume()));
140 ERRCHECK(channel->addFadePoint(parentclock + fade_length, target_fade_vol));
141 }
142
143 audio_data.SetVolume(new_volume);
144 return {};
145}

186auto audio::AudioSystem::load_event(std::string_view name, std::span<const std::pair<std::string_view, float>> params)
187 -> std::expected<void, AudioError> {
188 PROFILE_ZONE;
189 TOAST_INFO("[AudioSystem] Loading FMOD Studio Event {}", name);
190 FMOD::Studio::EventDescription* event_description = nullptr;
191 ERRCHECK(m.studio_system->getEvent(std::string(name).c_str(), &event_description));
192 // Create an instance of the event
193 FMOD::Studio::EventInstance* event_instance = nullptr;
194 ERRCHECK(event_description->createInstance(&event_instance));
195 // Apply initial parameter values to configure the event before playback
196 for (const auto& [param_name, param_value] : params) {
197 TOAST_INFO("[AudioSystem] Setting Event Instance Parameter {} to value: {}", param_name, param_value);
198 // Set the parameter values of the event instance
199 ERRCHECK(event_instance->setParameterByName(std::string(param_name).c_str(), param_value));
200 }
201 m.event_instances.insert({ std::string(name), event_instance });
202 m.event_descriptions.insert({ std::string(name), event_description });
203 return {};
204}
