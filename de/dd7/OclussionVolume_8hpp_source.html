80inline bool OclussionVolume::isAABBOnPlanes(const std::array<glm::vec4, 6>& planes, const renderer::BoundingBox& aabb) {
81 if (!aabb.isValid()) {
82 return true; // Invalid AABB, assume visible to be safe
83 }
84
85 for (int p = 0; p < 6; ++p) {
86 const glm::vec4& pl = planes[p];
87 glm::vec3 normal(pl.x, pl.y, pl.z);
88
89 // Find the positive vertex (P-vertex) - the corner most in the direction of the plane normal
90 glm::vec3 pVertex;
91 pVertex.x = (normal.x >= 0) ? aabb.max.x : aabb.min.x;
92 pVertex.y = (normal.y >= 0) ? aabb.max.y : aabb.min.y;
93 pVertex.z = (normal.z >= 0) ? aabb.max.z : aabb.min.z;
94
95 // If P-vertex is outside the plane, the AABB is completely outside
96 float dist = glm::dot(normal, pVertex) + pl.w;
97 if (dist < 0) {
98 return false;
99 }
100 }
101 return true;
102}

104inline bool OclussionVolume::isTransformedAABBOnPlanes(
105 const std::array<glm::vec4, 6>& planes, const renderer::BoundingBox& localAABB, const glm::mat4& worldTransform
106) {
107 if (!localAABB.isValid()) {
108 return true; // Invalid AABB, assume visible to be safe
109 }
110
111 // Transform AABB to world space by transforming all 8 corners and computing a new AABB
112 // This is a conservative approximation but fast
113 glm::vec3 corners[8] = {
114 glm::vec3(localAABB.min.x, localAABB.min.y, localAABB.min.z), glm::vec3(localAABB.max.x, localAABB.min.y, localAABB.min.z),
115 glm::vec3(localAABB.min.x, localAABB.max.y, localAABB.min.z), glm::vec3(localAABB.max.x, localAABB.max.y, localAABB.min.z),
116 glm::vec3(localAABB.min.x, localAABB.min.y, localAABB.max.z), glm::vec3(localAABB.max.x, localAABB.min.y, localAABB.max.z),
117 glm::vec3(localAABB.min.x, localAABB.max.y, localAABB.max.z), glm::vec3(localAABB.max.x, localAABB.max.y, localAABB.max.z),
118 };
119
120 renderer::BoundingBox worldAABB;
121 for (int i = 0; i < 8; ++i) {
122 glm::vec4 worldCorner = worldTransform * glm::vec4(corners[i], 1.0f);
123 worldAABB.expand(glm::vec3(worldCorner));
124 }
125
126 return isAABBOnPlanes(planes, worldAABB);
