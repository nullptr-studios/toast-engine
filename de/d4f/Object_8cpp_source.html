7void Object::Load(json_t j, const bool force_create) {
8 PROFILE_ZONE;
9
10 // Verify that the type for the object is the same as in the json
11 if (type() != j["type"].get<std::string>()) {
12 TOAST_ERROR("Trying to serialize for object {0} as type {1}. Expected {2}", name(), j["type"].get<std::string>(), type());
13 return;
14 }
15
16 std::string name = j["name"];
17 std::ranges::replace(name, ' ', '_');
18 m_name = name;
19 m_enabled = j["enabled"].get<bool>();
20
21 m_json = j; // Store for soft reloads
22
23 for (auto& json_c : j["children"]) {
24 const auto type = json_c["type"].get<std::string>();
25 std::string c_name = json_c["name"];
26
27 if (force_create) {
28 // Always create a new child without checking if it exists (default)
29 children.Add(type, c_name, json_c);
30 continue;
31 }
32
33 // Check if the child already exists by name
34 if (children.Has(c_name)) {
35 children.Get(c_name)->Load(json_c, false);
36 } else {
37 // If not, we still create a new one
38 children.Add(type, c_name, json_c);
39 }
40 }
41}

100void Object::RefreshBegin(const bool propagate) {
101 if (m_hasRunBegin) {
102 // If it has run begin, we need to re-schedule it

110 World::ScheduleBegin(this);
