26auto ClosestPointOnSegment(const dvec2& point, const Line& segment) -> dvec2 {
27 const dvec2 ab = segment.p2 - segment.p1;
28 const double ab_len_sq = dot(ab, ab);
29 if (ab_len_sq <= kCircleCollisionEpsSmall) {
30 return segment.p1;
31 }
32
33 const double t = clamp(dot(point - segment.p1, ab) / ab_len_sq, 0.0, 1.0);
34 return segment.p1 + ab * t;
35}

37auto CircleVsConvexEdges(const dvec2& center, const double radius, const std::vector<Line>& edges) -> std::optional<Manifold> {
38 if (edges.empty() || radius <= 0.0) {
39 return std::nullopt;
40 }
41
42 double max_separation = -std::numeric_limits<double>::infinity();
43 const Line* reference_edge = nullptr;
44
45 for (const Line& edge : edges) {
46 // Signed distance from center to supporting plane (positive means outside along outward normal)
47 const double separation = dot(center - edge.p1, edge.normal);
48 if (separation > radius) {
49 return std::nullopt;
50 }
51
52 if (separation > max_separation) {
53 max_separation = separation;
54 reference_edge = &edge;
55 }
56 }
57
58 if (reference_edge == nullptr) {
59 return std::nullopt;
60 }
61
62 dvec2 closest_point = ClosestPointOnSegment(center, *reference_edge);
63 dvec2 center_to_closest = center - closest_point;
64 double dist_sq = length2(center_to_closest);
65
66 dvec2 normal = reference_edge->normal;
67 double depth = 0.0;
68
69 // Center outside the polygon support plane of the reference edge: solve as circle vs segment.
70 if (max_separation > 0.0) {
71 if (dist_sq > radius * radius) {
72 return std::nullopt;
73 }
74
75 if (dist_sq > kCircleCollisionEpsSmall) {
76 normal = center_to_closest / std::sqrt(dist_sq);
77 }
78 depth = radius - std::sqrt(std::max(dist_sq, 0.0));
79 } else {
80 // Center is inside (or on boundary): push outward along nearest face normal.
81 depth = radius - max_separation;
82 closest_point = center - normal * max_separation;
83 }
84
85 const double normal_len_sq = length2(normal);
86 if (normal_len_sq <= kCircleCollisionEpsSmall) {
87 return std::nullopt;
88 }
89 if (!isNormalized(normal, 1e-4)) {
90 normal /= std::sqrt(normal_len_sq);
91 }
92
93 return Manifold { .normal = normal, .contact1 = closest_point, .contact2 = closest_point, .contactCount = 1, .depth = depth };
94}

111void RbKinematics(Rigidbody* rb) {
112 PROFILE_ZONE;
113
114 auto& velocity = rb->velocity;
115
116 // Avoid a Rigidbody having 0 mass
117 if (rb->mass < 0.1) {
118 rb->mass = 1;
119 }
120
121 // Sum forces/impulses
122 // Copy and clear under lock
123 std::deque<glm::dvec2> local_forces {};
124 {
125 std::lock_guard lock(rb->forcesMutex);
126 local_forces.swap(rb->forces);
127 }
128
129 glm::dvec2 forces_sum = std::accumulate(local_forces.begin(), local_forces.end(), glm::dvec2(0.0), std::plus());
130
131 // Apply impulses as instantaneous change in velocity
132 if (length2(forces_sum) > 0.0) {
133 velocity += forces_sum / rb->mass; // forces_sum is impulse (F*dt), so dividing by mass yields delta_v
134 }
135
136 // Apply gravity as acceleration over fixed timestep
137 dvec2 physics_system_gravity;
138 if (PhysicsSystem::gravity_type() != GravityType::POINT) {
139 physics_system_gravity = PhysicsSystem::gravity();
140 } else {
141 dvec2 dir = normalize(PhysicsSystem::gravity_point() - rb->GetPosition());
142 physics_system_gravity = dir * PhysicsSystem::gravity_point_scale();
143 }
144
145 const dvec2 gravity = physics_system_gravity * (rb->hasGravity ? dvec2 { rb->gravityScale } : glm::dvec2 { 0.0, 0.0 });
146 velocity += gravity * Time::fixed_delta();
147
148 // Apply drag over fixed timestep
149 const dvec2 damping = exp(rb->hasDrag ? dvec2 { -rb->drag } * Time::fixed_delta() : dvec2 { 0.0 });
150 velocity *= damping;
151
152 if (all(lessThan(abs(velocity), dvec2 { rb->minimumVelocity }))) {
153 velocity = { 0.0, 0.0 };
154 }
155}

170auto RbRbCollision(Rigidbody* rb1, Rigidbody* rb2) -> std::optional<Manifold> {
171 if (rb1->ignorePlayer or rb2->ignorePlayer) {
172 if ((rb1->flags & ColliderFlags::Player) == ColliderFlags::Player) {
173 return std::nullopt;
174 }
175
176 if ((rb2->flags & ColliderFlags::Player) == ColliderFlags::Player) {
177 return std::nullopt;
178 }
179 }
180
181 PROFILE_ZONE;
182 dvec2 pos1 = rb1->GetPosition();
183 dvec2 pos2 = rb2->GetPosition();
184
185 double dist = distance(pos1, pos2);
186 if (dist < 0.1f) {
187 return std::nullopt;
188 }
189
190 double penetration = (rb1->radius + rb2->radius) - dist;
191
192 // if penetration is negative we don't have a collision
193 if (penetration <= 0.0) {
194 return std::nullopt;
195 }
196
197 auto manifold = Manifold { .normal = normalize(pos1 - pos2), .depth = penetration };
198
199 // tangent direction perpendicular to normal
200 dvec2 base_point = pos2 + manifold.normal * (rb2->radius - manifold.depth);
201 manifold.contact1 = base_point;
202 manifold.contact2 = base_point;
203 manifold.contactCount = 1;
204
205 if (rb1->debug.showManifolds) {
206 manifold.Debug();
207 }
208 return manifold;

211void RbRbResolution(Rigidbody* rb1, Rigidbody* rb2, Manifold manifold) {
212#ifdef _DEBUG
213 if (rb1->debug.skipResolution || rb2->debug.skipResolution) {
214 return;
215 }
216#endif
217
218 PROFILE_ZONE;
219 dvec2 velocity1 = rb1->velocity;
220 dvec2 velocity2 = rb2->velocity;
221
222 dvec2 position1 = rb1->GetPosition();
223 dvec2 position2 = rb2->GetPosition();
224
225 // Early out if both bodies have infinite mass
226 if (rb1->mass <= 0.0 && rb2->mass <= 0.0) {
227 return;
228 }
229
230 double inv_mass1 = (rb1->mass > 0.0) ? 1.0 / rb1->mass : 0.0;
231 double inv_mass2 = (rb2->mass > 0.0) ? 1.0 / rb2->mass : 0.0;
232
233 double inv_mass_sum = inv_mass1 + inv_mass2;
234 if (inv_mass_sum <= 0.0) {
235 return;
236 }
237
238 dvec2 normal = manifold.normal;
239 dvec2 contact_tangent = { -normal.y, normal.x };
240
241 // Relative velocity
242 dvec2 relative_velocity = velocity1 - velocity2;
243
244 // Unfold velocity in normal and tangential components
245 double normal_speed = dot(relative_velocity, normal);
246 double tangent_speed = dot(relative_velocity, contact_tangent);
247
248 // Only resolve if bodies are moving towards each other
249 if (normal_speed < 0.0) {
250 // Restitution disabled below threshold to prevent jitter
251 double restitution1 = (std::abs(normal_speed) < rb1->restitutionThreshold) ? 0.0 : rb1->restitution;
252 double restitution2 = (std::abs(normal_speed) < rb2->restitutionThreshold) ? 0.0 : rb2->restitution;
253
254 double restitution = std::min(restitution1, restitution2);
255
256 // Normal impulse
257 // Jn = -(1 + e) * vn / (invMass1 + invMass2)
258 double normal_impulse = -(1.0 + restitution) * normal_speed / inv_mass_sum;
259
260 // Friction impulse
261 double friction = std::sqrt(rb1->friction * rb2->friction);
262 double max_friction_impulse = friction * std::abs(normal_impulse);
263
264 // Tangential impulse needed to cancel tangential speed
265 double tangential_impulse = -tangent_speed / inv_mass_sum;
266 tangential_impulse = clamp(tangential_impulse, -max_friction_impulse, max_friction_impulse);
267
268 // Apply impulses
269 dvec2 impulse = normal_impulse * normal + tangential_impulse * contact_tangent;
270
271 velocity1 += impulse * inv_mass1;
272 velocity2 -= impulse * inv_mass2;
273 }
274
275 // Positional correction
276 {
277 double depth_corr = manifold.depth - PhysicsSystem::pos_slop();
278 double penetration_correction = (depth_corr > 0.0) ? (depth_corr * PhysicsSystem::pos_ptc()) : 0.0;
279 dvec2 correction = (penetration_correction / inv_mass_sum) * normal;
280
281 position1 += correction * inv_mass1;
282 position2 -= correction * inv_mass2;
283 }
284
285 rb1->SetPosition(position1);
286 rb2->SetPosition(position2);
287
288 // Velocity correction
289 auto velocity_correction = [&](Rigidbody* rb, dvec2 v) -> dvec2 {
290 double normal_velocity = dot(v, normal);
291 if (std::abs(normal_velocity) < rb->minimumVelocity.y) {
292 // Kill tiny normal velocity
293 v -= normal_velocity * normal;
294 }
295
296 if (all(lessThan(abs(v), dvec2 { rb->minimumVelocity }))) {
297 v = { 0.0, 0.0 };
298 }
299
300 return v;
301 };
302
303 rb1->velocity = velocity_correction(rb1, velocity1);
304 rb2->velocity = velocity_correction(rb2, velocity2);
305}

332void RbMeshResolution(Rigidbody* rb, ConvexCollider* c, Manifold manifold) {
333#ifdef _DEBUG
334 if (rb->debug.skipResolution) {
335 return;
336 }
337#endif
338
339 PROFILE_ZONE;
340 dvec2 velocity = rb->velocity;
341 dvec2 position = rb->GetPosition();
342
343 // Early out if body is effectively infinite mass
344 if (rb->mass <= 0.0) {
345 return;
346 }
347
348 double inv_mass = 1.0 / rb->mass;
349
350 dvec2 contact_tangent = { manifold.normal.y, -manifold.normal.x };
351 double temp_force = c->throwForce;
352 if (c->forceLeft) {
353 temp_force = -temp_force;
354 }
355
356 if ((c->flags & ColliderFlags::Ramp) == ColliderFlags::Ramp) {
357 rb->AddForce(dvec2(temp_force * contact_tangent.x, temp_force * contact_tangent.y));
358 }
359
360 // unfold velocity in normal and tangencial
361 double normal_speed = dot(velocity, manifold.normal);
362 double tangent_speed = dot(velocity, contact_tangent);
363 // only apply restitution if we're moving faster than the restitution threshold
364 double restitution = (std::abs(normal_speed) < rb->restitutionThreshold) ? 0.0 : rb->restitution;
365
366 // only resolve if we're going towards the object
367 if (normal_speed < 0.0) {
368 // Normal impulse (bounce response)
369 // Jn = -(1 + e) * vn / invMass = -(1 + e) * vn * mass
370 double normal_impulse = -(1.0 + restitution) * normal_speed / inv_mass;
371
372 // Coulomb friction
373 double max_friction_impulse = c->friction * std::abs(normal_impulse);
374
375 // calculate tangential impulse to cancel tangential speed
376 double tangential_impulse = -tangent_speed / inv_mass;
377 tangential_impulse = clamp(tangential_impulse, -max_friction_impulse, max_friction_impulse);
378
379 // Apply impulses to velocity
380 velocity += (normal_impulse * inv_mass) * manifold.normal + (tangential_impulse * inv_mass) * contact_tangent;
381 }
382
383 // positional correction
384 {
385 double depth_corr = manifold.depth - PhysicsSystem::pos_slop();
386 double penetration_correction = (depth_corr > 0.0) ? (depth_corr * PhysicsSystem::pos_ptc()) : 0.0;
387 position += penetration_correction * manifold.normal;
388 }
389 rb->SetPosition(position);
390
391 // velocity correction
392 normal_speed = glm::dot(velocity, manifold.normal); // update normal speed
393 if (normal_speed < 0.0) {
394 velocity -= normal_speed * manifold.normal;
395 }
396
397 if (all(lessThan(abs(velocity), dvec2 { rb->minimumVelocity }))) {
398 velocity = { 0.0, 0.0 };
399 }
400
401 // if anybody reads this, i, Aleksey Gurov, am gay
402 rb->velocity = velocity;
403}

405std::optional<dvec2> RbRayCollision(Line* ray, Rigidbody* rb) {
406 std::optional<dvec2> result = std::nullopt;
407 dvec2 line = ray->p2 - ray->p1;
408 double t = dot(rb->GetPosition() - ray->p1, line) / dot(line, line);
409 t = clamp(t, 0.0, 1.0);
410 dvec2 closest_point = ray->p1 + t * line;
411
412 if (length2(closest_point - rb->GetPosition()) > rb->radius * rb->radius) {
413 return std::nullopt;
414 }
415
416 double distance = std::max(0.0, rb->radius - length(closest_point - rb->GetPosition()));
417
418 dvec2 pt1 = closest_point - ray->tangent * distance;
419 dvec2 pt2 = closest_point + ray->tangent * distance;
420 if (length2(pt2 - ray->p1) > length2(pt1 - ray->p1)) {
421 result = pt1;
422 } else {
423 result = pt2;
424 }
425
426 return result;
427}

429void RbTriggerCollision(Rigidbody* rb1, Trigger* t, std::list<std::function<void()>>& localCallbacks) {
430 PROFILE_ZONE;
431
432 if (!(rb1->flags & t->flags)) {
433 return;
434 }
435
436 auto* parent = rb1->parent();
437
438 // calculate points
439 const auto& tr = t->transform();
440 double left = tr->worldPosition().x - (tr->scale().x / 2);
441 double right = tr->worldPosition().x + (tr->scale().x / 2);
442 double top = tr->worldPosition().y + (tr->scale().y / 2);
443 double bottom = tr->worldPosition().y - (tr->scale().y / 2);
444
445 const auto& pos = rb1->GetPosition();
446 const double scl = rb1->radius;
447 float x_min = pos.x - scl;
448 float y_min = pos.y - scl;
449 float x_max = pos.x + scl;
450 float y_max = pos.y + scl;
451
452 // Actual collision check
453 if (x_min > right || x_max < left) {
454 goto NO_COLLISION; // NOLINT
455 }
456 if (y_min > top || y_max < bottom) {
457 goto NO_COLLISION; // NOLINT
458 }
459
460 // Dont dispatch callback if the rigidbody is already there
461 if (std::ranges::find(t->rigidbodies, rb1) != t->rigidbodies.end()) {
462 return;
463 }
464
465 t->rigidbodies.emplace_back(rb1);
466 localCallbacks.emplace_back([t, parent]() {
467 t->enterCallback(parent);
468 });
469 t->m.color = t->debug.collideColor;
470
471 if (t->debug.log) {
472 TOAST_INFO("{} entered the trigger {}", rb1->parent()->name(), t->name());
473 }
474 return;
475
476NO_COLLISION:
477 if (std::ranges::find(t->rigidbodies, rb1) != t->rigidbodies.end()) {
478 t->rigidbodies.remove(rb1);
479 localCallbacks.emplace_back([t, parent]() {
480 t->exitCallback(parent);
481 });
482 if (t->rigidbodies.empty()) {
483 t->m.color = t->debug.defaultColor;
484 }
485 if (t->debug.log) {
486 TOAST_INFO("{} exited the trigger {}", rb1->parent()->name(), t->name());
487 }
488 }
