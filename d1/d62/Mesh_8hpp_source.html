18struct BoundingBox {
19 glm::vec3 min = glm::vec3(std::numeric_limits<float>::max());
20 glm::vec3 max = glm::vec3(std::numeric_limits<float>::lowest());
21
23 [[nodiscard]]
24 glm::vec3 center() const {
25 return (min + max) * 0.5f;
26 }
27
29 [[nodiscard]]
30 glm::vec3 size() const {
31 return max - min;
32 }
33
35 [[nodiscard]]
36 glm::vec3 halfExtents() const {
37 return size() * 0.5f;
38 }
39
41 [[nodiscard]]
42 float boundingSphereRadius() const {
43 return glm::length(halfExtents());
44 }
45
47 void expand(const glm::vec3& point) {
48 min = glm::min(min, point);
49 max = glm::max(max, point);
50 }
51
53 [[nodiscard]]
54 bool isValid() const {
55 return min.x <= max.x && min.y <= max.y && min.z <= max.z;
56 }
57};
