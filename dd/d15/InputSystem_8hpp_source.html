68 template<typename ActionType, typename CallbackType>
69 void DispatchQueue(std::vector<ActionType*>& queue, CallbackType Listener::M::* map_ptr) {
70 PROFILE_ZONE;
71 for (auto* a : queue) {
72 a->CalculateValue();
73
74 // Call registered callbacks for this action name
75 for (auto* l : m.subscribers) {
76 auto& callbacks_map = l->m.*map_ptr;
77 const auto range = callbacks_map.equal_range(a->name);
78 for (auto it = range.first; it != range.second; ++it) {
79 it->second(a);
80 }
81 }
82
83 // Remove one-frame events (mouse scroll/position/delta) after dispatch
84 std::erase_if(a->m.pressedKeys, [](const auto& v) {
85 return v.first >= MOUSE_DELTA_CODE;
86 });
87
88 // Clear the queued flag so it can be re-queued next frame
89 a->m.queued = false;
90 }
91 queue.clear();
92 }
