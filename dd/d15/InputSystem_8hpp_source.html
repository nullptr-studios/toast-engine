47 void DispatchQueue(std::deque<ActionType*>& queue, CallbackType Listener::M::* map_ptr) {
48 while (!queue.empty()) {
49 ActionType* a = queue.front();
50 const auto& name = a->name;
51
52 a->CalculateValue();
53 std::erase_if(a->m.pressedKeys, [](const auto& v) {
54 return v.first == MOUSE_SCROLL_Y_CODE || v.first == MOUSE_SCROLL_X_CODE || v.first == MOUSE_POSITION_CODE;
55 });
56
57 for (auto* l : m.subscribers) {
58 auto& callbacks_map = l->m.*map_ptr;
59 const auto range = callbacks_map.equal_range(name);
60 for (auto it = range.first; it != range.second; ++it) {
61 it->second(a);
62 }
63 }
64 queue.pop_front();
65 }
