55 template<typename ActionType, typename CallbackType>
56 void DispatchQueue(std::deque<ActionType*>& queue, CallbackType Listener::M::* map_ptr) {
57 while (!queue.empty()) {
58 ActionType* a = queue.front();
59 const auto& name = a->name;
60
61 a->CalculateValue();
62
63 // Call registered callbacks for this action name
64 // Each listener can have multiple callbacks for the same action
65 for (auto* l : m.subscribers) {
66 auto& callbacks_map = l->m.*map_ptr;
67 const auto range = callbacks_map.equal_range(name);
68 for (auto it = range.first; it != range.second; ++it) {
69 it->second(a);
70 }
71 }
72
73 // Remove one-frame events (mouse scroll/position/delta) after dispatch
74 // These inputs should not persist across frames
75 std::erase_if(a->m.pressedKeys, [](const auto& v) {
76 return v.first >= MOUSE_DELTA_CODE;
77 });
78
79 queue.pop_front();
80 }
81 }
