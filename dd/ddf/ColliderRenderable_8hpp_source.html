148 void OnRender(renderer::IRenderablePass pass,const glm::mat4& viewProjection) noexcept override {
149 if (not enabled() || not m.show)
150 return;
151
152 if (not OclussionVolume::isTransformedAABBOnPlanes(m.boundingBox, GetWorldMatrix())) {
153 return;
154 }
155
156 PROFILE_ZONE;
157
158 // compute transform once
159 const glm::mat4 model = GetWorldMatrix();
160 const glm::mat4 mvp = viewProjection * model;
161
162 if (m.material && m.material->GetShader()) {
163 m.material->Use();
164 if (pass == renderer::IRenderablePass::GEOMETRY) {
165 auto shader = m.material->GetShader();
166 // upload world matrix for deferred / lighting passes
167 shader->Set("gWorld", model);
168 // set generic transform uniform
169 shader->Set("gMVP", mvp);
170 }else if (pass == renderer::IRenderablePass::OCCLUSION) {
171 if (!m.isOccluder)
172 return;
173
174 m.occlusionShader->Use();
175 // upload world matrix for deferred / lighting passes
176 m.occlusionShader->Set("gWorld", model);
177 // set generic transform uniform
178 m.occlusionShader->Set("gMVP", mvp);
179 }
180 // draw
181 m.mesh.DrawDynamicSpine(m.indices.size());
182 }
183
184 if (m.showTop && m.topMaterial && m.topMaterial->GetShader()) {
185 m.topMaterial->Use();
186 if (pass == renderer::IRenderablePass::GEOMETRY) {
187 auto shader = m.topMaterial->GetShader();
188 shader->Set("gWorld", model);
189 shader->Set("gMVP", mvp);
190 } else if (pass == renderer::IRenderablePass::OCCLUSION) {
191 m.occlusionShader->Use();
192 m.occlusionShader->Set("gWorld", model);
193 m.occlusionShader->Set("gMVP", mvp);
194 }
195 m.topMesh.DrawDynamicSpine(m.topIndices.size());
196 }
