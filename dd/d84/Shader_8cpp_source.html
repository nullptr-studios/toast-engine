24void Shader::Load() {
25 PROFILE_ZONE;
26 SetResourceState(resource::ResourceState::LOADING);
27 // Parse Json
28 std::istringstream in {};
29 if (!resource::Open(m_path, in)) {
30 TOAST_ERROR("Shader Failed to open shader file: {0}", m_path);
31 SetResourceState(resource::ResourceState::FAILED);
32 LoadErrorShader();
33 return;
34 }
35
36 nlohmann::ordered_json j = nlohmann::ordered_json::parse(in, nullptr, false);
37 if (j.is_discarded()) {
38 TOAST_ERROR("Shader Failed to parse shader JSON file: {0}", m_path);
39 SetResourceState(resource::ResourceState::FAILED);
40 LoadErrorShader();
41 return;
42 }
43
44 if (!j.contains("stageFiles") || !j["stageFiles"].is_array()) {
45 TOAST_ERROR("Shader JSON file missing 'stageFiles' array: {0}", m_path);
46 SetResourceState(resource::ResourceState::FAILED);
47 LoadErrorShader();
48 return;
49 }
50
51 std::vector<std::pair<Stage, std::filesystem::path>> stage_files;
52 for (const auto& item : j["stageFiles"]) {
53 if (!item.contains("stage") || !item.contains("path")) {
54 TOAST_ERROR("Shader JSON 'stageFiles' item missing 'stage' or 'path': {0}", m_path);
55 SetResourceState(resource::ResourceState::FAILED);
56 LoadErrorShader();
57 return;
58 }
59 std::string stage_str = item["stage"].get<std::string>();
60 std::string file_path = item["path"].get<std::string>();
61 Stage stage = Stage::Vertex;
62 if (stage_str == "vertex") {
63 stage = Stage::Vertex;
64 } else if (stage_str == "fragment") {
65 stage = Stage::Fragment;
66 } else if (stage_str == "geometry") {
67 stage = Stage::Geometry;
68 } else if (stage_str == "compute") {
69 stage = Stage::Compute;
70 } else {
71 TOAST_ERROR("Shader JSON unknown stage type: {0} in file {1}", stage_str, m_path);
72 SetResourceState(resource::ResourceState::FAILED);
73 LoadErrorShader();
74 return;
75 }
76 stage_files.emplace_back(stage, std::filesystem::path(file_path));
77 }
78 if (stage_files.empty()) {
79 TOAST_ERROR("Shader JSON file has no valid 'stageFiles': {0}", m_path);
80 SetResourceState(resource::ResourceState::FAILED);
81 LoadErrorShader();
82 return;
83 }
84 CreateFromFiles(stage_files, m_path);
85 SetResourceState(resource::ResourceState::LOADEDCPU);
86}

112void Shader::LoadMainThread() {
113 PROFILE_ZONE;
114 if (GetResourceState() == resource::ResourceState::UPLOADEDGPU) {
115 // already loaded
116 return;
117 }
118 SetResourceState(resource::ResourceState::UPLOADING);
119
120 std::vector<GLuint> compiled;
121 compiled.reserve(m_sourcesToLoad.size());
122 // Compile all stages
123 bool compilation_failed = false;
124 for (auto& kv : m_sourcesToLoad) {
125 GLuint shader = CompileSingleStage(kv.first, kv.second);
126 if (shader == 0) {
127 compilation_failed = true;
128 break;
129 }
130 compiled.push_back(shader);
131 }
132
133 if (compilation_failed) {
134 // Clean up any successfully compiled shaders
135 for (auto sh : compiled) {
136 if (sh != 0) {
137 glDeleteShader(sh);
138 }
139 }
140 TOAST_ERROR("Shader compilation failed, loading error shader");
141 LoadErrorShader();
142 return;
143 }
144
145 LinkProgram(compiled);
146
147 // Check if linking succeeded
148 if (!m_program) {
149 TOAST_ERROR("Shader linking failed, loading error shader");
150 LoadErrorShader();
151 return;
152 }
153
154 SetResourceState(resource::ResourceState::UPLOADEDGPU);
155}

165// Compile single shader stage
166GLuint Shader::CompileSingleStage(Stage stage, std::string_view source) const {
167 PROFILE_ZONE;
168 GLuint s = glCreateShader(static_cast<GLenum>(stage));
169
170 // source shader code
171 const char* src = source.data();
172 GLint len = static_cast<GLint>(source.size());
173
174 glShaderSource(s, 1, &src, &len);
175 glCompileShader(s);
176
177 GLint ok = GL_FALSE;
178 glGetShaderiv(s, GL_COMPILE_STATUS, &ok);
179
180 // error checking
181 if (ok == GL_FALSE) {
182 GLint log_len = 0;
183 glGetShaderiv(s, GL_INFO_LOG_LENGTH, &log_len);
184
185 std::string log(log_len, '\0');
186 glGetShaderInfoLog(s, log_len, nullptr, log.data());
187
188 std::string error_msg = "Shader compile error (" + StageToString(stage) + "):\n" + log;
189 glDeleteShader(s);
190
191 TOAST_ERROR("{0}", error_msg);
192 return 0; // Return 0 to indicate failure
193 }
194 return s;
195}

197void Shader::LinkProgram(const std::vector<GLuint>& shaders) {
198 PROFILE_ZONE;
199
200 // Check if any shader failed to compile (has id 0)
201 for (auto sh : shaders) {
202 if (sh == 0) {
203 TOAST_ERROR("Cannot link program: one or more shaders failed to compile");
204 return; // Don't attempt to link
205 }
206 }
207
208 // Create attach and link all programs
209 GLuint program = glCreateProgram();
210 for (auto sh : shaders) {
211 glAttachShader(program, sh);
212 }
213 glLinkProgram(program);
214
215 GLint ok = GL_FALSE;
216 glGetProgramiv(program, GL_LINK_STATUS, &ok);
217
218 // error checking
219 if (ok == GL_FALSE) {
220 GLint log_len = 0;
221 glGetProgramiv(program, GL_INFO_LOG_LENGTH, &log_len);
222
223 std::string log(log_len, '\0');
224 glGetProgramInfoLog(program, log_len, nullptr, log.data());
225
226 for (auto sh : shaders) {
227 glDeleteShader(sh);
228 }
229 glDeleteProgram(program);
230 TOAST_ERROR("Program link error:\n{0}", log);
231 return; // Don't throw, just return
232 }
233
234 // detach and delete shaders after successful link
235 for (auto sh : shaders) {
236 glDetachShader(program, sh);
237 glDeleteShader(sh);
238 }
239
240 // delete previous program and replace
241 ClearProgram();
242 m_program = program;
243 m_uniformLocationCache.clear();
244}

259void Shader::CreateFromFiles(const std::vector<std::pair<Stage, std::filesystem::path>>& stage_files, std::string_view debug_name) {
260 PROFILE_ZONE;
261 if (stage_files.empty()) {
262 TOAST_ERROR("No shader files provided");
263 LoadErrorShader();
264 return;
265 }
266
267 m_debugName = std::string(debug_name);
268 m_sourceFiles = stage_files; // store for reload
269 std::vector<std::pair<Stage, std::string>> loaded;
270 loaded.reserve(stage_files.size());
271 for (const auto& [stage, path] : stage_files) {
272 std::istringstream in {};
273 if (!resource::Open(path.string(), in)) {
274 TOAST_ERROR("Shader Failed to open shader file: {0}", path.string());
275 LoadErrorShader();
276 return;
277 }
278 loaded.emplace_back(stage, std::move(in.str()));
279 }
280
281 // convert and call createFromSources
282 std::vector<std::pair<Stage, std::string>> tmp = std::move(loaded);
283 CreateFromSources(tmp, m_debugName);
284}

286void Shader::Reload() {
287 if (m_sourceFiles.empty()) {
288 TOAST_WARN("Shader reload not available: Shader not constructed from files!!");
289 }
290
291 std::vector<std::pair<Stage, std::string>> loaded;
292 loaded.reserve(m_sourceFiles.size());
293 for (auto& kv : m_sourceFiles) {
294 std::istringstream in {};
295 if (!resource::Open(kv.second.string(), in)) {
296 TOAST_ERROR("Failed to open shader file: {0}", kv.second.string());
297 return;
298 }
299 loaded.emplace_back(kv.first, std::move(in.str()));
300 }
301
302 CreateFromSources(loaded, m_debugName);
303}

350void Shader::SetGenericAttrib(std::string_view name, const glm::vec4& v) {
351 if (!m_program) {
352 throw ToastException("SetGenericAttrib called on invalid program");
353 }
354 GLint loc = GetAttribLocation(name);
355 if (loc < 0) {
356 // attribute not found in program (optimized out or not declared) -> fallback to conventional location 3
357 loc = 3;
358 }
359 // set generic attrib value for the current VAO (or when no VAO attrib array enabled)
360 glVertexAttrib4f(static_cast<GLuint>(loc), v.r, v.g, v.b, v.a);
361}

437void Shader::LoadErrorShader() {
438 PROFILE_ZONE;
439 TOAST_WARN("Loading error fallback shader for shader: {0}", m_path);
440
441 const char* vertexSource = R"(
442#version 460 core
443
444layout (location = 0) in vec3 aPos;
445layout (location = 1) in vec3 aNormal;
446layout (location = 2) in vec2 aTexCoord;
447layout (location = 3) in vec4 aTangent; // xyz = tangent, w = handedness
448
449uniform mat4 gMVP;
450uniform mat4 gWorld;
451
452void main()
453{
454 gl_Position = gMVP * vec4(aPos, 1.0);
455}
456)";
457
458 const char* fragmentSource = R"(
459#version 460 core
460out vec4 FragColor;
461
462//in vec2 TexCoord;
463
464void main(void)
465{
466 vec2 resolution = vec2(800.0f, 800.f);
467 vec2 TexCoord = gl_FragCoord.st / resolution.xy;
468
469 float x = floor(TexCoord.x * 8.0);
470 float y = floor(TexCoord.y * 8.0);
471 float pattern = mod(x + y, 2.0);
472 vec3 color = mix(vec3(1.0f,0,0), vec3(0.0), pattern);
473 FragColor = vec4(color, 1.0);
474}
475)";
476
477 std::vector<std::pair<Stage, std::string>> errorStages;
478 errorStages.emplace_back(Stage::Vertex, std::string(vertexSource));
479 errorStages.emplace_back(Stage::Fragment, std::string(fragmentSource));
480
481 m_debugName = "ErrorShader";
482 m_sourceFiles.clear(); // Can't reload error shader from files
483
484 CreateFromSources(errorStages, m_debugName);
485 LoadMainThread();
486
487 SetResourceState(resource::ResourceState::UPLOADEDGPU);
488}
