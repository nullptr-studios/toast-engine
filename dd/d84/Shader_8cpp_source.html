24void Shader::Load() {
25 PROFILE_ZONE;
26 SetResourceState(resource::ResourceState::LOADING);
27 // Parse Json
28 std::istringstream in {};
29 if (!resource::Open(m_path, in)) {
30 throw ToastException("Shader Failed to open shader file: " + m_path);
31 }
32
33 nlohmann::ordered_json j = nlohmann::ordered_json::parse(in, nullptr, false);
34 if (j.is_discarded()) {
35 throw ToastException("Shader Failed to parse shader JSON file: " + m_path);
36 }
37
38 if (!j.contains("stageFiles") || !j["stageFiles"].is_array()) {
39 throw ToastException("Shader JSON file missing 'stageFiles' array: " + m_path);
40 }
41
42 std::vector<std::pair<Stage, std::filesystem::path>> stage_files;
43 for (const auto& item : j["stageFiles"]) {
44 if (!item.contains("stage") || !item.contains("path")) {
45 throw ToastException("Shader JSON 'stageFiles' item missing 'stage' or 'path': " + m_path);
46 }
47 std::string stage_str = item["stage"].get<std::string>();
48 std::string file_path = item["path"].get<std::string>();
49 Stage stage = Stage::Vertex;
50 if (stage_str == "vertex") {
51 stage = Stage::Vertex;
52 } else if (stage_str == "fragment") {
53 stage = Stage::Fragment;
54 } else if (stage_str == "geometry") {
55 stage = Stage::Geometry;
56 } else if (stage_str == "compute") {
57 stage = Stage::Compute;
58 } else {
59 throw ToastException("Shader JSON unknown stage type: " + stage_str + " in file " + m_path);
60 }
61 stage_files.emplace_back(stage, std::filesystem::path(file_path));
62 }
63 if (stage_files.empty()) {
64 throw ToastException("Shader JSON file has no valid 'stageFiles': " + m_path);
65 }
66 CreateFromFiles(stage_files, m_path);
67 SetResourceState(resource::ResourceState::LOADEDCPU);
68}

94void Shader::LoadMainThread() {
95 PROFILE_ZONE;
96 if (GetResourceState() == resource::ResourceState::UPLOADEDGPU) {
97 // already loaded
98 return;
99 }
100 SetResourceState(resource::ResourceState::UPLOADING);
101
102 std::vector<GLuint> compiled;
103 compiled.reserve(m_sourcesToLoad.size());
104 // Compile all stages
105 for (auto& kv : m_sourcesToLoad) {
106 compiled.push_back(CompileSingleStage(kv.first, kv.second));
107 }
108
109 LinkProgram(compiled);
110 SetResourceState(resource::ResourceState::UPLOADEDGPU);
111}

121// Compile single shader stage
122GLuint Shader::CompileSingleStage(Stage stage, std::string_view source) const {
123 PROFILE_ZONE;
124 GLuint s = glCreateShader(static_cast<GLenum>(stage));
125
126 // source shader code
127 const char* src = source.data();
128 GLint len = static_cast<GLint>(source.size());
129
130 glShaderSource(s, 1, &src, &len);
131 glCompileShader(s);
132
133 GLint ok = GL_FALSE;
134 glGetShaderiv(s, GL_COMPILE_STATUS, &ok);
135
136 // error checking
137 if (ok == GL_FALSE) {
138 GLint log_len = 0;
139 glGetShaderiv(s, GL_INFO_LOG_LENGTH, &log_len);
140
141 std::string log(log_len, '\0');
142 glGetShaderInfoLog(s, log_len, nullptr, log.data());
143
144 std::string error_msg = "Shader compile error (" + StageToString(stage) + "):\n" + log;
145 glDeleteShader(s);
146
147 TOAST_ERROR("{0}", error_msg);
148 //@TODO fallback to default error shader?
149 }
150 return s;
151}

153void Shader::LinkProgram(const std::vector<GLuint>& shaders) {
154 PROFILE_ZONE;
155 // Create attach and link all programs
156 GLuint program = glCreateProgram();
157 for (auto sh : shaders) {
158 glAttachShader(program, sh);
159 }
160 glLinkProgram(program);
161
162 GLint ok = GL_FALSE;
163 glGetProgramiv(program, GL_LINK_STATUS, &ok);
164
165 // error checking
166 if (ok == GL_FALSE) {
167 GLint log_len = 0;
168 glGetProgramiv(program, GL_INFO_LOG_LENGTH, &log_len);
169
170 std::string log(log_len, '\0');
171 glGetProgramInfoLog(program, log_len, nullptr, log.data());
172
173 for (auto sh : shaders) {
174 glDeleteShader(sh);
175 }
176 glDeleteProgram(program);
177 // TOAST_ERROR("Program link error:\n{0}", log);
178 throw ToastException("Program link error:\n" + log);
179 }
180
181 // detach and delete shaders after successful link
182 for (auto sh : shaders) {
183 glDetachShader(program, sh);
184 glDeleteShader(sh);
185 }
186
187 // delete previous program and replace
188 ClearProgram();
189 m_program = program;
190 m_uniformLocationCache.clear();
191}

204void Shader::CreateFromFiles(const std::vector<std::pair<Stage, std::filesystem::path>>& stage_files, std::string_view debug_name) {
205 PROFILE_ZONE;
206 if (stage_files.empty()) {
207 throw ToastException("No shader files provided");
208 }
209
210 m_debugName = std::string(debug_name);
211 m_sourceFiles = stage_files; // store for reload
212 std::vector<std::pair<Stage, std::string>> loaded;
213 loaded.reserve(stage_files.size());
214 for (const auto& [stage, path] : stage_files) {
215 std::istringstream in {};
216 if (!resource::Open(path.string(), in)) {
217 TOAST_ERROR("Shader Failed to open shader file: {0}", path.string());
218 }
219 loaded.emplace_back(stage, std::move(in.str()));
220 }
221
222 // convert and call createFromSources
223 std::vector<std::pair<Stage, std::string>> tmp = std::move(loaded);
224 CreateFromSources(tmp, m_debugName);
225}

227void Shader::Reload() {
228 if (m_sourceFiles.empty()) {
229 TOAST_WARN("Shader reload not available: Shader not constructed from files!!");
230 }
231
232 std::vector<std::pair<Stage, std::string>> loaded;
233 loaded.reserve(m_sourceFiles.size());
234 for (auto& kv : m_sourceFiles) {
235 std::istringstream in {};
236 if (!resource::Open(kv.second.string(), in)) {
237 TOAST_ERROR("Failed to open shader file: {0}", kv.second.string());
238 return;
239 }
240 loaded.emplace_back(kv.first, std::move(in.str()));
241 }
242
243 CreateFromSources(loaded, m_debugName);
244}

260GLint Shader::QueryUniformLocation(std::string_view name) {
261 auto it = m_uniformLocationCache.find(std::string(name));

291void Shader::SetGenericAttrib(std::string_view name, const glm::vec4& v) {
292 if (!m_program) {
293 throw ToastException("SetGenericAttrib called on invalid program");
294 }
295 GLint loc = GetAttribLocation(name);
296 if (loc < 0) {
297 // attribute not found in program (optimized out or not declared) -> fallback to conventional location 3
298 loc = 3;
299 }
300 // set generic attrib value for the current VAO (or when no VAO attrib array enabled)
301 glVertexAttrib4f(static_cast<GLuint>(loc), v.r, v.g, v.b, v.a);
302}
