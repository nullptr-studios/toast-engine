86// Top-level dispatcher
87inline auto KeycodeFromString(std::string& str) -> std::optional<std::pair<unsigned, Device>> {
88 // To lowercase
89 std::ranges::transform(str, str.begin(), [](unsigned char c) {
90 return std::tolower(c);
91 });
92
93 // Treat explicit controller/mouse prefixes first
94 if (str.rfind("mouse/", 0) == 0) {
95 auto result = MouseKeycodeFromString(str);
96 if (!result.has_value()) {
97 return std::nullopt;
98 }
99
100 int code = result.value();
101 if (code == -1) { // position sentinel from mouse helper
102 return std::make_pair(MOUSE_POSITION_CODE, Device::Mouse);
103 }
104 return std::make_pair(static_cast<unsigned>(code), Device::Mouse);
105 }
106
107 if (str.rfind("controller/button/", 0) == 0) {
108 auto result = ControllerButtonFromString(str);
109 if (!result.has_value()) {
110 return std::nullopt;
111 }
112 return std::make_pair(static_cast<unsigned>(result.value()), Device::ControllerButton);
113 }
114
115 if (str.rfind("controller/axis/", 0) == 0) {
116 auto result = ControllerAxisFromString(str);
117 if (!result.has_value()) {
118 return std::nullopt;
119 }
120 return std::make_pair(static_cast<unsigned>(result.value()), Device::ControllerAxis);
121 }
122
123 if (str.rfind("controller/stick/", 0) == 0) {
124 auto result = ControllerStickFromString(str);
125 if (!result.has_value()) {
126 return std::nullopt;
127 }
128 return std::make_pair(static_cast<unsigned>(result.value()), Device::ControllerStick);
129 }
130
131 // try keyboard parsing for anything else
132 auto result = KeyboardKeycodeFromString(str);
133 if (!result.has_value()) {
134 return std::nullopt;
135 }
136 return std::make_pair(static_cast<unsigned>(result.value()), Device::Keyboard);
137}

140inline auto KeyboardKeycodeFromString(std::string& str) -> std::optional<int> {
141 // trim
142 auto ltrim = [](std::string& s) {
143 s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](unsigned char ch) {
144 return !std::isspace(ch);
145 }));
146 };
147 auto rtrim = [](std::string& s) {
148 s.erase(
149 std::find_if(
150 s.rbegin(),
151 s.rend(),
152 [](unsigned char ch) {
153 return !std::isspace(ch);
154 }
155 ).base(),
156 s.end()
157 );
158 };
159 ltrim(str);
160 rtrim(str);
161 if (str.empty()) {
162 return std::nullopt;
163 }
164
165 // Helper to lowercase
166 auto to_lower = [](std::string s) {
167 std::transform(s.begin(), s.end(), s.begin(), [](unsigned char c) {
168 return std::tolower(c);
169 });
170 return s;
171 };
172
173 // Check for numpad prefix
174 std::string low = to_lower(str);
175 const std::string numpad_prefix = "numpad/";
176 if (low.rfind(numpad_prefix, 0) == 0) { // starts with "numpad/"
177 std::string key = low.substr(numpad_prefix.size());
178 // normalize common variants
179 for (auto& c : key) {
180 if (c == ' ' || c == '-') {
181 c = '_';
182 }
183 }
184
185 // digits
186 if (key.size() == 1 && key[0] >= '0' && key[0] <= '9') {
187 int digit = key[0] - '0';
188 switch (digit) {
189 case 0: return GLFW_KEY_KP_0;
190 case 1: return GLFW_KEY_KP_1;
191 case 2: return GLFW_KEY_KP_2;
192 case 3: return GLFW_KEY_KP_3;
193 case 4: return GLFW_KEY_KP_4;
194 case 5: return GLFW_KEY_KP_5;
195 case 6: return GLFW_KEY_KP_6;
196 case 7: return GLFW_KEY_KP_7;
197 case 8: return GLFW_KEY_KP_8;
198 case 9: return GLFW_KEY_KP_9;
199 }
200 }
201
202 if (key == "decimal" || key == "." || key == "dot") {
203 return GLFW_KEY_KP_DECIMAL;
204 }
205 if (key == "enter" || key == "return") {
206 return GLFW_KEY_KP_ENTER;
207 }
208 if (key == "add" || key == "+" || key == "plus") {
209 return GLFW_KEY_KP_ADD;
210 }
211 if (key == "subtract" || key == "sub" || key == "-" || key == "minus") {
212 return GLFW_KEY_KP_SUBTRACT;
213 }
214 if (key == "multiply" || key == "mul" || key == "*" || key == "asterisk") {
215 return GLFW_KEY_KP_MULTIPLY;
216 }
217 if (key == "divide" || key == "div" || key == "/") {
218 return GLFW_KEY_KP_DIVIDE;
219 }
220 if (key == "equal" || key == "=") {
221 return GLFW_KEY_KP_EQUAL;
222 }
223
224 return std::nullopt;
225 }
226
227 // Single-character shortcuts
228 if (str.size() == 1) {
229 unsigned char c = static_cast<unsigned char>(str[0]);
230 if (std::isalpha(c)) {
231 char lowc = static_cast<char>(std::tolower(c));
232 return GLFW_KEY_A + (lowc - 'a');
233 }
234 if (std::isdigit(c)) {
235 return GLFW_KEY_0 + (c - '0');
236 }
237
238 switch (c) {
239 case ' ': return GLFW_KEY_SPACE;
240 case '[': return GLFW_KEY_LEFT_BRACKET;
241 case ']': return GLFW_KEY_RIGHT_BRACKET;
242 case '\\': return GLFW_KEY_BACKSLASH;
243 case ';': return GLFW_KEY_SEMICOLON;
244 case '\'': return GLFW_KEY_APOSTROPHE;
245 case ',': return GLFW_KEY_COMMA;
246 case '.': return GLFW_KEY_PERIOD;
247 case '/': return GLFW_KEY_SLASH;
248 case '-': return GLFW_KEY_MINUS;
249 case '=': return GLFW_KEY_EQUAL;
250 case '`': return GLFW_KEY_GRAVE_ACCENT;
251 case '<': return GLFW_KEY_COMMA; // '<' is shift+','
252 case '>': return GLFW_KEY_PERIOD; // '>' is shift+'.'
253 case ':': return GLFW_KEY_SEMICOLON; // ':' is shift+';'
254 case '"': return GLFW_KEY_APOSTROPHE; // '"' is shift+'\''
255 case '|': return GLFW_KEY_BACKSLASH; // '|' is shift+'\'
256 case '?': return GLFW_KEY_SLASH; // '?' is shift+'/'
257 case '+': return GLFW_KEY_EQUAL; // '+' is shift+'=' -> map to EQUAL
258 case '_': return GLFW_KEY_MINUS; // '_' shift + '-'
259 case '~': return GLFW_KEY_GRAVE_ACCENT; // '~' shift+``
260 case '!': return GLFW_KEY_1; // '!' shift + '1'
261 case '@': return GLFW_KEY_2;
262 case '#': return GLFW_KEY_3;
263 case '$': return GLFW_KEY_4;
264 case '%': return GLFW_KEY_5;
265 case '^': return GLFW_KEY_6;
266 case '&': return GLFW_KEY_7;
267 case '*': return GLFW_KEY_8;
268 case '(': return GLFW_KEY_9;
269 case ')': return GLFW_KEY_0;
270 default: break;
271 }
272 }

274 // Normalize textual input
275 std::string n = to_lower(str);
276 for (auto& c : n) {
277 if (c == ' ' || c == '-') {
278 c = '_';
279 }
280 }
281
282 // Function keys F1..F25
283 if (n.size() >= 2 && n[0] == 'f') {
284 int idx = 0;

286 idx = std::stoi(n.substr(1));
287 } catch (...) { idx = 0; }
288 if (idx >= 1 && idx <= 25) {
289 return GLFW_KEY_F1 + (idx - 1);
290 }
291 }

376// Mouse parsing. str expected to start with "mouse/"
377inline auto MouseKeycodeFromString(std::string& str) -> std::optional<int> {
378 std::string s = normalize_name(str);
379 const std::string prefix = "mouse/";
380 if (s.rfind(prefix, 0) != 0) {
381 return std::nullopt;
382 }
383 std::string key = s.substr(prefix.size());
384 if (key.empty()) {
385 return std::nullopt;
386 }
387
388 // sentilen parsing
389 if (key == "position" || key == "pos") {
390 return MOUSE_POSITION_CODE;
391 }
392 if (key == "scrollx" || key == "x") {
393 return MOUSE_SCROLL_X_CODE;
394 }
395 if (key == "scrolly" || key == "scroll" || key == "y") {
396 return MOUSE_SCROLL_Y_CODE;
397 }
398 if (key == "raw") {
399 return MOUSE_RAW_CODE;
400 }
401 if (key == "delta") {
402 return MOUSE_DELTA_CODE;
403 }
404
405 // common names
406 if (key == "left") {
407 return GLFW_MOUSE_BUTTON_LEFT;
408 }
409 if (key == "right") {
410 return GLFW_MOUSE_BUTTON_RIGHT;
411 }
412 if (key == "middle" || key == "wheel") {
413 return GLFW_MOUSE_BUTTON_MIDDLE;
414 }
415
416 // "buttonN" or just numeric "0","1", etc.
417 if (key.rfind("button", 0) == 0) {
418 std::string num = key.substr(6); // after "button"
419 if (!num.empty() && std::all_of(num.begin(), num.end(), ::isdigit)) {
420 int idx = std::atoi(num.c_str());
421 if (idx >= 0 && idx <= GLFW_MOUSE_BUTTON_LAST) {
422 return idx;
423 }
424 }
425 return std::nullopt;
426 }
427
428 // numeric raw
429 if (!key.empty() && std::all_of(key.begin(), key.end(), ::isdigit)) {
430 int idx = std::atoi(key.c_str());
431 if (idx >= 0 && idx <= GLFW_MOUSE_BUTTON_LAST) {
432 return idx;
433 }
434 return std::nullopt;
435 }
436
437 // synonyms
438 if (key == "btn_left" || key == "l") {
439 return GLFW_MOUSE_BUTTON_LEFT;
440 }
441 if (key == "btn_right" || key == "r") {
442 return GLFW_MOUSE_BUTTON_RIGHT;
443 }
444 if (key == "btn_middle" || key == "m") {
445 return GLFW_MOUSE_BUTTON_MIDDLE;
446 }

448 return std::nullopt;
