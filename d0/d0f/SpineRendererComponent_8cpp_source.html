244void SpineRendererComponent::OnRender(const glm::mat4& precomputed_mat) noexcept {
245 if (!enabled()) {
246 return;
247 }
248
249 if (!m_skeleton) {
250 return;
251 }
252
253 PROFILE_ZONE;
254
255 spine::RenderCommand* command = SpineSkeletonRenderer::getRenderer().render(*m_skeleton);
256
257 const float z_step_cmd = 0.01f; // inter-command z offset step
258 const float z_step_vertex = 1e-4f; // per-triangle/vertex z offset step
259 float z_offset = 0.0f;
260
261 const glm::mat4 model = GetWorldMatrix();
262 const glm::mat4 mvp = precomputed_mat * model;
263
264 m_shader->Use();
265 m_shader->Set("transform", mvp);
266
267 // Reuse temporary buffers
268 m_tempVerts.clear();
269 m_tempIndices.clear();
270
271 // First pass: collect all vertices to compute bounding box for frustum culling
272 {
273 spine::RenderCommand* cmd = command;
274 size_t totalVerts = 0;
275 while (cmd) {
276 totalVerts += cmd->numVertices;
277 cmd = cmd->next;
278 }
279
280 // Build temporary vertex positions for bounding box computation
281 m_tempVerts.reserve(totalVerts);
282 cmd = command;
283 while (cmd) {
284 for (int i = 0; i < cmd->numVertices; ++i) {
285 renderer::SpineVertex v {};
286 v.position = glm::vec3(cmd->positions[(i * 2) + 0], cmd->positions[(i * 2) + 1], 0.0f);
287 v.texCoord = glm::vec2(cmd->uvs[(i * 2) + 0], cmd->uvs[(i * 2) + 1]);
288 v.colorABGR = cmd->colors[i];
289 m_tempVerts.push_back(v);
290 }
291 cmd = cmd->next;
292 }
293
294 // Compute dynamic bounding box
295 m_dynamicMesh.ComputeSpineBoundingBox(m_tempVerts.data(), m_tempVerts.size());
296
297 // Frustum culling using the dynamic AABB
298 const auto& frustumPlanes = renderer::IRendererBase::GetInstance()->GetFrustumPlanes();
299 if (!OclussionVolume::isTransformedAABBOnPlanes(frustumPlanes, m_dynamicMesh.dynamicBoundingBox(), model)) {
300 return; // Outside frustum, skip rendering
301 }
302 }
303
304 // Reset buffers for actual rendering pass
305 m_tempVerts.clear();
306 m_tempIndices.clear();
307
308 // cache last bound texture to avoid redundant binds
309 m_lastBoundTexture = 0;
310
311 auto flush_batch = [&]() {
312 if (m_tempIndices.empty()) {
313 return;
314 }
315 // Update GPU buffer
316 m_dynamicMesh.UpdateDynamicSpine(m_tempVerts.data(), m_tempVerts.size(), m_tempIndices.data(), m_tempIndices.size());
317 // Draw mesh
318 m_dynamicMesh.DrawDynamicSpine(m_tempIndices.size());
319 // clear for next batch
320 m_tempVerts.clear();
321 m_tempIndices.clear();
322 };
323
324 while (command) {
325 if (m_tempVerts.capacity() < static_cast<size_t>(command->numVertices)) {
326 m_tempVerts.reserve(static_cast<size_t>(command->numVertices));
327 }
328 if (m_tempIndices.capacity() < static_cast<size_t>(command->numIndices)) {
329 m_tempIndices.reserve(static_cast<size_t>(command->numIndices));
330 }
331
332 // Determine command texture
333 std::shared_ptr<Texture>* tex_ptr = static_cast<std::shared_ptr<Texture>*>(command->texture);
334 unsigned int tex_id = tex_ptr->get()->id();
335
336 if (tex_id != m_lastBoundTexture && !m_tempIndices.empty()) {
337 flush_batch();
338 }
339
340 // bind texture if needed
341 if (tex_id != m_lastBoundTexture) {
342 if (tex_id != 0) {
343 tex_ptr->get()->Bind(0);
344 }
345 m_lastBoundTexture = tex_id;
346 }
347
348 // append vertices
349 size_t start_vert = m_tempVerts.size();
350 m_tempVerts.resize(start_vert + command->numVertices);
351 for (int i = 0; i < command->numVertices; ++i) {
352 auto& v = m_tempVerts[start_vert + i];
353 v.position = glm::vec3(command->positions[(i * 2) + 0], command->positions[(i * 2) + 1], 0.0f);
354 v.texCoord = glm::vec2(command->uvs[(i * 2) + 0], command->uvs[(i * 2) + 1]);
355 v.colorABGR = command->colors[i];
356 }
357
358 // append indices
359 size_t start_idx = m_tempIndices.size();
360 m_tempIndices.resize(start_idx + command->numIndices);
361 for (int i = 0; i < command->numIndices; ++i) {
362 m_tempIndices[start_idx + i] = static_cast<uint16_t>(command->indices[i] + start_vert);
363 }
364
365 // Per-triangle Z layering
366 for (size_t i = 0; i + 2 < command->numIndices; i += 3) {
367 uint16_t ia = static_cast<uint16_t>(command->indices[i + 0] + start_vert);
368 uint16_t ib = static_cast<uint16_t>(command->indices[i + 1] + start_vert);
369 uint16_t ic = static_cast<uint16_t>(command->indices[i + 2] + start_vert);
370 float z = z_offset;
371 m_tempVerts[ia].position.z = z;
372 m_tempVerts[ib].position.z = z;
373 m_tempVerts[ic].position.z = z;
374 z_offset += z_step_vertex;
375 }
376
377 // Increment Z between commands
378 z_offset += z_step_cmd;
379
380 command = command->next;
381 }
382
383 // flush any remaining geometry
384 flush_batch();
385}
