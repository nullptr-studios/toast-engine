277void SpineRendererComponent::OnRender(renderer::IRenderablePass pass, const glm::mat4& precomputed_mat) noexcept {
278 if (!enabled()) {
279 return;
280 }
281
282 if (!m.skeleton) {
283 return;
284 }
285
286 if (pass != renderer::IRenderablePass::GEOMETRY && pass != renderer::IRenderablePass::OCCLUSION) {
287 return;
288 }
289
290 // this is really unoptimiced lmao
291 PROFILE_ZONE_C(0xFF0000); // Red for rendering
292
293 spine::RenderCommand* command = SpineSkeletonRenderer::getRenderer().render(*m.skeleton);
294
295 const float z_step_cmd = 0.01f; // inter-command z offset step
296 const float z_step_vertex = 1e-4f; // per-triangle/vertex z offset step
297 float z_offset = 0.0f;
298
299 const glm::mat4 model = GetWorldMatrix();
300 const glm::mat4 mvp = precomputed_mat * model;
301
302 // done just at occlusion step and reused
303 if (pass == renderer::IRenderablePass::OCCLUSION) {
304 // First pass: collect all vertices to compute bounding box for frustum culling
305 {
306 // Compute dynamic bounding box
307 m.dynamicMesh.ComputeSpineBoundingBox(m.tempVerts.data(), m.tempVerts.size());
308
309 // Frustum culling using the dynamic AABB
310 m.onScreen = OclussionVolume::isTransformedAABBOnPlanes(m.dynamicMesh.dynamicBoundingBox(), model);
311 }
312
313 if (!m.isOccluder) {
314 return;
315 }
316 }
317
318 // cache last bound texture to avoid redundant binds
319 m.lastBoundTexture = 0;
320
321 if (pass == renderer::IRenderablePass::GEOMETRY) {
322 m.shader->Use();
323 m.shader->Set("transform", mvp);
324 } else if (pass == renderer::IRenderablePass::OCCLUSION) {
325 if (!m.isOccluder) {
326 return; // Skip occlusion pass if not an occluder
327 }
328
329 m.occlusionShader->Use();
330 m.occlusionShader->Set("gWorld", model);
331
332 // set generic transform uniform
333 m.occlusionShader->Set("gMVP", mvp);
334 }
335
336 if (!m.onScreen) {
337 return;
338 }
339
340 auto flush_batch = [&]() {
341 if (m.tempIndices.empty()) {
342 return;
343 }
344 // Update GPU buffer
345 m.dynamicMesh.UpdateDynamicSpine(m.tempVerts.data(), m.tempVerts.size(), m.tempIndices.data(), m.tempIndices.size());
346 // Draw mesh
347 m.dynamicMesh.DrawDynamicSpine(m.tempIndices.size());
348 };
349
350 m.tempVerts.clear();
351 m.tempIndices.clear();
352
353 while (command) {
354 if (m.tempVerts.capacity() < static_cast<size_t>(command->numVertices)) {
355 m.tempVerts.reserve(static_cast<size_t>(command->numVertices));
356 }
357 if (m.tempIndices.capacity() < static_cast<size_t>(command->numIndices)) {
358 m.tempIndices.reserve(static_cast<size_t>(command->numIndices));
359 }
360
361 // Determine command texture
362 std::shared_ptr<Texture>* tex_ptr = static_cast<std::shared_ptr<Texture>*>(command->texture);
363 unsigned int tex_id = tex_ptr->get()->id();
364
365 if (tex_id != m.lastBoundTexture && !m.tempIndices.empty()) {
366 flush_batch();
367 }
368
369 // bind texture if needed
370 if (tex_id != m.lastBoundTexture) {
371 if (tex_id != 0) {
372 tex_ptr->get()->Bind(0);
373 }
374 m.lastBoundTexture = tex_id;
375 }
376
377 // append vertices
378 size_t start_vert = m.tempVerts.size();
379 m.tempVerts.resize(start_vert + command->numVertices);
380 for (int i = 0; i < command->numVertices; ++i) {
381 auto& v = m.tempVerts[start_vert + i];
382 v.position = glm::vec3(command->positions[(i * 2) + 0], command->positions[(i * 2) + 1], 0.0f);
383 v.texCoord = glm::vec2(command->uvs[(i * 2) + 0], 1.0f - command->uvs[(i * 2) + 1]);
384 v.colorABGR = command->colors[i];
385 }
386
387 // append indices
388 size_t start_idx = m.tempIndices.size();
389 m.tempIndices.resize(start_idx + command->numIndices);
390 for (int i = 0; i < command->numIndices; ++i) {
391 m.tempIndices[start_idx + i] = static_cast<uint16_t>(command->indices[i] + start_vert);
392 }
393
394 // Per-triangle Z layering
395 for (int i = 0; i + 2 < command->numIndices; i += 3) {
396 uint16_t ia = static_cast<uint16_t>(command->indices[i + 0] + start_vert);
397 uint16_t ib = static_cast<uint16_t>(command->indices[i + 1] + start_vert);
398 uint16_t ic = static_cast<uint16_t>(command->indices[i + 2] + start_vert);
399 float z = z_offset;
400 m.tempVerts[ia].position.z = z;
401 m.tempVerts[ib].position.z = z;
402 m.tempVerts[ic].position.z = z;
403 z_offset += z_step_vertex;
404 }
405
406 // Increment Z between commands
407 z_offset += z_step_cmd;
408
409 command = command->next;
410 }
411
412 // flush any remaining geometry
413 if (m.drawToDepth) {
414 glDepthMask(GL_TRUE);
415 } else {
416 glDepthMask(GL_FALSE);
417 }
418
419 if (pass != renderer::IRenderablePass::OCCLUSION) {
420 glEnable(GL_BLEND);
421 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
422 } else {
423 glDisable(GL_BLEND);
424 }
425 flush_batch();
426}
