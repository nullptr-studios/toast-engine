60// Trigger effect modes from DS5 protocol
61constexpr uint8_t DS5_TRIGGER_MODE_OFF = 0x05; // Turn off and reset
62constexpr uint8_t DS5_TRIGGER_MODE_FEEDBACK = 0x21; // Continuous resistance from a point
63constexpr uint8_t DS5_TRIGGER_MODE_WEAPON = 0x25; // Resistance with "click" at end (single shot)
64constexpr uint8_t DS5_TRIGGER_MODE_VIBRATION = 0x26;// Continuous vibration at a frequency
65constexpr uint8_t DS5_TRIGGER_MODE_MACHINE = 0x27; // Oscillating vibration (perfect for recoil bounce)
66
67// Enable bits for trigger effects
68constexpr uint8_t DS5_ENABLE_RIGHT_TRIGGER = 0x04;
69constexpr uint8_t DS5_ENABLE_LEFT_TRIGGER = 0x08;

72auto send_ds5_effect(const DS5EffectPacket& packet) -> bool {
73 auto* gamepad = input::InputSystem::GetFirstConnectedGamepad();
74 if (not gamepad) {
75 return false;
76 }
77
78 // SDL_SendGamepadEffect sends raw HID data to the controller
79 return SDL_SendGamepadEffect(gamepad, &packet, sizeof(packet));
80}

83auto apply_trigger_effect(Trigger trigger, const uint8_t effect[11]) -> void {
84 if (not is_dualsense()) {
85 TOAST_INFO("Haptics: adaptive triggers only work on DualSense controllers");
86 return;
87 }
88
89 DS5EffectPacket packet {};
90
91 if (trigger == Trigger::Left or trigger == Trigger::Both) {
92 packet.ucEnableBits1 |= DS5_ENABLE_LEFT_TRIGGER;
93 std::memcpy(packet.rgucLeftTriggerEffect, effect, 11);
94 }
95 if (trigger == Trigger::Right or trigger == Trigger::Both) {
96 packet.ucEnableBits1 |= DS5_ENABLE_RIGHT_TRIGGER;
97 std::memcpy(packet.rgucRightTriggerEffect, effect, 11);
98 }
99
100 if (not send_ds5_effect(packet)) {
101 TOAST_WARN("Haptics: failed to send DualSense trigger effect");
102 }
103}

183auto TriggerResistance(Trigger trigger, float start_position, float strength) -> void {
184 uint8_t effect[11] = {};
185
186 // Feedback mode
187 // Uses zone bitmasks for precise control
188 const uint8_t zone = static_cast<uint8_t>(start_position * 9.0f); // 0-9 zones
189 const uint8_t force = static_cast<uint8_t>(std::clamp(strength * 7.0f, 0.0f, 7.0f)); // 0-7 force
190
191 // Build zone bitmask and force values
192 uint16_t active_zones = 0;
193 uint32_t force_zones = 0;
194 for (int i = zone; i < 10; ++i) {
195 active_zones |= static_cast<uint16_t>(1 << i);
196 force_zones |= static_cast<uint32_t>(force << (3 * i));
197 }
198
199 effect[0] = DS5_TRIGGER_MODE_FEEDBACK;
200 effect[1] = static_cast<uint8_t>(active_zones & 0xFF);
201 effect[2] = static_cast<uint8_t>((active_zones >> 8) & 0xFF);
202 effect[3] = static_cast<uint8_t>(force_zones & 0xFF);
203 effect[4] = static_cast<uint8_t>((force_zones >> 8) & 0xFF);
204 effect[5] = static_cast<uint8_t>((force_zones >> 16) & 0xFF);
205 effect[6] = static_cast<uint8_t>((force_zones >> 24) & 0xFF);
206
207 apply_trigger_effect(trigger, effect);
208}

210auto TriggerClick(Trigger trigger, float click_position, float strength) -> void {
211 uint8_t effect[11] = {};
212
213 const uint8_t start_zone = static_cast<uint8_t>(std::clamp(click_position * 9.0f, 2.0f, 7.0f));
214 const uint8_t end_zone = static_cast<uint8_t>(std::min(start_zone + 1, 8));
215 const uint8_t force = static_cast<uint8_t>(std::clamp(strength * 7.0f, 0.0f, 7.0f));
216
217 // Weapon mode uses zone bitmask with start and end points
218 const uint16_t zones = static_cast<uint16_t>((1 << start_zone) | (1 << end_zone));
219
220 effect[0] = DS5_TRIGGER_MODE_WEAPON;
221 effect[1] = static_cast<uint8_t>(zones & 0xFF);
222 effect[2] = static_cast<uint8_t>((zones >> 8) & 0xFF);
223 effect[3] = force;
224
225 apply_trigger_effect(trigger, effect);
226}

228auto TriggerWeapon(Trigger trigger, float start_position, float strength, uint32_t pulse_ms) -> void {
229 uint8_t effect[11] = {};
230
231 // Vibration mode
232 const uint8_t zone = static_cast<uint8_t>(std::clamp(start_position * 9.0f, 0.0f, 9.0f));
233 const uint8_t amp = static_cast<uint8_t>(std::clamp(strength * 7.0f, 0.0f, 7.0f));
234
235 // Convert milliseconds to Hz: freq = 1000 / ms
236 // Clamp to valid range (1-255 Hz), and handle edge cases
237 const uint32_t clamped_ms = std::clamp(pulse_ms, 4u, 1000u); // 4ms = 250Hz max, 1000ms = 1Hz min
238 const uint8_t freq_hz = static_cast<uint8_t>(1000u / clamped_ms);
239
240 // Build zone bitmask and amplitude values for all zones from start onwards
241 uint16_t active_zones = 0;
242 uint32_t amp_zones = 0;
243 for (int i = zone; i < 10; ++i) {
244 active_zones |= static_cast<uint16_t>(1 << i);
245 amp_zones |= static_cast<uint32_t>(amp << (3 * i));
246 }
247
248 effect[0] = DS5_TRIGGER_MODE_VIBRATION;
249 effect[1] = static_cast<uint8_t>(active_zones & 0xFF);
250 effect[2] = static_cast<uint8_t>((active_zones >> 8) & 0xFF);
251 effect[3] = static_cast<uint8_t>(amp_zones & 0xFF);
252 effect[4] = static_cast<uint8_t>((amp_zones >> 8) & 0xFF);
253 effect[5] = static_cast<uint8_t>((amp_zones >> 16) & 0xFF);
254 effect[6] = static_cast<uint8_t>((amp_zones >> 24) & 0xFF);
255 effect[9] = freq_hz;
256
257 apply_trigger_effect(trigger, effect);
258}

267auto TriggerClearAll() -> void {
268 // Clear both triggers with a single packet for efficiency
269 if (not is_dualsense()) {
270 return;
271 }
272
273 DS5EffectPacket packet {};
274 packet.ucEnableBits1 = DS5_ENABLE_LEFT_TRIGGER | DS5_ENABLE_RIGHT_TRIGGER;
275 // Set both triggers to OFF mode explicitly
276 packet.rgucLeftTriggerEffect[0] = DS5_TRIGGER_MODE_OFF;
277 packet.rgucRightTriggerEffect[0] = DS5_TRIGGER_MODE_OFF;
278
279 send_ds5_effect(packet);
280}
