130inline void DrawDebugArrow(const glm::vec2& start, const glm::vec2& direction, float length, const glm::vec4& color) {
131 glm::vec2 end = start + direction * length;
132
133 // Draw main line
134 renderer::DebugLine(start, end, color);
135
136 // Draw arrowhead
137 glm::vec2 perpendicular(-direction.y, direction.x);
138 float arrowSize = length * 0.2f;
139 glm::vec2 arrowPoint1 = end - direction * arrowSize + perpendicular * arrowSize * 0.5f;
140 glm::vec2 arrowPoint2 = end - direction * arrowSize - perpendicular * arrowSize * 0.5f;
141
142 renderer::DebugLine(end, arrowPoint1, color);
143 renderer::DebugLine(end, arrowPoint2, color);
144}
