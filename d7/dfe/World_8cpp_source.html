271void World::UnloadScene(const unsigned id) {
272 auto* w = Instance();
273
274 // Check if scene still exists
275 if (!w->m.children.Has(id)) {
276 return; // Already unloaded, nothing to do
277 }
278
279 // Get scene safely
280 auto* scene = w->m.children.Get(id);
281 if (!scene) {
282 return; // Scene doesn't exist or is being destroyed
283 }
284
285 // Disable the scene manually
286 if (scene->enabled()) {
287 scene->_enabled(false);
288 }
289
290 // Remove from tickables immediately so it stops being processed
291 w->m.tickableScenes.erase(id);
292
293 // Just schedule for destruction
294 toast::World::ScheduleDestroy(scene);

453void World::RunBeginQueue() {
454 if (!m.simulateWorld) {
455 return;
456 }
457
458 PROFILE_ZONE;
459
460 // Swap the queue into a local list under lock so other threads can enqueue while we process
461 std::list<Object*> local {};
462 {
463 std::lock_guard lock(m.queueMutex);
464 local.swap(m.beginQueue);
465 }
466
467 for (auto* obj : local) {
468 if (!obj) {
469 continue;
470 }
471
472 // We need to check this so scene begin is not run while the scene is being loaded
473 if (!m.tickableScenes.contains(obj->scene()->id())) {
474 std::lock_guard lock(m.queueMutex);
475 m.beginQueue.push_back(obj);
476 continue;
477 }
478
479 obj->_Begin();
480 // If begin didn't run, reschedule it for later
481 if (!obj->has_run_begin()) {
482 std::lock_guard lock(m.queueMutex);
483 m.beginQueue.push_back(obj);
484 }
485 }
486}

488void World::RunDestroyQueue() {
489 PROFILE_ZONE;
490
491 // Move the destroy queue into a local list under lock and process without holding the lock
492 std::list<Object*> local {};
493 {
494 std::lock_guard lock(m.queueMutex);
495 local.swap(m.destroyQueue);
496 }
497
498 if (local.empty()) {
499 return;
500 }
501
502 for (auto* obj : local) {
503 if (!obj) {
504 continue;
505 }
506
507 // Call _Destroy() - it will handle double-destruction internally
508 obj->_Destroy();
509
510 // Remove the object from its parent's children map
511 if (obj->parent()) {
512 obj->parent()->children.erase(obj->id());
513 } else {
514 // If it's a root-level object (likely a scene), remove from world's children
515 m.children.erase(obj->id());
516 }
517 }
