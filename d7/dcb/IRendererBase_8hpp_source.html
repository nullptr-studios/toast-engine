202 void LoadRenderSettings() {
203 std::string configData;
204 if (!resource::ResourceManager::LoadConfig(".\\config\\Renderer.settings", configData)) {
205 TOAST_WARN("Failed to load renderer settings file... creating a default one!");
206 SaveRenderSettings();
207 ApplyRenderSettings();
208 return;
209 }
210
211 try {
212 auto j = json_t::parse(configData);
213 if (j.contains("resolutionScale")) {
214 m_config.resolutionScale = j["resolutionScale"].get<float>();
215 } else {
216 m_config.resolutionScale = 1.0f;
217 }
218 if (j.contains("lightResolutionScale")) {
219 m_config.lightResolutionScale = j["lightResolutionScale"].get<float>();
220 } else {
221 m_config.lightResolutionScale = .75f;
222 }
223 if (j.contains("vSync")) {
224 m_config.vSync = j["vSync"].get<bool>();
225 } else {
226 m_config.vSync = true;
227 }
228 if (j.contains("fullscreen")) {
229 m_config.currentDisplayMode = j["fullscreen"].get<toast::DisplayMode>();
230 } else {
231 m_config.currentDisplayMode = toast::DisplayMode::WINDOWED;
232 }
233 if (j.contains("resolution")) {
234 m_config.resolution = j["resolution"].get<glm::uvec2>();
235 } else {
236 m_config.resolution = glm::uvec2(1920, 1080);
237 }
238 TOAST_TRACE("SUCCESFULLY LOADED RENDERER SETTINGS!... now applying");
239 ApplyRenderSettings();
240
241 } catch (const std::exception& e) { TOAST_ERROR("Error parsing renderer settings: {0}", e.what()); }
242 }
243
244 void SaveRenderSettings() {
245 json_t j {};
246 j["resolutionScale"] = m_config.resolutionScale;
247 j["lightResolutionScale"] = m_config.lightResolutionScale;
248 j["vSync"] = m_config.vSync;
249 j["fullscreen"] = m_config.currentDisplayMode;
250 j["resolution"] = m_config.resolution;
251
252 if (!resource::ResourceManager::SaveConfig(".\\config\\Renderer.settings", j.dump(1))) {
253 TOAST_ERROR("Failed to save renderer settings file!");
254 } else {
255 // TOAST_TRACE("SUCCESFULLY SAVED RENDERER SETTINGS!");
256 }
257 }
