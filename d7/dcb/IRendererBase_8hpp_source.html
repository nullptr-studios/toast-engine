211 void LoadRenderSettings() {
212 std::string configData;
213 if (!resource::ResourceManager::LoadConfig("Renderer.settings", configData)) {
214 TOAST_WARN("Failed to load renderer settings file... creating a default one!");
215 SaveRenderSettings();
216 ApplyRenderSettings();
217 return;
218 }
219
220 try {
221 auto j = json_t::parse(configData);
222 if (j.contains("resolutionScale")) {
223 m_config.resolutionScale = j["resolutionScale"].get<float>();
224 } else {
225 m_config.resolutionScale = 1.0f;
226 }
227 if (j.contains("lightResolutionScale")) {
228 m_config.lightResolutionScale = j["lightResolutionScale"].get<float>();
229 } else {
230 m_config.lightResolutionScale = .75f;
231 }
232 if (j.contains("vSync")) {
233 m_config.vSync = j["vSync"].get<bool>();
234 } else {
235 m_config.vSync = true;
236 }
237 if (j.contains("fullscreen")) {
238 m_config.currentDisplayMode = j["fullscreen"].get<toast::DisplayMode>();
239 } else {
240 m_config.currentDisplayMode = toast::DisplayMode::WINDOWED;
241 }
242 if (j.contains("resolution")) {
243 m_config.resolution = j["resolution"].get<glm::uvec2>();
244 } else {
245 m_config.resolution = glm::uvec2(1920, 1080);
246 }
247 if (j.contains("MaxFPS")) {
248 m_config.maxFPS = j["MaxFPS"].get<unsigned>();
249 } else {
250 m_config.maxFPS = 300;
251 }
252 TOAST_TRACE("SUCCESFULLY LOADED RENDERER SETTINGS!... now applying");
253 ApplyRenderSettings();
254
255 } catch (const std::exception& e) { TOAST_ERROR("Error parsing renderer settings: {0}", e.what()); }
256 }
257
258 void SaveRenderSettings() {
259 json_t j {};
260 j["resolutionScale"] = m_config.resolutionScale;
261 j["lightResolutionScale"] = m_config.lightResolutionScale;
262 j["vSync"] = m_config.vSync;
263 j["fullscreen"] = m_config.currentDisplayMode;
264 j["resolution"] = m_config.resolution;
265 j["MaxFPS"] = m_config.maxFPS;
266
267 if (!resource::ResourceManager::SaveConfig("Renderer.settings", j.dump(1))) {
268 TOAST_ERROR("Failed to save renderer settings file!");
269 } else {
270 // TOAST_TRACE("SUCCESFULLY SAVED RENDERER SETTINGS!");
271 }
272 }

276 void ToggleFullscreen() {

278 if (window->GetDisplayMode() == toast::DisplayMode::FULLSCREEN) {
279 window->SetDisplayMode(toast::DisplayMode::WINDOWED);
280 m_config.currentDisplayMode = toast::DisplayMode::WINDOWED;
281 } else {
282 window->SetDisplayMode(toast::DisplayMode::FULLSCREEN);
283 m_config.currentDisplayMode = toast::DisplayMode::FULLSCREEN;
284 }
285 SaveRenderSettings();
