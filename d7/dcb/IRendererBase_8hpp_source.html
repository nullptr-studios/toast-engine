228 void LoadRenderSettings() {
229 std::string configData;
230 if (!resource::ResourceManager::LoadConfig("Renderer.settings", configData)) {
231 TOAST_WARN("Renderer.settings not found, using defaults");
232 ApplySafeRenderSettings();
233 return;
234 }
235
236 try {
237 auto j = json_t::parse(configData);
238
239 if (j.contains("resolutionScale")) {
240 m_config.resolutionScale = j["resolutionScale"].get<float>();
241 }
242 if (j.contains("vSync")) {
243 m_config.vSync = j["vSync"].get<bool>();
244 }
245 if (j.contains("fullscreen")) {
246 m_config.currentDisplayMode = j["fullscreen"].get<toast::DisplayMode>();
247 }
248 if (j.contains("resolution")) {
249 m_config.resolution = j["resolution"].get<glm::uvec2>();
250 }
251 if (j.contains("MaxFPS")) {
252 m_config.maxFPS = j["MaxFPS"].get<unsigned>();
253 }
254 if (j.contains("msaaSamples")) {
255 m_config.msaaSamples = std::max(1u, j["msaaSamples"].get<unsigned>());
256 }
257 if (j.contains("anisotropyLevel")) {
258 m_config.anisotropyLevel = std::max(1.0f, j["anisotropyLevel"].get<float>());
259 }
260 if (j.contains("shadowMapResolution")) {
261 m_config.shadowMapResolution = std::clamp(j["shadowMapResolution"].get<unsigned>(), 64u, 8192u);
262 }
263 if (j.contains("shadowRaymarchSteps")) {
264 m_config.shadowRaymarchSteps = std::clamp(j["shadowRaymarchSteps"].get<unsigned>(), 1u, 256u);
265 }
266
267 TOAST_TRACE("Renderer.settings loaded successfully");
268
269 // Apply only the settings that don't resize the OS window.
270 ApplySafeRenderSettings();
271
272 } catch (const std::exception& e) {
273 TOAST_ERROR("Error parsing Renderer.settings: {0} — using defaults", e.what());
274 ApplySafeRenderSettings();
275 }
276 }

299 TOAST_ERROR("Failed to save renderer settings file!");
