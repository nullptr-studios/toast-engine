215 void LoadRenderSettings() {
216 std::string configData;
217 if (!resource::ResourceManager::LoadConfig("Renderer.settings", configData)) {
218 TOAST_WARN("Renderer.settings not found, using defaults");
219 ApplySafeRenderSettings();
220 return;
221 }
222
223 try {
224 auto j = json_t::parse(configData);
225
226 if (j.contains("resolutionScale")) {
227 m_config.resolutionScale = j["resolutionScale"].get<float>();
228 }
229 if (j.contains("vSync")) {
230 m_config.vSync = j["vSync"].get<bool>();
231 }
232 if (j.contains("fullscreen")) {
233 m_config.currentDisplayMode = j["fullscreen"].get<toast::DisplayMode>();
234 }
235 if (j.contains("resolution")) {
236 m_config.resolution = j["resolution"].get<glm::uvec2>();
237 }
238 if (j.contains("MaxFPS")) {
239 m_config.maxFPS = j["MaxFPS"].get<unsigned>();
240 }
241 if (j.contains("msaaSamples")) {
242 m_config.msaaSamples = std::max(1u, j["msaaSamples"].get<unsigned>());
243 }
244 if (j.contains("anisotropyLevel")) {
245 m_config.anisotropyLevel = std::max(1.0f, j["anisotropyLevel"].get<float>());
246 }
247 if (j.contains("shadowMapResolution")) {
248 m_config.shadowMapResolution = std::clamp(j["shadowMapResolution"].get<unsigned>(), 64u, 8192u);
249 }
250 if (j.contains("shadowRaymarchSteps")) {
251 m_config.shadowRaymarchSteps = std::clamp(j["shadowRaymarchSteps"].get<unsigned>(), 1u, 256u);
252 }
253
254 TOAST_TRACE("Renderer.settings loaded successfully");
255
256 // Apply only the settings that don't resize the OS window.
257 ApplySafeRenderSettings();
258
259 } catch (const std::exception& e) {
260 TOAST_ERROR("Error parsing Renderer.settings: {0} — using defaults", e.what());
261 ApplySafeRenderSettings();
262 }
263 }

267 auto* window = toast::Window::GetInstance();
268 window->SetVSync(m_config.vSync);
269 window->SetRefreshFrameTime(1000.0 / m_config.maxFPS);

285 TOAST_ERROR("Failed to save renderer settings file!");
