116 bool ReadFile(const std::string& raw_path, std::vector<uint8_t>& out) {
117 // canonicalize lookup path same as packer
118 std::u8string path = canonical_path_for_pack(raw_path);
119
120 uint64_t h = fnv1a_hash64(path);
121 auto it = std::lower_bound(hashes_.begin(), hashes_.end(), h);
122 // not found quickly
123 if (it == hashes_.end() || *it != h) {
124 // fallback: linear scan to help debugging/compatibility
125 for (size_t i = 0; i < paths_.size(); ++i) {
126 if (paths_[i] == path) { // found by exact string despite hash mismatch
127 return read_at_index(i, out);
128 }
129 }
130 return false;
131 }
132
133 size_t idx = static_cast<size_t>(it - hashes_.begin());
134 // collision scan
135 size_t lo = idx, hi = idx;
136 while (lo > 0 && hashes_[lo - 1] == h) {
137 --lo;
138 }
139 while (hi + 1 < hashes_.size() && hashes_[hi + 1] == h) {
140 ++hi;
141 }
142
143 for (size_t i = lo; i <= hi; ++i) {
144 if (paths_[i] == path) { return read_at_index(i, out); }
145 }
146 return false;
147 }
