11std::string ResolveWorldHudShaderPath() {

14 constexpr const char* kDefaultShader = "SHADERS/default.shader";
15
16 if (resource::Open(kWorldHudShader).has_value()) {

20 return kWorldHudShaderLower;

49void HUDWorldRendererComponent::OnRender(renderer::IRenderablePass pass, const glm::mat4& precomputed_mat) noexcept {
50 if (!enabled() || pass != renderer::IRenderablePass::GEOMETRY) {
51 return;
52 }
53 if (!m_mesh || !m_shader) {
54 return;
55 }
56
57 if (m_view && m_hudLayer) {
58 m_textureId = m_hudLayer->GetViewTextureGL(m_view);
59 }
60 if (m_textureId == 0) {
61 return;
62 }
63
64 const glm::mat4 model = GetWorldMatrix();
65 const glm::mat4 mvp = precomputed_mat * model;
66
67 m_shader->Use();
68 m_shader->Set("gWorld", model);
69 m_shader->Set("gMVP", mvp);
70 m_shader->Set("gColor", glm::vec4(1.0f));
71
72 glActiveTexture(GL_TEXTURE0);
73 glBindTexture(GL_TEXTURE_2D, m_textureId);
74 m_shader->SetSampler("gTexture", 0);
75
76 glDepthMask(m_drawToDepth ? GL_TRUE : GL_FALSE);
77 glEnable(GL_BLEND);
78 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
79
80 m_mesh->Draw();
81}
