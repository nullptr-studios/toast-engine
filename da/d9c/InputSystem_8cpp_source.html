75void InputSystem::Tick() {
76 // Check all connected controllers for button/axis changes
77 PollControllers();
78
79 // For held keys, we need to dispatch Ongoing events every frame
80 // Add all actions with pressed keys to the dispatch queue
81 if (HasActiveLayout()) {
82 // 0D actions that are held
83 for (auto& action : m.activeLayout->m.actions0d) {
84 if (!action.m.pressedKeys.empty() && action.CheckState(m.currentState)) {
85 AddToQueue(m.dispatch0DQueue, &action);
86 }
87 }
88
89 // 1D actions that are held
90 for (auto& action : m.activeLayout->m.actions1d) {
91 if (!action.m.pressedKeys.empty() && action.CheckState(m.currentState)) {
92 AddToQueue(m.dispatch1DQueue, &action);
93 }
94 }
95
96 // 2D actions that are held
97 for (auto& action : m.activeLayout->m.actions2d) {
98 if (!action.m.pressedKeys.empty() && action.CheckState(m.currentState)) {
99 AddToQueue(m.dispatch2DQueue, &action);
100 }
101 }
102 }
103
104 // Dispatch all queued actions
105 DispatchQueue(m.dispatch0DQueue, &Listener::M::callbacks0d);
106 DispatchQueue(m.dispatch1DQueue, &Listener::M::callbacks1d);
107 DispatchQueue(m.dispatch2DQueue, &Listener::M::callbacks2d);
108}

172bool InputSystem::Handle0DAction(int key_code, int action, int mods, Device device) {
173 // Ignore OS key repeat events (action == 2 = GLFW_REPEAT)
174 // We handle held keys ourselves by dispatching every frame in Tick()
175 if (action == 2) {
176 return false;
177 }
178
179 // Map button/keypress to 0D actions (boolean-like)
180 for (auto& act : m.activeLayout->m.actions0d) {
181 if (!act.CheckState(m.currentState)) {
182 continue;
183 }
184 for (const auto& bind : act.m.binds) {
185 if (bind.device != device) {
186 continue;
187 }
188 if (!bind.keys.contains(key_code)) {
189 continue;
190 }
191
192 act.device = bind.device;
193 act.modifiers = static_cast<ModifierKey>(mods);
194
195 // action: 0 = release, 1 = press
196 if (action == 0) {
197 // Key released - remove from pressed keys
198 act.m.pressedKeys.erase(key_code);
199 } else if (action == 1) {
200 // Key pressed - add to pressed keys
201 act.m.pressedKeys.emplace(key_code, true);
202 }
203
204 // Add to queue - will be dispatched in Tick()
205 AddToQueue(m.dispatch0DQueue, &act);
206 return true;
207 }
208 }
209 return false;
210}

311bool InputSystem::OnMousePosition(event::WindowMousePosition* e) {
312 // Store mouse delta and mouse position
313 m.oldMousePosition = m.mousePosition;
314 m.mousePosition = glm::vec2 { e->x, e->y };
315 m.mouseDelta = m.mousePosition - m.oldMousePosition;
316
317 // Mouse position as a 2D action (normalized to NDC)
318 // NOTE: Mouse position should NOT have Started/Ongoing states like keys
319 // It's a continuous value that updates every frame
320 if (!HasActiveLayout()) {
321 return false;
322 }
323
324 for (auto& action : m.activeLayout->m.actions2d) {
325 if (!action.CheckState(m.currentState)) {
326 continue;
327 }
328 for (const auto& bind : action.m.binds) {
329 if (bind.device != Device::Mouse) {
330 continue;
331 }
332 for (const auto& [key, _] : bind.keys) {
333 // Mouse position in normalized device coordinates [-1, 1]
334 if (key == MOUSE_POSITION_CODE) {
335 action.device = bind.device;
336
337 glm::vec2 value = m.mousePosition;
338
339#if defined(__linux__)
340 // Adjust for display scale on Linux
341 const auto [sx, sy] = toast::Window::GetInstance()->GetDisplayScale();
342 value.x *= sx;
343 value.y *= sy;
344#endif
345
346 // Convert screen coords to NDC [-1, 1]
347 const auto s = toast::Window::GetInstance()->GetFramebufferSize();
348 value.x = (value.x / s.x) - 0.5f;
349 value.y = (value.y / s.y) - 0.5f;
350 value *= 2.0f;
351
352 // Store as transient input (will be cleared after dispatch)
353 action.m.pressedKeys[MOUSE_POSITION_CODE] = value;
354 AddToQueue(m.dispatch2DQueue, &action);
355 return true;
356 }
357
358 // Raw mouse position (screen coordinates)
359 if (key == MOUSE_RAW_CODE) {
360 action.device = bind.device;
361 action.m.pressedKeys[MOUSE_RAW_CODE] = m.mousePosition;
362 AddToQueue(m.dispatch2DQueue, &action);
363 return true;
364 }
365
366 // Mouse delta (movement since last frame)
367 if (key == MOUSE_DELTA_CODE) {
368 action.device = bind.device;
369 action.m.pressedKeys[MOUSE_DELTA_CODE] = m.mouseDelta;
370 AddToQueue(m.dispatch2DQueue, &action);
371 return true;
372 }
373 }
374 }
375 }
376 return false;

488void InputSystem::PollControllers() {
489 // Skip if no active layout
490 if (!HasActiveLayout()) {
491 return;
492 }
493
494 for (auto& [jid, state] : m.controllers) {
495 // Refresh controller state from GLFW
496 state.previous = state.current;
497 glfwGetGamepadState(jid, &state.current);
498
499 // Button transitions (press/release)
500 for (int i = 0; i <= GLFW_GAMEPAD_BUTTON_LAST; ++i) {
501 const bool was_pressed = state.previous.buttons[i] == GLFW_PRESS;
502 const bool is_pressed = state.current.buttons[i] == GLFW_PRESS;
503 if (!was_pressed && is_pressed) {
504 ControllerButton(i, true);
505 } else if (was_pressed && !is_pressed) {
506 ControllerButton(i, false);
507 }
508 }
509
510 // Axis changes (with deadzone and normalization)
511 std::array<float, 6> axes {};
512 for (int i = 0; i <= GLFW_GAMEPAD_AXIS_LAST; ++i) {
513 // Apply deadzone to previous and current values
514 const float prev = std::abs(state.previous.axes[i]) > AXIS_DEADZONE ? state.previous.axes[i] : 0.0f;
515 const float curr = std::abs(state.current.axes[i]) > AXIS_DEADZONE ? state.current.axes[i] : 0.0f;
516
517 // Only process if the axis value changed meaningfully
518 // The 0.001f threshold prevents micro-jitter
519 if (std::abs(curr - prev) > 0.001f) {
520 // Copy all axes and apply transformations
521 std::ranges::copy(state.current.axes, axes.begin());
522
523 // Invert Y axes to match standard coordinate system (up = positive)
524 axes[1] *= -1.0f; // Left stick Y
525 axes[3] *= -1.0f; // Right stick Y
526
527 // Normalize trigger values from [-1, 1] to [0, 1]
528 // GLFW reports triggers as axis values but they should be unidirectional
529 axes[4] = (axes[4] * 0.5f) + 0.5f; // Left trigger (L2)
530 axes[5] = (axes[5] * 0.5f) + 0.5f; // Right trigger (R2)
531
532 // Apply deadzone to all processed axes
533 // This ensures values below threshold are clamped to zero
534 std::ranges::for_each(axes, [](float& ax) {
535 if (std::abs(ax) < AXIS_DEADZONE) {
536 ax = 0.0f;
537 }
538 });
539
540 // Route axis changes to appropriate action handlers
541 ControllerAxis(i, axes);
542 }
543 }
544 }

623void InputSystem::ControllerAxis(int id, std::array<float, 6> axes) {
624 // Controller axes -> 1D
625 for (auto& action : m.activeLayout->m.actions1d) {
626 if (!action.CheckState(m.currentState)) {
627 continue;
628 }
629 for (const auto& bind : action.m.binds) {
630 if (bind.device != Device::ControllerAxis) {
631 continue;
632 }
633 for (const auto& [key, direction] : bind.keys) {
634 if (key != id) {
635 continue;
636 }
637 action.device = bind.device;
638 const float value = axes[id];
639
640 // Key offset: Add large constant to avoid collision with other input types
641 const int key_code = id + static_cast<int>(2e7);
642
643 if (value != 0.0f) {
644 // Axis is active - store the scaled value
645 action.m.pressedKeys[key_code] = bind.GetFloatValue(direction) * value;
646 } else {
647 // Axis returned to neutral - clear it
648 action.m.pressedKeys.erase(key_code);
649 }
650 AddToQueue(m.dispatch1DQueue, &action);
651 return;
652 }
653 }
654 }
655
656 // Controller axes & sticks -> 2D
657 for (auto& action : m.activeLayout->m.actions2d) {
658 if (!action.CheckState(m.currentState)) {
659 continue;
660 }
661 for (const auto& bind : action.m.binds) {
662 if (bind.device != Device::ControllerAxis && bind.device != Device::ControllerStick) {
663 continue;
664 }
665 for (const auto& [key, direction] : bind.keys) {
666 if (bind.device == Device::ControllerAxis && key == id) {
667 action.device = bind.device;
668 const float value = axes[id];
669
670 // Key offset for single axis
671 const int key_code = id + static_cast<int>(2e7);
672
673 if (value != 0.0f) {
674 // Store scaled 2D value
675 action.m.pressedKeys[key_code] = bind.GetVec2Value(direction) * value;
676 } else {
677 // Clear when axis returns to neutral
678 action.m.pressedKeys.erase(key_code);
679 }
680 AddToQueue(m.dispatch2DQueue, &action);
681 return;
682 }
683 if (bind.device == Device::ControllerStick) {
684 // Left Stick: axes 0 (X) and 1 (Y)
685 // Key 0 indicates left stick binding
686 if (key == 0 && (id == 0 || id == 1)) {
687 action.device = Device::ControllerStick;
688
689 // Get individual axis values
690 // Note: axes[1] is already inverted in PollControllers()
691 const float x_value = axes[0];
692 const float y_value = axes[1];
693
694 const int key_code = static_cast<int>(2e8);
695
696 // Check if stick is outside deadzone (any axis active)
697 if (std::abs(x_value) > m.triggerDeadzone || std::abs(y_value) > m.triggerDeadzone) {
698 // Stick is active - store the full 2D vector as a single entry
699 // This ensures accurate analog values without artificial clamping
700 action.m.pressedKeys[key_code] = glm::vec2 { x_value, y_value };
701 } else {
702 // Stick returned to neutral, clear the entry
703 action.m.pressedKeys.erase(key_code);
704 }
705 AddToQueue(m.dispatch2DQueue, &action);
706 return;
707 }
708 // Right stick (axes 2,3) mapped when key == 1
709 if (key == 1 && (id == 2 || id == 3)) {
710 action.device = Device::ControllerStick;
711
712 // Get individual axis values
713 // Note: axes[3] is already inverted in PollControllers()
714 const float x_value = axes[2];
715 const float y_value = axes[3];
716
717 // Use a single key for the stick
718 const int key_code = static_cast<int>(2e8) + 1;
719
720 // Check if stick is outside deadzone (any axis active)
721 if (std::abs(x_value) > m.triggerDeadzone || std::abs(y_value) > m.triggerDeadzone) {
722 // Stick is active - store the full 2D vector as a single entry
723 action.m.pressedKeys[key_code] = glm::vec2 { x_value, y_value };
