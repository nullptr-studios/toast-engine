126bool InputSystem::Handle0DAction(int key_code, int action, int mods) {
127 // Map button/keypress to 0D actions (boolean-like)
128 for (auto& act : m.activeLayout->m.actions0d) {
129 if (!act.CheckState(m.currentState)) {
130 continue;
131 }
132 for (const auto& bind : act.m.binds) {
133 if (!bind.keys.contains(key_code) || action == 2) {
134 continue;
135 }
136
137 act.device = bind.device;
138 act.modifiers = static_cast<ModifierKey>(mods);
139
140 // 0 = release, 1 = press
141 if (action == 0) {
142 act.m.pressedKeys.erase(key_code);
143 } else if (action == 1) {
144 act.m.pressedKeys.emplace(key_code, true);
145 }
146
147 AddToQueue(m.dispatch0DQueue, &act);
148 return true;
149 }
150 }
151 return false;
152}

232bool InputSystem::OnMousePosition(event::WindowMousePosition* e) {
233 // Mouse position as a 2D action (normalized to NDC)
234 if (!HasActiveLayout()) {
235 return false;
236 }
237
238 for (auto& action : m.activeLayout->m.actions2d) {
239 if (!action.CheckState(m.currentState)) {
240 continue;
241 }
242 for (const auto& bind : action.m.binds) {
243 if (bind.device != Device::Mouse) {
244 continue;
245 }
246 for (const auto& [key, _] : bind.keys) {
247 if (key != MOUSE_POSITION_CODE) {
248 continue;
249 }
250
251 action.device = bind.device;
252 action.state = Action2D::Ongoing;
253
254 glm::vec2 value = { e->x, e->y };
255
256#if defined(__linux__)
257 // Adjust for display scale on Linux
258 const auto [sx, sy] = toast::Window::GetInstance()->GetDisplayScale();
259 value.x *= sx;
260 value.y *= sy;
261#endif
262
263 // Convert screen coords to NDC [-1, 1]
264 const auto [w, h] = toast::Window::GetInstance()->GetFramebufferSize();
265 value.x = (value.x / w) - 0.5f;
266 value.y = (value.y / h) - 0.5f;
267 value *= 2.0f;
268 value.y *= -1.0f; // y up
269
270 action.m.pressedKeys.emplace(MOUSE_POSITION_CODE, value);
271 AddToQueue(m.dispatch2DQueue, &action);
272 return true;
273 }

389void InputSystem::PollControllers() {
390 // Skip if no active layout
391 if (!HasActiveLayout()) {
392 return;
393 }
394
395 for (auto& [jid, state] : m.controllers) {
396 // Refresh controller state
397 state.previous = state.current;
398 glfwGetGamepadState(jid, &state.current);
399
400 // Button transitions (press/release)
401 for (int i = 0; i <= GLFW_GAMEPAD_BUTTON_LAST; ++i) {
402 const bool was_pressed = state.previous.buttons[i] == GLFW_PRESS;
403 const bool is_pressed = state.current.buttons[i] == GLFW_PRESS;
404 if (!was_pressed && is_pressed) {
405 ControllerButton(i, true);
406 } else if (was_pressed && !is_pressed) {
407 ControllerButton(i, false);
408 }
409 }
410
411 // Axis changes (with deadzone and normalization)
412 std::array<float, 6> axes {};
413 for (int i = 0; i <= GLFW_GAMEPAD_AXIS_LAST; ++i) {
414 const float prev = std::abs(state.previous.axes[i]) > AXIS_DEADZONE ? state.previous.axes[i] : 0.0f;
415 const float curr = std::abs(state.current.axes[i]) > AXIS_DEADZONE ? state.current.axes[i] : 0.0f;
416 if (std::abs(curr - prev) > 0.001f) {
417 std::ranges::copy(state.current.axes, axes.begin());
418 axes[1] *= -1; // invert Y on left stick
419 axes[3] *= -1; // invert Y on right stick
420 axes[4] = (axes[4] * 0.5f) + 0.5f; // trigger L2 to [0,1]
421 axes[5] = (axes[5] * 0.5f) + 0.5f; // trigger R2 to [0,1]
422 std::ranges::for_each(axes, [](float& ax) {
423 if (std::abs(ax) < AXIS_DEADZONE) {
424 ax = 0.0f;
425 }
426 });
427 ControllerAxis(i, axes);
428 }
429 }
430 }

509void InputSystem::ControllerAxis(int id, std::array<float, 6> axes) {
510 // Controller axes -> 1D
511 for (auto& action : m.activeLayout->m.actions1d) {
512 if (!action.CheckState(m.currentState)) {
513 continue;
514 }
515 for (const auto& bind : action.m.binds) {
516 if (bind.device != Device::ControllerAxis) {
517 continue;
518 }
519 for (const auto& [key, direction] : bind.keys) {
520 if (key != id) {
521 continue;
522 }
523 action.device = bind.device;
524 const float value = axes[id];
525 if (value != 0.0f) {
526 action.m.pressedKeys[id + 2e7] = bind.GetFloatValue(direction) * value;
527 } else {
528 action.m.pressedKeys.erase(id + 2e7);
529 }
530 AddToQueue(m.dispatch1DQueue, &action);
531 return;
532 }
533 }
534 }
535
536 // Controller axes & sticks -> 2D
537 for (auto& action : m.activeLayout->m.actions2d) {
538 if (!action.CheckState(m.currentState)) {
539 continue;
540 }
541 for (const auto& bind : action.m.binds) {
542 if (bind.device != Device::ControllerAxis && bind.device != Device::ControllerStick) {
543 continue;
544 }
545 for (const auto& [key, direction] : bind.keys) {
546 if (bind.device == Device::ControllerAxis && key == id) {
547 action.device = bind.device;
548 const float value = axes[id];
549 if (value != 0.0f) {
550 action.m.pressedKeys.emplace(id + 2e7, bind.GetVec2Value(direction) * value);
551 } else {
552 action.m.pressedKeys.erase(id + 2e7);
553 }
554 AddToQueue(m.dispatch2DQueue, &action);
555 return;
556 }
557 if (bind.device == Device::ControllerStick) {
558 // Left stick (axes 0,1) mapped when key == 0
559 if (key == 0 && (id == 0 || id == 1)) {
560 action.device = Device::ControllerStick;
561 glm::vec2 value = { axes[0], -axes[1] };
562 if (value != glm::vec2 { 0.0f, 0.0f }) {
563 action.m.pressedKeys[id + 2e8] = value;
564 } else {
565 action.m.pressedKeys.erase(id + 2e8);
566 }
567 AddToQueue(m.dispatch2DQueue, &action);
568 return;
569 }
570 // Right stick (axes 2,3) mapped when key == 1
571 if (key == 1 && (id == 2 || id == 3)) {
572 action.device = Device::ControllerStick;
573 glm::vec2 value = { axes[2], -axes[3] };
574 if (value != glm::vec2 { 0.0f, 0.0f }) {
575 action.m.pressedKeys[id + 2e8] = value;
576 } else {
577 action.m.pressedKeys.erase(id + 2e8);
578 }
579 AddToQueue(m.dispatch2DQueue, &action);
580 return;
581 }
582 }
583 }
