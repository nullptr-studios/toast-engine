175void TransformComponent::ComputeParentWorldTRS(glm::vec3& outPos, glm::quat& outRot, glm::vec3& outScl) const noexcept {
176 // If world matrix is clean, reuse cached parent TRS
177 if (!m_dirtyWorldMatrix) {
178 outPos = m_cachedParentWorldPos;
179 outRot = m_cachedParentWorldRot;
180 outScl = m_cachedParentWorldScl;
181 return;
182 }
183
184 // Recompute and update cache when world is dirty
185 glm::vec3 accPos = glm::vec3(0.0f);
186 glm::quat accRot = glm::identity<glm::quat>();
187 glm::vec3 accScl = glm::vec3(1.0f);
188
189 Object* object_ptr = parent();
190 while (object_ptr) {
191 TransformComponent* parent_transform = nullptr;
192
193 if (Actor* parent_actor = dynamic_cast<Actor*>(object_ptr)) {
194 // Skip using our own transform as parent if parent is our owning Actor
195 if (parent_actor->transform() == this) {
196 object_ptr = object_ptr->parent();
197 continue;
198 }
199 parent_transform = parent_actor->transform();
200 } else if (TransformComponent* parent_comp = dynamic_cast<TransformComponent*>(object_ptr)) {
201 parent_transform = parent_comp;
202 } else {
203 // Skip non-transform parents
204 object_ptr = object_ptr->parent();
205 continue;
206 }
207
208 if (parent_transform) {
209 accPos = parent_transform->rotationQuat() * (accPos * parent_transform->scale()) + parent_transform->position();
210 accRot = parent_transform->rotationQuat() * accRot;
211 accScl *= parent_transform->scale();
212 }
213
214 object_ptr = object_ptr->parent();
215 }
216
217 // Update cache and output
218 m_cachedParentWorldPos = accPos;
219 m_cachedParentWorldRot = accRot;
220 m_cachedParentWorldScl = accScl;
221
222 outPos = accPos;
223 outRot = accRot;
224 outScl = accScl;
225}

347glm::mat4 TransformComponent::GetInverse() noexcept {
348 if (!m_dirtyInverse) {
349 return m_cachedInverse;
350 }
351
352 constexpr float EPSILON = 1e-6f;
353 glm::vec3 inv_scale { std::abs(m_scale.x) > EPSILON ? 1.0f / m_scale.x : 0.0f,
354 std::abs(m_scale.y) > EPSILON ? 1.0f / m_scale.y : 0.0f,
355 std::abs(m_scale.z) > EPSILON ? 1.0f / m_scale.z : 0.0f };
356
357 // If any inverse scale is zero, ignore position along that axis to avoid inf in translation
358 glm::vec3 safe_position { inv_scale.x != 0.0f ? -m_position.x : 0.0f,
359 inv_scale.y != 0.0f ? -m_position.y : 0.0f,
360 inv_scale.z != 0.0f ? -m_position.z : 0.0f };
361
362 // S*R*T
363 mat4 pos_mat = translate(identity<mat4>(), safe_position);
364 mat4 rot_mat = mat4_cast(inverse(m_rotation));
365 mat4 scl_mat = glm::scale(identity<mat4>(), inv_scale);
366 m_cachedInverse = scl_mat * rot_mat * pos_mat;
367
368 m_dirtyInverse = false;
369 return m_cachedInverse;
370}

372glm::mat4 TransformComponent::GetWorldMatrix() noexcept {
373 if (!m_dirtyWorldMatrix) {
374 return m_cachedWorldMatrix;
375 }
376
377 // If no transform-bearing parent, world == local
378 if (!parent()) {
379 m_cachedWorldMatrix = GetMatrix();
380 m_dirtyWorldMatrix = false;
381 return m_cachedWorldMatrix;
382 }
383
384 glm::vec3 world_pos = m_position;
385 glm::quat world_rot = m_rotation;
386 glm::vec3 world_scl = m_scale;
387
388 Object* object_ptr = parent();
389 while (object_ptr) {
390 TransformComponent* parent_transform = nullptr;
391
392 if (Actor* parent_actor = dynamic_cast<Actor*>(object_ptr)) {
393 if (parent_actor->transform() == this) {
394 object_ptr = object_ptr->parent();
395 continue;
396 }
397 parent_transform = parent_actor->transform();
398 } else if (TransformComponent* parent_comp = dynamic_cast<TransformComponent*>(object_ptr)) {
399 parent_transform = parent_comp;
400 } else {
401 // Skip non-transform parents
402 object_ptr = object_ptr->parent();
403 continue;
404 }
405
406 if (parent_transform) {
407 // world_child = T_parent * R_parent * S_parent applied to local/world accumulator
408 world_pos = parent_transform->rotationQuat() * (world_pos * parent_transform->scale()) + parent_transform->position();
409 world_rot = parent_transform->rotationQuat() * world_rot;
410 world_scl *= parent_transform->scale();
411 }
412
413 object_ptr = object_ptr->parent();
414 }
415
416 mat4 pos_mat = translate(identity<mat4>(), world_pos);
417 mat4 rot_mat = mat4_cast(world_rot);
418 mat4 scl_mat = glm::scale(identity<mat4>(), world_scl);
419 m_cachedWorldMatrix = pos_mat * rot_mat * scl_mat;
420
421 m_dirtyWorldMatrix = false;
422 return m_cachedWorldMatrix;
423}
