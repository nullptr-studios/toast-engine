22std::optional<glm::dvec2> LineLineCollision(const Line& a, const Line& b) {
23 // directions
24 glm::dvec2 a_vec = a.p2 - a.p1;
25 glm::dvec2 b_vec = b.p2 - b.p1;
26
27 glm::dvec2 start_delta = b.p1 - a.p1;
28
29 double cross_a_b = determinant(dmat2 { a_vec, b_vec }); // a_vec x b_vec
30 double cross_delta_b = determinant(dmat2 { start_delta, b_vec }); // (b_start - a_start) x b_vec
31 double cross_delta_a = determinant(dmat2 { start_delta, a_vec }); // (b_start - a_start) x a_vec
32
33 // Parallel (including colinear)
34 if (std::abs(cross_a_b) < PhysicsSystem::eps_small()) {
35 return std::nullopt;
36 }
37
38 double t_on_a = cross_delta_b / cross_a_b; // position along segment a
39 double t_on_b = cross_delta_a / cross_a_b; // position along segment b
40
41 // Intersection must lie within both segments
42 if (t_on_a < 0.0 || t_on_a > 1.0 || t_on_b < 0.0 || t_on_b > 1.0) {
43 return std::nullopt;
44 }
45
46 return dvec2 { a.p1 } + t_on_a * a_vec;
47}

49void BoxKinematics(BoxRigidbody* rb) {
50 auto& velocity = rb->velocity;
51 auto& angular_velocity = rb->angularVelocity;
52
53 // Avoid boxes with 0 or less mass
54 if (rb->mass < 0.1) {
55 rb->mass = 1.0;
56 }
57
58 // Sum torques
59 double torques = std::ranges::fold_left(rb->torques, 0.0, std::plus {});
60 // Compute inertia
61 dvec2 half_size = rb->size / 2.0;
62 double inertia = (rb->mass * (half_size.x * half_size.x + half_size.y * half_size.y)) / 12.0;
63 inertia = (inertia < 0.001) ? 1.0 : inertia;
64 // Calculate angular acceleration
65 double angular_accel = torques / inertia;
66
67 // Sum forces
68 glm::dvec2 forces = std::ranges::fold_left(rb->forces, glm::dvec2 { 0.0 }, std::plus {});
69 glm::dvec2 accel = (forces / rb->mass) + (PhysicsSystem::gravity() * dvec2 { rb->gravityScale });
70
71 // Integrate velocities
72 velocity += accel * Time::fixed_delta();
73 angular_velocity += angular_accel * Time::fixed_delta();
74
75 // Apply drag
76 const double damping = std::exp(-rb->linearDrag * Time::fixed_delta());
77 velocity *= damping;
78
79 const double angular_damping = std::exp(-rb->angularDrag * Time::fixed_delta());
80 angular_velocity *= angular_damping;
81
82 // Stop the object if less than minimum velocity
83 if (all(lessThan(abs(velocity), rb->minimumVelocity))) {
84 velocity = { 0.0, 0.0 };
85 }
86 if (angular_velocity < rb->minimumAngularVelocity) {
87 angular_velocity = 0.0;
88 }
89}

115static auto ClipLineSegmentToLine(dvec2 p1, dvec2 p2, dvec2 normal, dvec2 offset) -> std::vector<dvec2> {
116 std::vector<dvec2> points;
117 points.reserve(2);
118
119 double distance1 = dot(p1 - offset, normal);
120 double distance2 = dot(p2 - offset, normal);
121
122 // if the points are behind the plane, don't clip
123 if (distance1 <= 0.0) {
124 points.emplace_back(p1);
125 }
126 if (distance2 <= 0.0) {
127 points.emplace_back(p2);
128 }
129
130 // if one is in front of the plane, clip it to the intersection point
131 if (std::signbit(distance1) != std::signbit(distance2) && points.size() < 2) {
132 const double pct_across = distance2 / (distance2 - distance1);
133 const dvec2 intersection_point = (p1 - p2) * pct_across;
134 points.emplace_back(intersection_point);
135 }
136
137 return points;
138}

140auto BoxMeshCollision(BoxRigidbody* rb, ConvexCollider* c) -> std::optional<BoxManifold> {
141 dvec2 rb_pos = rb->GetPosition();
142 std::list<BoxManifold> manifolds;
143
144 int edge_count = 0;
145 // This method will use the SAT algorithm
146 for (const auto& edge : c->edges) {
147 // get the maximum and minimum points of the collider projected onto the edge normal
148 double min_collider = dot(dvec2 { c->vertices[0] }, edge.normal);
149 double max_collider = min_collider;
150
151 for (int i = 1; i < c->vertices.size(); i++) {
152 double proj = dot(dvec2 { c->vertices[i] }, edge.normal);
153 min_collider = std::min(proj, min_collider);
154 max_collider = std::max(proj, max_collider);
155 }
156
157 // get the rigidbody OBB vertices
158 std::vector<vec2> rb_points = rb->GetPoints();
159
160 // project rigidbody vertices onto the same axis
161 double min_rb = dot(dvec2 { rb_points[0] }, edge.normal);
162 double max_rb = min_rb;
163
164 for (int i = 1; i < rb_points.size(); i++) {
165 double proj = dot(dvec2 { rb_points[i] }, edge.normal);
166 min_rb = std::min(proj, min_rb);
167 max_rb = std::max(proj, max_rb);
168 }
169
170 // if theres no overlap on this axis, there is no collision at all
171 if (max_rb < min_collider || min_rb > max_collider) {
172 return std::nullopt;
173 }
174
175 // compute overlap along the axis (penetration depth candidate)
176 double overlap = std::min(max_collider - min_rb, max_rb - min_collider);
177
178 // find closest rigidbody vertex to the edge segment (used for biasing)
179 dvec2 a = edge.p1;
180 dvec2 b = edge.p2;
181 dvec2 ab = b - a;
182
183 double best_dist2 = std::numeric_limits<double>::max();
184 for (dvec2 p : rb_points) {
185 double t = dot(p - a, ab) / dot(ab, ab);
186 t = std::clamp(t, 0.0, 1.0);
187 dvec2 closest = a + ab * t;
188 double dist2 = dot(p - closest, p - closest);
189 best_dist2 = std::min(best_dist2, dist2);
190 }
191
192 // bias depth so nearer parallel edges win
193 overlap += PhysicsSystem::eps() * best_dist2;
194
195 // store candidate manifold
196 // clang-format off
197 auto manifold = BoxManifold{
198 .normal = edge.normal,
199 .contact1 = {0.0, 0.0},
200 .contact2 = {0.0, 0.0},
201 .contactCount = -1,
202 .depth = overlap
203 };
204 // clang-format on
205
206 std::vector<dvec2> points;
207 for (const auto& e : rb->GetEdges()) {
208 auto p = LineLineCollision(e, edge);
209 if (p.has_value()) {
210 points.emplace_back(p.value());
211 }
212 }
213
214 if (points.size() >= 2) {
215 manifold.contact1 = points[0];
216 manifold.contact2 = points[1];
217 manifold.contactCount = 2;
218 } else if (points.size() == 1) {
219 manifold.contact1 = points[0];
220 manifold.contactCount = 1;
221 } else {
222 manifold.contact1 = rb->GetPosition();
223 manifold.contact2 = rb->GetPosition();
224 manifold.contactCount = 1;
225 }
226
227 manifolds.emplace_back(manifold);
228 }
229
230 // select the axis with the least penetration depth
231 auto it = std::ranges::min_element(manifolds, {}, &BoxManifold::depth);
232 if (it == manifolds.end()) {
233 return std::nullopt;
234 }
235 auto best = *it;
236
237 if (rb->debug.showManifolds) {
238 best.Debug();
239 }
240 return best;
241}

243void BoxMeshResolution(BoxRigidbody* rb, ConvexCollider* c, BoxManifold manifold) {
244 dvec2 position = rb->GetPosition();
245 dvec2& velocity = rb->velocity;
246 double& angular_velocity = rb->angularVelocity;
247
248 // Skip if body has infinite mass
249 if (rb->mass <= 0.0) {
250 return;
251 }
252 double inv_mass = 1.0 / rb->mass;
253
254 // moment of inertia for box
255 double width = rb->size.x / 2;
256 double height = rb->size.y / 2;
257 double inertia = (rb->mass * (width * width + height * height)) / 12.0;
258 double inv_inertia = inertia > 0.0 ? 1.0 / inertia : 0.0;
259
260 dvec2 contact = (manifold.contactCount == 2) ? (manifold.contact1 + manifold.contact2) * 0.5 : manifold.contact1;
261 dvec2 r = contact - position;
262
263 // unfold velocity in normal and tangencial
264 dvec2 contact_tangent = { -manifold.normal.y, manifold.normal.x };
265 double normal_speed = dot(velocity, manifold.normal);
266 double tangent_speed = dot(velocity, contact_tangent);
267
268 // only apply restitution if we're moving faster than the restitution threshold
269 double restitution = (std::abs(normal_speed) < rb->restitutionThreshold) ? 0.0 : rb->restitution;
270
271 // only resolve if we're going towards the object
272 if (normal_speed < 0.0) {
273 double normal_lever_arm = determinant(dmat2 { r, manifold.normal });
274 double tangent_lever_arm = determinant(dmat2 { r, contact_tangent });
275 double normal_effective_mass = inv_mass + (normal_lever_arm * normal_lever_arm * inv_inertia);
276 double tangent_effective_mass = inv_mass + (tangent_lever_arm * tangent_lever_arm * inv_inertia);
277
278 // Normal impulse (bounce response)
279 double normal_impulse = -(1.0 + restitution) * normal_speed / normal_effective_mass;
280
281 // Coulomb friction
282 double max_friction_impulse = c->friction * std::abs(normal_impulse);
283
284 // calculate tangential impulse to cancel tangential speed
285 double tangential_impulse = -tangent_speed / tangent_effective_mass;
286 tangential_impulse = clamp(tangential_impulse, -max_friction_impulse, max_friction_impulse);
287
288 // total impulse
289 dvec2 impulse = normal_impulse * manifold.normal + tangential_impulse * contact_tangent;
290
291 // Apply impulses to velocity
292 velocity += impulse * inv_mass;
293
294 // Apply angular impulse with slight damping to smooth spikes
295 dmat2 torque_mat = { r, impulse };
296 double torque_impulse = determinant(torque_mat);
297 const double angular_impulse_blend = 0.6;
298 angular_velocity += torque_impulse * inv_inertia * angular_impulse_blend;
299 }
300
301 // positional correction
302 double penetration_correction = std::max(manifold.depth - PhysicsSystem::pos_slop(), 0.0) * PhysicsSystem::pos_ptc();
303 position += penetration_correction * manifold.normal;
304 rb->SetPosition(position);
305
306 // velocity correction
307 double residual_normal_speed = dot(velocity, manifold.normal);
308 if (residual_normal_speed < 0.0) {
309 velocity -= residual_normal_speed * manifold.normal;
310 }
311}
