180auto BoxRigidbody::GetPoints() const -> std::vector<glm::vec2> {
181 glm::dvec2 position = GetPosition();
182 double rotation = GetRotation();
183 glm::dvec2 scale = size;
184
185 std::vector<glm::vec2> points(4);
186
187 // Half-extents for centering
188 glm::dvec2 h = scale * 0.5;
189 double cos_r = std::cos(rotation);
190 double sin_r = std::sin(rotation);
191
192 // Define local corners relative to center
193 std::array<glm::dvec2, 4> corners = {
194 glm::dvec2 { -h.x, -h.y },
195 glm::dvec2 { h.x, -h.y },
196 glm::dvec2 { h.x, h.y },
197 glm::dvec2 { -h.x, h.y }
198 };
199
200 for (int i = 0; i < 4; i++) {
201 // Rotation -> Position
202 points[i].x = (float)((corners[i].x * cos_r) - (corners[i].y * sin_r) + position.x);
203 points[i].y = (float)((corners[i].x * sin_r) + (corners[i].y * cos_r) + position.y);
204 }
205
206 return points;
207}

209auto BoxRigidbody::GetEdges() const -> std::vector<Line> {
210 std::vector<glm::vec2> points = GetPoints();
211 std::vector<Line> lines;
212 lines.reserve(points.size());
213
214 for (int i = 0; i < points.size(); i++) {
215 const glm::vec2& p1 = points[i];
216 const glm::vec2& p2 = points[(i + 1) % points.size()];
217
218 // compute edge direction
219 glm::vec2 edge = p2 - p1;
220
221 // compute perpendicular vector
222 glm::vec2 normal = glm::normalize(glm::vec2 { -edge.y, edge.x });
223
224 // clang-format off
225 lines.emplace_back(Line {
226 .p1 = p1,
227 .p2 = p2,
228 .normal = normal,
229 .tangent = {-normal.y, normal.x},
230 .length = glm::distance(p1, p2)
231 });
232 // clang-format on
233 }
234
235 return lines;
236}

246void BoxRigidbody::AddForce(glm::dvec2 force, glm::dvec2 position) {
247 // Get position with [0, 0] as center of mass
248 glm::dvec2 rel_pos = position - GetPosition();
249
250 // Decompose into force + torque
251 // Linear force always apply
252 AddForce(force);
253
254 // torque = cross product position and force
255 glm::dmat2 mat = { rel_pos, force };
256 AddTorque(glm::determinant(mat));
257}
