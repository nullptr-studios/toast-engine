53void Texture::Load() {
54 PROFILE_ZONE;
55 SetResourceState(resource::ResourceState::LOADING);
56
57 std::vector<uint8_t> f {};
58 if (!resource::Open(m_path, f)) {
59 TOAST_ERROR("Failed to load texture: {0}", m_path);
60 SetResourceState(resource::ResourceState::FAILED);
61 return;
62 }
63
64 int channels = 0;
65
66 // Decode without forcing 4 channels
67 unsigned char* pixels = stbi_load_from_memory(f.data(), static_cast<int>(f.size()), &m_width, &m_height, &channels, 0);
68 if (!pixels) {
69 TOAST_ERROR("Failed to load texture from memory: {0}", m_path);
70 SetResourceState(resource::ResourceState::FAILED);
71 return;
72 }
73
74 if (m_width <= 0 || m_height <= 0 || channels <= 0) {
75 TOAST_ERROR("Invalid texture dimensions or channels: {0}", m_path);
76 stbi_image_free(pixels);
77 SetResourceState(resource::ResourceState::FAILED);
78 return;
79 }
80
81 m_channels = channels;
82 m_pixels = pixels; // Take ownership
83
84 SetResourceState(resource::ResourceState::LOADEDCPU);
85}

96void Texture::LoadPlaceholderTexture() {
97 // generate a raw array of pixeles and upload directly to opengl
98 // generate a checkerdoabr purple and black pattern 32x32
99 constexpr int size = 32;
100 constexpr int channels = 4;
101 unsigned char pixels[size * size * channels];
102 for (int y = 0; y < size; ++y) {
103 for (int x = 0; x < size; ++x) {
104 int index = (y * size + x) * channels;
105 if (((x / 4) + (y / 4)) % 2 == 0) {
106 // purple
107 pixels[index + 0] = 255; // R
108 pixels[index + 1] = 0; // G
109 pixels[index + 2] = 255; // B
110 pixels[index + 3] = 255; // A
111 } else {
112 // black
113 pixels[index + 0] = 0; // R
114 pixels[index + 1] = 0; // G
115 pixels[index + 2] = 0; // B
116 pixels[index + 3] = 255; // A
117 }
118 }
119 }
120 m_width = size;
121 m_height = size;
122 m_channels = channels;
123
124 // Create OpenGL texture
125 glActiveTexture(GL_TEXTURE0);
126 glGenTextures(1, const_cast<GLuint*>(&m_textureId));
127 glBindTexture(GL_TEXTURE_2D, m_textureId);
128
129 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
130
131 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, m_width, m_height, 0, GL_RGBA, GL_UNSIGNED_BYTE, reinterpret_cast<void*>(pixels));
132
133 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
134 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
135 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
136 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
137
138 glBindTexture(GL_TEXTURE_2D, 0);
139
140 // No CPU pixel data to free since we generated it here
141}

143void Texture::CreateOpenGLTexture() {
144 PROFILE_ZONE;
145 if (m_textureId) {
146 TOAST_ERROR("Tried to generate a texture but it was already generated!?!?!?");
147 return;
148 }
149
150 SetResourceState(resource::ResourceState::UPLOADING);
151
152 if (m_pixels == nullptr) {
153 SetResourceState(resource::ResourceState::FAILED);
154 TOAST_ERROR("Trying to create OpenGL texture but no pixel data is available!?!");
155 LoadPlaceholderTexture();
156 return;
157 }
158
159 // Determine appropriate GL formats
160 GLenum format = GL_RGBA;
161 GLenum internalFormat = GL_RGBA8;
162 switch (m_channels) {
163 case 1:
164 format = GL_RED;
165 internalFormat = GL_R8;
166 break; // grayscale
167 case 2:
168 format = GL_RG;
169 internalFormat = GL_RG8;
170 break; // treat as RG
171 case 3:
172 format = GL_RGB;
173 internalFormat = GL_RGB8;
174 break; // RGB
175 case 4:
176 format = GL_RGBA;
177 internalFormat = GL_RGBA8;
178 break; // RGBA
179 default: {
180 TOAST_ERROR("Unsupported channel count {0} for texture: {1}", m_channels, m_path);
181 SetResourceState(resource::ResourceState::FAILED);
182 stbi_image_free(m_pixels);
183 m_pixels = nullptr;
184 LoadPlaceholderTexture();
185 return;
186 }
187 }
188
189 glActiveTexture(GL_TEXTURE0);
190 glGenTextures(1, const_cast<GLuint*>(&m_textureId));
191 glBindTexture(GL_TEXTURE_2D, m_textureId);
192
193 // Ensure tight packing for arbitrary widths
194 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
195
196 glTexImage2D(GL_TEXTURE_2D, 0, internalFormat, m_width, m_height, 0, format, GL_UNSIGNED_BYTE, reinterpret_cast<void*>(m_pixels));
197
198 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
199 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
200 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
201 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
202
203 glBindTexture(GL_TEXTURE_2D, 0);
204
205 // Release CPU pixel data now that it's on the GPU
206 if (m_pixels) {
207 stbi_image_free(m_pixels);
208 m_pixels = nullptr;
209 }
210
211 SetResourceState(resource::ResourceState::UPLOADEDGPU);
212}
