27void Collider::CalculatePoints() {
28 // We require at leats 3 points to make a mesh
29 if (m.points.size() < 3) {
30 return;
31 }
32
33 // update world position data
34 glm::vec2 world_position = static_cast<toast::Actor*>(parent())->transform()->worldPosition();
35 data.worldPosition = world_position;
36
37 for (auto* c : m.convexShapes) {
38 delete c;
39 }
40 m.convexShapes.clear();
41
42 // simple_meshes are a list of points marked true if concave
43 using simple_mesh = std::list<std::pair<glm::vec2, bool>>;
44 simple_mesh start_mesh;
45
46 // Figure out which points are concave
47 int sign = ShoelaceArea(m.points) >= 0 ? 1 : -1;
48 std::list<std::list<glm::vec2>::iterator> concave_points;
49 for (auto it = m.points.begin(); it != m.points.end(); ++it) {
50 auto it_prev = it != m.points.begin() ? std::prev(it) : std::prev(m.points.end());
51 auto it_next = std::next(it) != m.points.end() ? std::next(it) : m.points.begin();
52
53 glm::vec2 curr = *it;
54 glm::vec2 prev = *it_prev;
55 glm::vec2 next = *it_next;
56
57 glm::mat2 mat = { curr - prev, next - curr };
58 float det = sign * glm::determinant(mat);
59
60 // we mark the point as concave if the determinant is negative
61 std::pair<glm::vec2, bool> point = { curr, det < 0.0f };
62 start_mesh.push_back(point);
63 }
64
65 // Create convex meshes
66 std::list<simple_mesh> meshes_list = { start_mesh };
67
68 // for every mesh, if we find a point marked as concave, we will divide the
69 // mesh into two by creating a new point from the intersection of the direction of
70 // the vector created by normalize((concave_pt - prev) + (concave_pt - next)) and the
71 // line of the mesh it intersects with, if it exactly happened to be another point
72 // rather than a line we will pick that point instead of creating a new one at the
73 // intersection point
74 // This will mark that point as not concave since its not concave anymore
75 //
76 // We will run this recursivelly until there are no points marked as concave
77
78 // Recompute concavity flags for a mesh
79 auto recompute_concave = [&](simple_mesh& mesh) {
80 for (auto it = mesh.begin(); it != mesh.end(); ++it) {
81 auto it_prev = it != mesh.begin() ? std::prev(it) : std::prev(mesh.end());
82 auto it_next = std::next(it) != mesh.end() ? std::next(it) : mesh.begin();
83
84 glm::vec2 curr = it->first;
85 glm::vec2 prev = it_prev->first;
86 glm::vec2 next = it_next->first;
87
88 glm::mat2 mat = { curr - prev, next - curr };
89 float det = sign * glm::determinant(mat);
90 it->second = det < 0.0f; // concave flag
91 }
92 };
93
94 const float epsilon = 1e-5f;
95 bool changed = true;
96 while (changed) {
97 changed = false;
98
99 // Iterate over each mesh we currently have
100 for (auto mesh_it = meshes_list.begin(); mesh_it != meshes_list.end();) {
101 auto& mesh = *mesh_it;
102
103 // Find the first concave point in this mesh
104 auto concave_it = std::ranges::find_if(mesh, [](const auto& p) {
105 return p.second;
106 });
107
108 // If no concave points, skip to the next mesh
109 if (concave_it == mesh.end()) {
110 ++mesh_it;
111 continue;
112 }
113
114 // Find concave point and its neighbors
115 glm::vec2 concave_pt = concave_it->first;
116 auto prev_it = concave_it != mesh.begin() ? std::prev(concave_it) : std::prev(mesh.end());
117 auto next_it = std::next(concave_it) != mesh.end() ? std::next(concave_it) : mesh.begin();
118
119 // Direction is the bisector of the two vectors
120 glm::vec2 dir = (concave_pt - prev_it->first) + (concave_pt - next_it->first);
121 float dir_len = glm::length(dir);
122 if (dir_len > epsilon) {
123 dir = glm::normalize(dir);
124 } else {
125 // perpendicular direction to prev->concave if thedirection doesn't make sense
126 glm::vec2 edge = concave_pt - prev_it->first;
127 float edge_len = glm::length(edge);
128 if (edge_len > epsilon) {
129 dir = glm::normalize(glm::vec2(-edge.y, edge.x));
130 } else {
131 dir = glm::vec2(1.0f, 0.0f); // safe fallback
132 }
133 }
134
135 // Track the closest intersection along direction
136 float best_t = std::numeric_limits<float>::max();
137 glm::vec2 best_point = concave_pt;
138 auto best_edge_start = mesh.end();
139
140 // Test intersection with every edge that does NOT include the concave point
141 for (auto it = mesh.begin(); it != mesh.end(); ++it) {
142 auto it_next = std::next(it) != mesh.end() ? std::next(it) : mesh.begin();
143
144 // Skip edges incident to the concave point
145 if (it == concave_it || it_next == concave_it) {
146 continue;
147 }
148
149 glm::vec2 q = it->first; // edge start
150 glm::vec2 s = it_next->first - q; // edge direction
151
152 float denom = glm::determinant(glm::mat2 { dir, s });
153 if (std::abs(denom) < epsilon) {
154 continue; // Parallel, no intersection
155 }
156
157 // Solve for intersection concave_pt + t*dir = q + u*s
158 float t = glm::determinant(glm::mat2 { q - concave_pt, s }) / denom;
159 float u = glm::determinant(glm::mat2 { q - concave_pt, dir }) / denom;
160
161 // Accept only forward intersections and within the edge length
162 if (t > epsilon && u >= -epsilon && u <= 1.0f + epsilon) {
163 if (t < best_t) {
164 best_t = t;
165 best_point = concave_pt + dir * t;
166 best_edge_start = it;
167 }
168 }
169 }
170
171 // If no intersection found, mark concave resolved and continue
172 if (best_edge_start == mesh.end()) {
173 concave_it->second = false;
174 ++mesh_it;
175 continue;
176 }
177
178 auto best_edge_end = std::next(best_edge_start) != mesh.end() ? std::next(best_edge_start) : mesh.begin();
179
180 // If the intersection is very close to an existing vertex, snap to it
181 if (glm::length(best_point - best_edge_start->first) < epsilon) {
182 best_point = best_edge_start->first;
183 } else if (glm::length(best_point - best_edge_end->first) < epsilon) {
184 best_point = best_edge_end->first;
185 }
186
187 // This new point will be shared by the two new meshes and is not concave
188 std::pair<glm::vec2, bool> new_point = { best_point, false };
189
190 // Build two new meshes by walking the old mesh in two directions from the concave point
191 simple_mesh mesh_a;
192 simple_mesh mesh_b;
193
194 // mesh_a: from concave -> ... -> best_edge_start -> new_point
195 for (auto it = concave_it;; it = (std::next(it) != mesh.end() ? std::next(it) : mesh.begin())) {
196 mesh_a.push_back(*it);
197 if (it == best_edge_start) {
198 break;
199 }
200 }
201 mesh_a.push_back(new_point);
202
203 // mesh_b: new_point -> best_edge_end -> ... -> concave
204 mesh_b.push_back(new_point);
205 for (auto it = best_edge_end;; it = (std::next(it) != mesh.end() ? std::next(it) : mesh.begin())) {
206 mesh_b.push_back(*it);
207 if (it == concave_it) {
208 break;
209 }
210 }
211
212 // Re-evaluate concavity after splitting
213 recompute_concave(mesh_a);
214 recompute_concave(mesh_b);
215
216 // Replace the current mesh with the two new meshes
217 mesh_it = meshes_list.erase(mesh_it);
218 mesh_it = meshes_list.insert(mesh_it, mesh_b);
219 mesh_it = meshes_list.insert(mesh_it, mesh_a);
220
221 changed = true;
222 break; // restart scanning meshes after modification
223 }
224 }
225
226 // Finally, create convex colliders for every convex mesh we produced
227 for (const auto& points : meshes_list) {
228 m.convexShapes.emplace_back(new ConvexCollider(points, data));
229 }
230}
