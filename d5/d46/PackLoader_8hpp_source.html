36inline uint64_t fnv1a_hash64(const std::u8string& s) {
37 PROFILE_ZONE;
38 uint64_t hash = 14695981039346656037ULL;
39 for (unsigned char c : s) {
40 hash ^= static_cast<uint64_t>(c);
41 hash *= 1099511628211ULL;
42 }
43 return hash;

128 bool ReadFile(const std::string& raw_path, std::vector<uint8_t>& out) {
129 PROFILE_ZONE;
