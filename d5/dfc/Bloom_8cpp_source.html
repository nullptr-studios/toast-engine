58void Bloom::Execute(Framebuffer* inputFBO, Framebuffer* outputFBO) {
59 if (!m_extractShader || !m_blurShader || !m_compositeShader || !inputFBO || !outputFBO) {
60 return;
61 }
62 EnsureBuffers(outputFBO->Width(), outputFBO->Height());
63
64 if (!m_extractFbo || !m_blurFboA || !m_blurFboB) {
65 return;
66 }
67
68 const float blendIntensity = m_intensity * GetBlend();
69 if (blendIntensity <= 0.0f) {
70 inputFBO->BlitTo(outputFBO, GL_COLOR_BUFFER_BIT, GL_LINEAR);
71 return;
72 }
73
74 // 1) Extract HDR highlights above threshold (values > 1 are kept and boosted naturally).
75 m_extractFbo->bindDraw();
76 glViewport(0, 0, m_extractFbo->Width(), m_extractFbo->Height());
77 glScissor(0, 0, m_extractFbo->Width(), m_extractFbo->Height());
78 glDisable(GL_BLEND);
79 glDisable(GL_DEPTH_TEST);
80 glDisable(GL_CULL_FACE);
81 glDepthMask(GL_FALSE);
82 glBindTextureUnit(0, inputFBO->GetColorTexture(0));
83 m_extractShader->Use();
84 m_extractShader->SetSampler("uInputTex", 0);
85 m_extractShader->Set("uThreshold", m_threshold);
86 m_extractShader->Set("uKnee", m_knee);
87 renderer::IRendererBase::GetInstance()->DrawScreenQuad(false, false);
88
89 // 2) Separable Gaussian blur using convolution kernel.
90 Framebuffer* srcBlurFbo = m_extractFbo;
91 Framebuffer* dstBlurFbo = m_blurFboA;
92 const int blurPasses = std::max(1, m_iterations * 2);
93 for (int pass = 0; pass < blurPasses; ++pass) {
94 const bool horizontal = (pass % 2) == 0;
95 dstBlurFbo->bindDraw();
96 glViewport(0, 0, dstBlurFbo->Width(), dstBlurFbo->Height());
97 glScissor(0, 0, dstBlurFbo->Width(), dstBlurFbo->Height());
98 glBindTextureUnit(0, srcBlurFbo->GetColorTexture(0));
99 m_blurShader->Use();
100 m_blurShader->SetSampler("uInputTex", 0);
101 m_blurShader->Set("uHorizontal", horizontal ? 1 : 0);
102 m_blurShader->Set("uSigma", m_blurSigma);
103 renderer::IRendererBase::GetInstance()->DrawScreenQuad(false, false);
104
105 srcBlurFbo = dstBlurFbo;
106 dstBlurFbo = (dstBlurFbo == m_blurFboA) ? m_blurFboB : m_blurFboA;
107 }
108
109 // 3) Composite original + blurred bloom into output.
110 outputFBO->bindDraw();
111 glViewport(0, 0, outputFBO->Width(), outputFBO->Height());
112 glScissor(0, 0, outputFBO->Width(), outputFBO->Height());
113
114 glDisable(GL_BLEND);
115 glDisable(GL_DEPTH_TEST);
116 glDisable(GL_CULL_FACE);
117 glDepthMask(GL_FALSE);
118
119 glBindTextureUnit(0, inputFBO->GetColorTexture(0));
120 glBindTextureUnit(1, srcBlurFbo->GetColorTexture(0));
121 m_compositeShader->Use();
122 m_compositeShader->SetSampler("uInputTex", 0);
123 m_compositeShader->SetSampler("uBloomTex", 1);
124 m_compositeShader->Set("uIntensity", blendIntensity);
125
126 renderer::IRendererBase::GetInstance()->DrawScreenQuad(false, false);
127
128 glBindTextureUnit(1, 0);
129 glBindTextureUnit(0, 0);
130 Framebuffer::unbind();
131}
