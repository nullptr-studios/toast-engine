17glm::vec3 ResolveLightDirection(GlobalLight* light) {
18 if (!light || !light->transform()) {
19 return glm::normalize(glm::vec3(-0.35f, -1.0f, -0.25f));
20 }
21 const glm::vec3 forward = light->transform()->GetFrontVector();
22 const glm::vec3 candidate = -forward;
23 const float len2 = glm::dot(candidate, candidate);
24 if (len2 <= 1e-6f) {
25 return glm::normalize(glm::vec3(-0.35f, -1.0f, -0.25f));
26 }
27 return candidate / glm::sqrt(len2);
28}

30glm::vec3 NormalizeSafe(const glm::vec3& direction) {
31 const float len2 = glm::dot(direction, direction);
32 if (len2 <= 1e-6f) {
33 return glm::normalize(glm::vec3(-0.35f, -1.0f, -0.25f));
34 }
35 return direction / glm::sqrt(len2);
36}
