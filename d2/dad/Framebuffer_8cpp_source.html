30void Framebuffer::Build() {
31 if (m_specs.width <= 0 || m_specs.height <= 0) {
32 throw ToastException("Framebuffer dimensions must be > 0");
33 }
34
35 bind();
36
37 // Clean up previous attachments
38 for (GLuint t : m_colorTextures) {
39 if (t) {
40 glDeleteTextures(1, &t);
41 }
42 }
43 m_colorTextures.clear();
44
45 if (m_depthTexture) {
46 glDeleteTextures(1, &m_depthTexture);
47 m_depthTexture = 0;
48 }
49 if (m_depthRbo) {
50 glDeleteRenderbuffers(1, &m_depthRbo);
51 m_depthRbo = 0;
52 }
53
54 // Create color attachments
55 GLenum texTarget = m_specs.multisample ? GL_TEXTURE_2D_MULTISAMPLE : GL_TEXTURE_2D;
56 m_colorTextures.resize(m_colorSpecs.size(), 0);
57
58 for (size_t i = 0; i < m_colorSpecs.size(); ++i) {
59 GLuint tex = 0;
60 glGenTextures(1, &tex);
61 if (!tex) {
62 throw ToastException("Failed to create color texture for FBO");
63 }
64
65 glBindTexture(texTarget, tex);
66 if (m_specs.multisample) {
67 glTexImage2DMultisample(texTarget, m_specs.samples, m_colorSpecs[i].internalFormat, m_specs.width, m_specs.height, GL_TRUE);
68 } else {
69 glTexImage2D(
70 texTarget, 0, m_colorSpecs[i].internalFormat, m_specs.width, m_specs.height, 0, m_colorSpecs[i].format, m_colorSpecs[i].type, nullptr
71 );
72
73 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
74 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
75 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
76 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
77 }
78
79 // attach
80 glFramebufferTexture2D(GL_FRAMEBUFFER, static_cast<GLenum>(GL_COLOR_ATTACHMENT0 + i), texTarget, tex, 0);
81
82 m_colorTextures[i] = tex;
83 }
84
85 // Depth attachment
86 if (m_depthMode == DepthMode::Renderbuffer) {
87 glGenRenderbuffers(1, &m_depthRbo);
88 glBindRenderbuffer(GL_RENDERBUFFER, m_depthRbo);
89 if (m_specs.multisample) {
90 glRenderbufferStorageMultisample(GL_RENDERBUFFER, m_specs.samples, m_depthInternalFormat, m_specs.width, m_specs.height);
91 } else {
92 glRenderbufferStorage(GL_RENDERBUFFER, m_depthInternalFormat, m_specs.width, m_specs.height);
93 }
94 glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_DEPTH_STENCIL_ATTACHMENT, GL_RENDERBUFFER, m_depthRbo);
95 // unbind rbo
96 glBindRenderbuffer(GL_RENDERBUFFER, 0);
97 } else if (m_depthMode == DepthMode::Texture) {
98 GLuint tex = 0;
99 glGenTextures(1, &tex);
100 GLenum texTargetDepth = m_specs.multisample ? GL_TEXTURE_2D_MULTISAMPLE : GL_TEXTURE_2D;
101 glBindTexture(texTargetDepth, tex);
102 if (m_specs.multisample) {
103 glTexImage2DMultisample(texTargetDepth, m_specs.samples, m_depthInternalFormat, m_specs.width, m_specs.height, GL_TRUE);
104 } else {
105 GLenum format = (m_depthInternalFormat == GL_DEPTH24_STENCIL8) ? GL_DEPTH_STENCIL : GL_DEPTH_COMPONENT;
106 GLenum type = (m_depthInternalFormat == GL_DEPTH24_STENCIL8) ? GL_UNSIGNED_INT_24_8 : GL_FLOAT;
107 glTexImage2D(texTargetDepth, 0, m_depthInternalFormat, m_specs.width, m_specs.height, 0, format, type, nullptr);
108 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
109 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
110 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
111 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
112 }
113
114 // attach as depth or depth-stencil depending on format
115 if (m_depthInternalFormat == GL_DEPTH24_STENCIL8) {
116 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_DEPTH_STENCIL_ATTACHMENT, texTargetDepth, tex, 0);
117 } else {
118 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, texTargetDepth, tex, 0);
119 }
120
121 m_depthTexture = tex;
122 glBindTexture(texTargetDepth, 0);
123 }
124
125 if (!m_colorSpecs.empty()) {
126 std::vector<GLenum> draws;
127 draws.reserve(m_colorSpecs.size());
128 for (size_t i = 0; i < m_colorSpecs.size(); ++i) {
129 draws.push_back(GL_COLOR_ATTACHMENT0 + static_cast<GLenum>(i));
130 }
131 glDrawBuffers(static_cast<GLsizei>(draws.size()), draws.data());
132 } else {
133 // no color attachments -> disable draw buffer
134 glDrawBuffer(GL_NONE);
135 glReadBuffer(GL_NONE);
136 }
137
138 // Check complete
139 GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
140 if (status != GL_FRAMEBUFFER_COMPLETE) {
141 std::string msg = "Framebuffer is not complete: 0x" + intToHex(status);
142 unbind();
143 throw ToastException(msg);
144 }
145
146 // restore binding
147 unbind();
148}

150void Framebuffer::BlitTo(const Framebuffer* target, GLbitfield mask, GLenum filter, unsigned srcAttachment, unsigned dstAttachment) const {
151 // Preserve previous framebuffer bindings and buffer enums
152 GLint prevReadFbo = 0, prevDrawFbo = 0;
153 GLint prevReadBuf = 0, prevDrawBuf = 0;
154 glGetIntegerv(GL_READ_FRAMEBUFFER_BINDING, &prevReadFbo);
155 glGetIntegerv(GL_DRAW_FRAMEBUFFER_BINDING, &prevDrawFbo);
156 glGetIntegerv(GL_READ_BUFFER, &prevReadBuf);
157 glGetIntegerv(GL_DRAW_BUFFER, &prevDrawBuf);
158
159 // bind read and draw
160 glBindFramebuffer(GL_READ_FRAMEBUFFER, m_fbo);
161 glBindFramebuffer(GL_DRAW_FRAMEBUFFER, target ? target->m_fbo : 0);
162
163 // If we're blitting color bits, select appropriate color attachments
164 if (mask & GL_COLOR_BUFFER_BIT) {
165 // validate indices
166 if (srcAttachment >= m_colorTextures.size()) {
167 TOAST_WARN("BlitTo: srcAttachment {0} out of range, clamping to 0", srcAttachment);
168 srcAttachment = 0;
169 }
170 if (target && dstAttachment >= target->m_colorTextures.size()) {
171 TOAST_WARN("BlitTo: dstAttachment {0} out of range for target, clamping to 0", dstAttachment);
172 dstAttachment = 0;
173 }
174
175 glReadBuffer(static_cast<GLenum>(GL_COLOR_ATTACHMENT0 + srcAttachment));
176 if (target) {
177 glDrawBuffer(static_cast<GLenum>(GL_COLOR_ATTACHMENT0 + dstAttachment));
178 } else {
179 // if target is null (default framebuffer) map to back buffer
180 glDrawBuffer(GL_BACK);
181 }
182 }
183
184 glBlitFramebuffer(
185 0,
186 0,
187 m_specs.width,
188 m_specs.height,
189 0,
190 0,
191 target ? target->m_specs.width : m_specs.width,
192 target ? target->m_specs.height : m_specs.height,
193 mask,
194 filter
195 );
196
197 // restore previous read/draw buffers and framebuffer bindings
198 if (mask & GL_COLOR_BUFFER_BIT) {
199 glReadBuffer(static_cast<GLenum>(prevReadBuf));
200 glDrawBuffer(static_cast<GLenum>(prevDrawBuf));
201 }
202
203 glBindFramebuffer(GL_READ_FRAMEBUFFER, prevReadFbo);
204 glBindFramebuffer(GL_DRAW_FRAMEBUFFER, prevDrawFbo);
205}

207uint32_t Framebuffer::ReadPixel(unsigned colorAttachmentIndex, int x, int y) const {
208 if (colorAttachmentIndex >= m_colorTextures.size()) {
209 throw ToastException("colorAttachmentIndex out of range");
210 }
211 if (m_specs.multisample) {
212 throw ToastException("ReadPixel: framebuffer is multisampled; blit to resolved FBO first");
213 }
214
215 GLint prev = 0;
216 glGetIntegerv(GL_READ_FRAMEBUFFER_BINDING, &prev);
217 glBindFramebuffer(GL_READ_FRAMEBUFFER, m_fbo);
218 glReadBuffer(static_cast<GLenum>(GL_COLOR_ATTACHMENT0 + colorAttachmentIndex));
219
220 unsigned char pixel[4] = { 0, 0, 0, 0 };
221 glReadPixels(x, y, 1, 1, GL_RGBA, GL_UNSIGNED_BYTE, pixel);
222
223 // restore
224 glBindFramebuffer(GL_READ_FRAMEBUFFER, prev);
225
226 uint32_t out = static_cast<uint32_t>(pixel[0]) | (static_cast<uint32_t>(pixel[1]) << 8) | (static_cast<uint32_t>(pixel[2]) << 16) |
227 (static_cast<uint32_t>(pixel[3]) << 24);
228 return out;
229}

233 for (GLuint t : m_colorTextures) {
