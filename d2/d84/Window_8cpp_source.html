17unsigned DecodeUTF8Codepoint(const char* text, int& consumed) {
18 const unsigned char c0 = static_cast<unsigned char>(text[0]);
19 if (c0 == 0) {
20 consumed = 0;
21 return 0;
22 }
23 if ((c0 & 0x80u) == 0) {
24 consumed = 1;
25 return c0;
26 }
27 if ((c0 & 0xE0u) == 0xC0u) {
28 consumed = 2;
29 return ((c0 & 0x1Fu) << 6) | (static_cast<unsigned char>(text[1]) & 0x3Fu);
30 }
31 if ((c0 & 0xF0u) == 0xE0u) {
32 consumed = 3;
33 return ((c0 & 0x0Fu) << 12) | ((static_cast<unsigned char>(text[1]) & 0x3Fu) << 6) | (static_cast<unsigned char>(text[2]) & 0x3Fu);
34 }
35 consumed = 4;
36 return ((c0 & 0x07u) << 18) | ((static_cast<unsigned char>(text[1]) & 0x3Fu) << 12) | ((static_cast<unsigned char>(text[2]) & 0x3Fu) << 6) |
37 (static_cast<unsigned char>(text[3]) & 0x3Fu);
38}

139void Window::PollEventsOnly() {
140 SDL_Event ev;
141 while (SDL_PollEvent(&ev)) {
142#ifdef TOAST_EDITOR
143#endif
144#ifdef TOAST_EDITOR
145 ImGui_ImplSDL3_ProcessEvent(&ev);
146#endif
147 switch (ev.type) {
148 case SDL_EVENT_QUIT: m_shouldClose = true; break;
149 case SDL_EVENT_WINDOW_CLOSE_REQUESTED:
150 if (ev.window.windowID == SDL_GetWindowID(m_sdlWindow)) {
151 m_shouldClose = true;
152 }
153 break;
154 case SDL_EVENT_KEY_DOWN:
155 case SDL_EVENT_KEY_UP: {
156 const int action = (ev.type == SDL_EVENT_KEY_UP) ? event::WINDOW_INPUT_RELEASED
157 : (ev.key.repeat ? event::WINDOW_INPUT_REPEATED : event::WINDOW_INPUT_PRESSED);
158 event::WindowKey::Callback(static_cast<int>(ev.key.key), static_cast<int>(ev.key.scancode), action, static_cast<int>(SDL_GetModState()));
159 break;
160 }
161 case SDL_EVENT_TEXT_INPUT: {
162 const char* text = ev.text.text;
163 int i = 0;
164 while (text[i] != '\0') {
165 int consumed = 0;
166 const unsigned cp = DecodeUTF8Codepoint(text + i, consumed);
167 if (consumed <= 0) {
168 break;
169 }
170 event::WindowChar::Callback(cp);
171 i += consumed;
172 }
173 break;
174 }
175 case SDL_EVENT_MOUSE_MOTION: event::WindowMousePosition::Callback(ev.motion.x, ev.motion.y); break;
176 case SDL_EVENT_MOUSE_BUTTON_DOWN:
177 case SDL_EVENT_MOUSE_BUTTON_UP: {
178 const int action = (ev.type == SDL_EVENT_MOUSE_BUTTON_DOWN) ? event::WINDOW_INPUT_PRESSED : event::WINDOW_INPUT_RELEASED;
179 event::WindowMouseButton::Callback(static_cast<int>(ev.button.button), action, static_cast<int>(SDL_GetModState()));
180 break;
181 }
182 case SDL_EVENT_MOUSE_WHEEL: event::WindowMouseScroll::Callback(ev.wheel.x, ev.wheel.y); break;
183 case SDL_EVENT_GAMEPAD_ADDED:
184 event::WindowInputDevice::Callback(static_cast<int>(ev.gdevice.which), event::WINDOW_INPUT_DEVICE_CONNECTED);
185 break;
186 case SDL_EVENT_GAMEPAD_REMOVED:
187 event::WindowInputDevice::Callback(static_cast<int>(ev.gdevice.which), event::WINDOW_INPUT_DEVICE_DISCONNECTED);
188 break;
189 case SDL_EVENT_DROP_FILE: {
190 std::vector<std::string> files;
191 files.emplace_back(ev.drop.data ? ev.drop.data : "");
192 if (ev.drop.data) {
193 SDL_free(const_cast<char*>(ev.drop.data));
194 }
195 event::WindowDrop::Callback(std::move(files));
196 break;
197 }
198 case SDL_EVENT_WINDOW_PIXEL_SIZE_CHANGED:
199 if (ev.window.windowID == SDL_GetWindowID(m_sdlWindow)) {
200 event::WindowResize::Callback(ev.window.data1, ev.window.data2);
201 }
202 break;
203 default: break;
204 }
205 }
206}

214void Window::SetDisplayMode(DisplayMode modeScreen) {
215 if (modeScreen == m_currentDisplayMode) {
216 return;
217 }
218
219 if (modeScreen == DisplayMode::WINDOWED) {
220 TOAST_INFO("Switching to WINDOWED mode");
221 SDL_SetWindowFullscreen(m_sdlWindow, false);
222 SDL_SetWindowPosition(m_sdlWindow, m_windowedPos.x, m_windowedPos.y);
223 SDL_SetWindowSize(m_sdlWindow, static_cast<int>(m_windowedSize.x), static_cast<int>(m_windowedSize.y));
224 } else if (modeScreen == DisplayMode::FULLSCREEN) {
225 TOAST_INFO("Switching to FULLSCREEN mode");
226 if (m_currentDisplayMode == DisplayMode::WINDOWED) {
227 SDL_GetWindowPosition(m_sdlWindow, &m_windowedPos.x, &m_windowedPos.y);
228 int w = 0;
229 int h = 0;
230 SDL_GetWindowSize(m_sdlWindow, &w, &h);
231 m_windowedSize = { static_cast<unsigned>(w), static_cast<unsigned>(h) };
232 }
233 SDL_SetWindowFullscreen(m_sdlWindow, true);
234 }
235
236 m_currentDisplayMode = modeScreen;
