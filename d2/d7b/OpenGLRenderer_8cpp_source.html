130bool IsInsideDirectionalShadowBounds(IRenderable* renderable, const glm::mat4& shadowMatrix) {
131 if (!renderable) {
132 return false;
133 }
134
135 const glm::mat4 world = renderable->GetWorldMatrix();
136 const glm::vec3 worldPos = glm::vec3(world[3]);
137 const glm::vec4 clip = shadowMatrix * glm::vec4(worldPos, 1.0f);
138 if (!std::isfinite(clip.x) || !std::isfinite(clip.y) || !std::isfinite(clip.z) || !std::isfinite(clip.w)) {
139 return false;
140 }
141
142 if (std::abs(clip.w) <= 1e-6f) {
143 return false;
144 }
145
146 const glm::vec3 ndc = glm::vec3(clip) / clip.w;
147 constexpr float kShadowCullMargin = 0.03f;
148 return ndc.x >= -1.0f - kShadowCullMargin && ndc.x <= 1.0f + kShadowCullMargin &&
149 ndc.y >= -1.0f - kShadowCullMargin && ndc.y <= 1.0f + kShadowCullMargin &&
150 ndc.z >= -1.0f - kShadowCullMargin && ndc.z <= 1.0f + kShadowCullMargin;
151}

511void OpenGLRenderer::Render() {
512 InvalidateCachedGlState();
513 if (toast::Window::GetInstance()->IsMinimized()) {
514 return;
515 }
516 PROFILE_ZONE;
517
518#ifdef TRACY_ENABLE
519 TracyGpuZone("Main Render");
520#endif
521
522 // Update view matrix only when a camera is active
523 if (m_activeCamera) {
524 SetViewMatrix(m_activeCamera->GetViewMatrix());
525 }
526
527 // Compute combined matrix once
528 m_multipliedMatrix = m_projectionMatrix * m_viewMatrix;
529
530 // Extract frustum planes for culling
531 OclussionVolume::extractFrustumPlanesNormalized(m_multipliedMatrix, m_frustumPlanes);
532
533 // Rebuild enabled renderables only when scene membership or enable-state changes.
534 if (m_renderablesSortDirty) {
535 m.combinedRenderables.clear();
536 m.combinedRenderables.reserve(m_renderables.size());
537 for (auto* r : m_renderables) {
538 if (r && m_disabledRenderables.find(r) == m_disabledRenderables.end()) {
539 m.combinedRenderables.push_back(r);
540 }
541 }
542
543
544 // Depth can change every frame, so keep sort per-frame but only for enabled list.
545 // HACK: FOR OPTIMIZATION PURPOSES WE JUST SORT WHEN ADDING OBJECTS
546 if (m.combinedRenderables.size() > 1) {
547 std::stable_sort(m.combinedRenderables.begin(), m.combinedRenderables.end(), [](IRenderable* a, IRenderable* b) {
548 const int priority_a = a->GetGeometrySortPriority();
549 const int priority_b = b->GetGeometrySortPriority();
550 if (priority_a != priority_b) {
551 return priority_a < priority_b;
552 }
553 return a->GetDepth() < b->GetDepth();
554 });
555 }
556 m_renderablesSortDirty = false;
557 }
558
559 if (m_transparentsSortDirty) {
560 m.combinedTransparents.clear();
561 m.combinedTransparents.reserve(m_transparentRenderables.size());
562 for (auto* r : m_transparentRenderables) {
563 if (r && m_disabledTransparents.find(r) == m_disabledTransparents.end()) {
564 m.combinedTransparents.push_back(r);
565 }
566 }
567 m_transparentsSortDirty = false;
568 }
569
570 if (m.combinedTransparents.size() > 1) {
571 std::stable_sort(m.combinedTransparents.begin(), m.combinedTransparents.end(), [&](IRenderable* a, IRenderable* b) {
572 const int priority_a = a->GetTransparentSortPriority();
573 const int priority_b = b->GetTransparentSortPriority();
574 if (priority_a != priority_b) {
575 return priority_a < priority_b;
576 }
577
578 const float depthA = a->GetTransparentSortDepth(m_viewMatrix);
579 const float depthB = b->GetTransparentSortDepth(m_viewMatrix);
580 if (std::abs(depthA - depthB) > 1e-4f) {
581 // View-space: smaller Z is farther, so draw farther first for blending.
582 return depthA < depthB;
583 }
584 return a < b;
585 });
586
587 }
588
589 if (m_watersSortDirty) {
590 m.combinedWaters.clear();
591 m.combinedWaters.reserve(m_waterRenderables.size());
592 for (auto* r : m_waterRenderables) {
593 if (r && m_disabledWaters.find(r) == m_disabledWaters.end()) {
594 m.combinedWaters.push_back(r);
595 }
596 }
597 m_watersSortDirty = false;
598 }
599
600 if (m.combinedWaters.size() > 1) {
601 std::stable_sort(m.combinedWaters.begin(), m.combinedWaters.end(), [&](IRenderable* a, IRenderable* b) {
602 const int priority_a = a->GetTransparentSortPriority();
603 const int priority_b = b->GetTransparentSortPriority();
604 if (priority_a != priority_b) {

737 TracyGpuZone("Occlusion Pass");

741 const glm::ivec2 occlusionResolution { m.occlusionFramebuffer->Width(), m.occlusionFramebuffer->Height() };
742 const GLuint dispatchX = static_cast<GLuint>((occlusionResolution.x + 15) / 16);
743 const GLuint dispatchY = static_cast<GLuint>((occlusionResolution.y + 15) / 16);
744 glViewport(0, 0, occlusionResolution.x, occlusionResolution.y);
745 glScissor(0, 0, occlusionResolution.x, occlusionResolution.y);
746
747 glClear(GL_COLOR_BUFFER_BIT);

752 r->OnRender(renderer::IRenderablePass::OCCLUSION, m_multipliedMatrix);

754 for (auto* r : m.combinedTransparents) {
755 r->OnRender(renderer::IRenderablePass::OCCLUSION, m_multipliedMatrix);
756 }
757
758 m.jfaInitComputeShader->Use();
759 glBindTextureUnit(0, m.occlusionFramebuffer->GetColorTexture(0));
760 glBindImageTexture(1, m.jfaTex, 0, GL_FALSE, 0, GL_WRITE_ONLY, GL_RG32F);
761 m.jfaInitComputeShader->Set("resolution", occlusionResolution);
762 glDispatchCompute(dispatchX, dispatchY, 1);
763 glMemoryBarrier(GL_SHADER_IMAGE_ACCESS_BARRIER_BIT | GL_TEXTURE_FETCH_BARRIER_BIT);
764
765 GLuint inputTex = m.jfaTex;
766 GLuint outputTex = m.pingPongTex;
767
768 const int maxDim = std::max(occlusionResolution.x, occlusionResolution.y);
769 int step = maxDim / 2;
770 m.jfaComputeShader->Use();
771 m.jfaComputeShader->Set("resolution", occlusionResolution);
772
773 while (step > 0) {
774 m.jfaComputeShader->Set("stepValue", step);
775
776 glBindTextureUnit(0, inputTex);
777 glBindImageTexture(1, outputTex, 0, GL_FALSE, 0, GL_WRITE_ONLY, GL_RG32F);
778
779 glDispatchCompute(dispatchX, dispatchY, 1);
780 glMemoryBarrier(GL_SHADER_IMAGE_ACCESS_BARRIER_BIT | GL_TEXTURE_FETCH_BARRIER_BIT);
781
782 std::swap(inputTex, outputTex);
783
784 step /= 2;
785 }
786
787 m.finalComputeShader->Use();
788 glBindTextureUnit(0, inputTex);
789 glBindImageTexture(1, m.sdfTex, 0, GL_FALSE, 0, GL_WRITE_ONLY, GL_R32F);
790 m.finalComputeShader->Set("resolution", occlusionResolution);
791 glDispatchCompute(dispatchX, dispatchY, 1);
792 glMemoryBarrier(GL_SHADER_IMAGE_ACCESS_BARRIER_BIT | GL_TEXTURE_FETCH_BARRIER_BIT);

1174void OpenGLRenderer::UpdateDirectionalShadowMatrix() {
1175 glm::vec3 cameraPos = glm::vec3(0.0f);
1176 if (m_activeCamera) {
1177 cameraPos = m_activeCamera->transform()->worldPosition();
1178 }
1179
1180 const glm::vec3 lightDir = GetGlobalLightDirection();
1181 const glm::mat4 viewProj = m_projectionMatrix * m_viewMatrix;
1182 const bool cameraMoved = glm::length(cameraPos - m.lastDirectionalCameraPos) > kDirectionalShadowSnapEpsilon;
1183 const bool lightChanged = glm::length(lightDir - m.lastDirectionalLightDir) > 1e-4f;
1184 const bool viewProjChanged = glm::length(glm::vec4(viewProj[0] - m.lastDirectionalViewProj[0])) > 1e-4f ||
1185 glm::length(glm::vec4(viewProj[1] - m.lastDirectionalViewProj[1])) > 1e-4f ||
1186 glm::length(glm::vec4(viewProj[2] - m.lastDirectionalViewProj[2])) > 1e-4f ||
1187 glm::length(glm::vec4(viewProj[3] - m.lastDirectionalViewProj[3])) > 1e-4f;
1188 if (!m.directionalShadowMatrixDirty && !cameraMoved && !lightChanged && !viewProjChanged) {
1189 return;
1190 }
1191
1192 float nearClip = 0.1f;
1193 float farClip = 1000.0f;
1194 {
1195 const float a = m_projectionMatrix[2][2];
1196 const float b = m_projectionMatrix[3][2];
1197 nearClip = std::max(0.01f, b / (a - 1.0f));
1198 farClip = std::max(nearClip + 1.0f, b / (a + 1.0f));
1199 }
1200
1201 const float projectionY = std::max(1e-4f, std::abs(m_projectionMatrix[1][1]));
1202 const float projectionX = std::max(1e-4f, std::abs(m_projectionMatrix[0][0]));
1203 const float fovY = 2.0f * std::atan(1.0f / projectionY);
1204 const float aspect = std::max(0.1f, projectionY / projectionX);
1205 const float directionalShadowDistance = std::max(16.0f, m_config.directionalShadowDistance);
1206 const float directionalShadowRange = std::max(8.0f, m_config.directionalShadowRange);
1207 const float coverageNear = nearClip;
1208 const float coverageFar = std::clamp(directionalShadowDistance, coverageNear + 12.0f, farClip);
1209
1210 const glm::mat4 invView = glm::inverse(m_viewMatrix);
1211 const glm::vec3 camPos = glm::vec3(invView[3]);
1212 const glm::vec3 camRight = glm::normalize(glm::vec3(invView[0]));
1213 const glm::vec3 camUp = glm::normalize(glm::vec3(invView[1]));
1214 const glm::vec3 camForward = glm::normalize(-glm::vec3(invView[2]));
1215
1216 auto buildFrustumCorners = [&](float zNear, float zFar) -> std::array<glm::vec3, 8> {
1217 const float tanHalfFovY = std::tan(fovY * 0.5f);
1218 const float nearHalfY = zNear * tanHalfFovY;
1219 const float nearHalfX = nearHalfY * aspect;
1220 const float farHalfY = zFar * tanHalfFovY;
1221 const float farHalfX = farHalfY * aspect;
1222
1223 const glm::vec3 nearCenter = camPos + camForward * zNear;
1224 const glm::vec3 farCenter = camPos + camForward * zFar;

1249 }

1252 const glm::mat4 lightView = glm::lookAt(lightPos, frustumCenter, up);

1270 const float centerY = (minLs.y + maxLs.y) * 0.5f;
1271 const float halfX = std::max((maxLs.x - minLs.x) * 0.5f, minHalfExtent);
1272 const float halfY = std::max((maxLs.y - minLs.y) * 0.5f, minHalfExtent);
1273 minLs.x = centerX - halfX;
1274 maxLs.x = centerX + halfX;
1275 minLs.y = centerY - halfY;
1276 maxLs.y = centerY + halfY;
1277
1278 const float resolution = static_cast<float>(std::max(1, shadowRes));
1279 const float extentX = std::max(maxLs.x - minLs.x, 1.0f);
1280 const float extentY = std::max(maxLs.y - minLs.y, 1.0f);
1281 const float texelX = extentX / resolution;
1282 const float texelY = extentY / resolution;

1286 centerLs.y = std::floor(centerLs.y / texelY) * texelY;

1288 minLs.x = centerLs.x - extentX * 0.5f;
1289 maxLs.x = centerLs.x + extentX * 0.5f;
1290 minLs.y = centerLs.y - extentY * 0.5f;
1291 maxLs.y = centerLs.y + extentY * 0.5f;

1295 float nearPlane = -maxLs.z - padZ;

1299 nearPlane = 0.01f;
1300 farPlane = nearPlane + 500.0f;

1303 return lightProj * lightView;
1304 };
