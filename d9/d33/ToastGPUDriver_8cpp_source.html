86static GLuint LoadShaderFromSource(GLenum shader_type, const char* source, const char* name) {
87 // Check that we have a valid GL context
88 if (!glfwGetCurrentContext()) {
89 GPU_FATAL("No GL context current when loading shader: " << name);
90 }
91
92 // Clear any pending GL errors
93 while (glGetError() != GL_NO_ERROR) { }
94
95 GLuint shader_id = glCreateShader(shader_type);
96 if (shader_id == 0) {
97 GLenum err = glGetError();
98 GPU_FATAL("glCreateShader failed for: " << name << "\n\tError: " << glErrorString(err));
99 }
100
101 glShaderSource(shader_id, 1, &source, nullptr);
102 glCompileShader(shader_id);
103
104 GLint compileStatus;
105 glGetShaderiv(shader_id, GL_COMPILE_STATUS, &compileStatus);
106 if (compileStatus == GL_FALSE) {
107 std::string log = GetShaderLog(shader_id);
108 glDeleteShader(shader_id);
109 GPU_FATAL("Unable to compile shader: " << name << "\n\tLog: " << log);
110 }
111 return shader_id;
112}

142void ToastGPUDriver::CreateTexture(uint32_t texture_id, ultralight::RefPtr<ultralight::Bitmap> bitmap) {
143 PROFILE_ZONE_C(tracy::Color::Orange);
144
145 // Ensure correct GL context is current
146 glfwMakeContextCurrent(context_->active_window());
147
148 CHECK_GL();
149
150 if (bitmap->IsEmpty()) {
151 CreateFBOTexture(texture_id, bitmap);
152 return;
153 }
154
155 CHECK_GL();
156
157 TextureEntry& entry = texture_map_[texture_id];
158 glGenTextures(1, &entry.tex_id);
159 glActiveTexture(GL_TEXTURE0);
160 glBindTexture(GL_TEXTURE_2D, entry.tex_id);
161
162 CHECK_GL();
163
164 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
165 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
166 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
167 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
168
169 CHECK_GL();
170
171 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
172 glPixelStorei(GL_UNPACK_ROW_LENGTH, bitmap->row_bytes() / bitmap->bpp());
173
174 CHECK_GL();
175
176 if (bitmap->format() == ultralight::BitmapFormat::A8_UNORM) {
177 const void* pixels = bitmap->LockPixels();
178 glTexImage2D(GL_TEXTURE_2D, 0, GL_R8, bitmap->width(), bitmap->height(), 0, GL_RED, GL_UNSIGNED_BYTE, pixels);
179 bitmap->UnlockPixels();
180
181 // A8 textures are used for font glyphs - don't generate mipmaps
182 // and use LINEAR filtering for smooth text
183 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
184 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
185
186 // The shader's fillGlyph() reads the glyph coverage from texture.r directly
187 // so we DON'T need any swizzle - just keep R8 as-is
188 // The shader handles the alpha multiplication with vertex color
189
190 entry.width = bitmap->width();
191 entry.height = bitmap->height();
192
193 glPixelStorei(GL_UNPACK_ROW_LENGTH, 0); // restore default
194 return; // Skip mipmap generation for A8 textures
195 } else if (bitmap->format() == ultralight::BitmapFormat::BGRA8_UNORM_SRGB) {
196 const void* pixels = bitmap->LockPixels();
197 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, bitmap->width(), bitmap->height(), 0, GL_BGRA, GL_UNSIGNED_BYTE, pixels);
198 bitmap->UnlockPixels();
199 } else {
200 GPU_FATAL("Unhandled texture format: " << static_cast<int>(bitmap->format()));
201 }
202
203 CHECK_GL();
204 glGenerateMipmap(GL_TEXTURE_2D);
205 CHECK_GL();
206
207 entry.width = bitmap->width();
208 entry.height = bitmap->height();
209
210 glPixelStorei(GL_UNPACK_ROW_LENGTH, 0); // restore default
211}

213void ToastGPUDriver::UpdateTexture(uint32_t texture_id, ultralight::RefPtr<ultralight::Bitmap> bitmap) {
214 PROFILE_ZONE_C(tracy::Color::Orange);
215
216 // Ensure correct GL context is current
217 glfwMakeContextCurrent(context_->active_window());
218
219 glActiveTexture(GL_TEXTURE0);
220 auto it = texture_map_.find(texture_id);
221
222 // If texture hasn't been created yet, create it using CreateTexture path
223 if (it == texture_map_.end() || it->second.tex_id == 0) {
224 // Forward to CreateTexture which will allocate and initialize the GL texture
225 CreateTexture(texture_id, bitmap);
226 return;
227 }
228
229 TextureEntry& entry = it->second;
230
231 // Check if dimensions changed - if so, we need to reallocate
232 bool needs_realloc = (entry.width != bitmap->width() || entry.height != bitmap->height());
233
234 glBindTexture(GL_TEXTURE_2D, entry.tex_id);
235
236 CHECK_GL();
237
238 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
239 glPixelStorei(GL_UNPACK_ROW_LENGTH, bitmap->row_bytes() / bitmap->bpp());
240
241 if (!bitmap->IsEmpty()) {
242 if (bitmap->format() == ultralight::BitmapFormat::A8_UNORM) {
243 const void* pixels = bitmap->LockPixels();
244 if (needs_realloc) {
245 // Texture size changed - reallocate
246 glTexImage2D(GL_TEXTURE_2D, 0, GL_R8, bitmap->width(), bitmap->height(), 0, GL_RED, GL_UNSIGNED_BYTE, pixels);
247 entry.width = bitmap->width();
248 entry.height = bitmap->height();
249 } else {
250 // Same size - use subimage for efficiency
251 glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, bitmap->width(), bitmap->height(), GL_RED, GL_UNSIGNED_BYTE, pixels);
252 }
253 bitmap->UnlockPixels();
254 // Don't generate mipmaps for A8 textures
255 } else if (bitmap->format() == ultralight::BitmapFormat::BGRA8_UNORM_SRGB) {
256 const void* pixels = bitmap->LockPixels();
257 if (needs_realloc) {
258 // Texture size changed - reallocate
259 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, bitmap->width(), bitmap->height(), 0, GL_BGRA, GL_UNSIGNED_BYTE, pixels);
260 entry.width = bitmap->width();
261 entry.height = bitmap->height();
262 } else {
263 // Same size - use subimage for efficiency
264 glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, bitmap->width(), bitmap->height(), GL_BGRA, GL_UNSIGNED_BYTE, pixels);
265 }
266 bitmap->UnlockPixels();
267
268 CHECK_GL();
269 glGenerateMipmap(GL_TEXTURE_2D);
270 } else {
271 GPU_FATAL("Unhandled texture format: " << static_cast<int>(bitmap->format()));
272 }
273 }
274
275 glPixelStorei(GL_UNPACK_ROW_LENGTH, 0); // restore default
276 CHECK_GL();
277}

279void ToastGPUDriver::BindTexture(uint8_t texture_unit, uint32_t texture_id) {
280 glActiveTexture(GL_TEXTURE0 + texture_unit);
281
282 if (texture_id == 0) {
283 // Bind fallback texture instead of nothing
284 glBindTexture(GL_TEXTURE_2D, fallback_texture_id_);
285 return;
286 }
287
288 auto it = texture_map_.find(texture_id);
289 if (it == texture_map_.end()) {
290 // Log warning about missing texture
291 static std::set<uint32_t> warned_missing;
292 if (warned_missing.find(texture_id) == warned_missing.end()) {
293 TOAST_WARN("[GPU] BindTexture: texture_id {} not found in texture_map_! (unit {}) - using fallback", texture_id, texture_unit);
294 warned_missing.insert(texture_id);
295 }
296 // Bind fallback texture instead of nothing
297 glBindTexture(GL_TEXTURE_2D, fallback_texture_id_);
298 return;
299 }
300
301 TextureEntry& entry = it->second;
302
303 // Verify the GL texture is valid
304 if (entry.tex_id == 0 || !glIsTexture(entry.tex_id)) {
305 static std::set<uint32_t> warned_invalid;
306 if (warned_invalid.find(texture_id) == warned_invalid.end()) {
307 TOAST_WARN("[GPU] BindTexture: texture_id {} has invalid GL texture {} (unit {}) - using fallback", texture_id, entry.tex_id, texture_unit);
308 warned_invalid.insert(texture_id);
309 }
310 // Bind fallback texture instead of nothing
311 glBindTexture(GL_TEXTURE_2D, fallback_texture_id_);
312 return;
313 }
314
315 ResolveIfNeeded(entry.render_buffer_id);
316 glBindTexture(GL_TEXTURE_2D, entry.tex_id);
317
318 // Only set filter/wrap params, don't touch swizzle (it's set at creation time)
319 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
320 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
321 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
322 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
323
324 CHECK_GL();
325}

533void ToastGPUDriver::DrawCommandList() {
534 PROFILE_ZONE_C(tracy::Color::Orange);
535
536 // Copy command list under lock, then process outside the lock to avoid holding
537 // the mutex during GL calls (which could be re-entrant via callbacks).
538 std::vector<ultralight::Command> local_commands;
539 {
540 std::lock_guard<std::mutex> lock(command_list_mutex_);
541 if (command_list_.empty()) {
542 return;
543 }
544 local_commands = command_list_;
545 command_list_.clear();
546 }
547
548 glfwMakeContextCurrent(context_->active_window());
549
550 CHECK_GL();
551
552 batch_count_ = 0;
553
554 glEnable(GL_BLEND);
555 glDisable(GL_SCISSOR_TEST);
556 glDisable(GL_DEPTH_TEST);
557 glDepthFunc(GL_NEVER);
558 glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA);
559
560 CHECK_GL();
561
562 for (const auto& cmd : local_commands) {
563 switch (cmd.command_type) {
564 case ultralight::CommandType::DrawGeometry: DrawGeometry(cmd.geometry_id, cmd.indices_count, cmd.indices_offset, cmd.gpu_state); break;
565 case ultralight::CommandType::ClearRenderBuffer: ClearRenderBuffer(cmd.gpu_state.render_buffer_id); break;
566 }
567 }
568
569 // local_commands cleared on scope exit
570 glDisable(GL_SCISSOR_TEST);
571
572 glBindFramebuffer(GL_FRAMEBUFFER, 0);
573 CHECK_GL();
574}

576void ToastGPUDriver::DrawGeometry(uint32_t geometry_id, uint32_t indices_count, uint32_t indices_offset, const ultralight::GPUState& state) {
577 PROFILE_ZONE_C(tracy::Color::Orange);
578
579 glfwMakeContextCurrent(context_->active_window());
580
581 if (programs_.empty()) {
582 LoadPrograms();
583 }
584
585 BindRenderBuffer(state.render_buffer_id);
586
587 SetViewport(state.viewport_width, state.viewport_height);
588
589 GeometryEntry& geometry = geometry_map_[geometry_id];
590 SelectProgram(static_cast<ProgramType>(state.shader_type));
591 UpdateUniforms(state);
592
593 CHECK_GL();
594
595 CreateVAOIfNeededForActiveContext(geometry_id);
596 auto vao_entry = geometry.vao_map[glfwGetCurrentContext()];
597 glBindVertexArray(vao_entry);
598 CHECK_GL();
599
600 BindTexture(0, state.texture_1_id);
601 BindTexture(1, state.texture_2_id);
602 BindTexture(2, state.texture_3_id);
603
604 CHECK_GL();
605
606 if (state.enable_scissor) {
607 glEnable(GL_SCISSOR_TEST);
608 const ultralight::IntRect& r = state.scissor_rect;
609 glScissor(r.left, r.top, r.right - r.left, r.bottom - r.top);
610 } else {
611 glDisable(GL_SCISSOR_TEST);
612 }
613
614 if (state.enable_blend) {
615 glEnable(GL_BLEND);
616 } else {
617 glDisable(GL_BLEND);
618 }
619
620 CHECK_GL();
621 glDrawElements(GL_TRIANGLES, indices_count, GL_UNSIGNED_INT, reinterpret_cast<GLvoid*>(indices_offset * sizeof(unsigned int)));
622 CHECK_GL();
623 glBindVertexArray(0);
624
625 batch_count_++;
626
627 CHECK_GL();
628}

651void ToastGPUDriver::LoadProgram(ProgramType type) {
652 ProgramEntry prog;
653
654 if (type == ultralight::ShaderType::Fill) {
655 prog.vert_shader_id =
656 LoadShaderFromSource(GL_VERTEX_SHADER, ultralight::shaders::shader_v2f_c4f_t2f_t2f_d28f_vert().c_str(), "shader_v2f_c4f_t2f_t2f_d28f.vert");
657 prog.frag_shader_id = LoadShaderFromSource(GL_FRAGMENT_SHADER, ultralight::shaders::shader_fill_frag().c_str(), "shader_fill.frag");
658 } else if (type == ultralight::ShaderType::FillPath) {
659 prog.vert_shader_id = LoadShaderFromSource(GL_VERTEX_SHADER, ultralight::shaders::shader_v2f_c4f_t2f_vert().c_str(), "shader_v2f_c4f_t2f.vert");
660 prog.frag_shader_id = LoadShaderFromSource(GL_FRAGMENT_SHADER, ultralight::shaders::shader_fill_path_frag().c_str(), "shader_fill_path.frag");
661 }
662
663 prog.program_id = glCreateProgram();
664 glAttachShader(prog.program_id, prog.vert_shader_id);
665 glAttachShader(prog.program_id, prog.frag_shader_id);
666
667 glBindAttribLocation(prog.program_id, 0, "in_Position");
668 glBindAttribLocation(prog.program_id, 1, "in_Color");
669 glBindAttribLocation(prog.program_id, 2, "in_TexCoord");
670
671 if (type == ultralight::ShaderType::Fill) {
672 glBindAttribLocation(prog.program_id, 3, "in_ObjCoord");
673 glBindAttribLocation(prog.program_id, 4, "in_Data0");
674 glBindAttribLocation(prog.program_id, 5, "in_Data1");
675 glBindAttribLocation(prog.program_id, 6, "in_Data2");
676 glBindAttribLocation(prog.program_id, 7, "in_Data3");
677 glBindAttribLocation(prog.program_id, 8, "in_Data4");
678 glBindAttribLocation(prog.program_id, 9, "in_Data5");
679 glBindAttribLocation(prog.program_id, 10, "in_Data6");
680 }
681
682 glLinkProgram(prog.program_id);
683
684 GLint linkStatus;
685 glGetProgramiv(prog.program_id, GL_LINK_STATUS, &linkStatus);
686 if (linkStatus == GL_FALSE) {
687 GPU_FATAL("Unable to link shader program.\n\tLog: " << GetProgramLog(prog.program_id));
688 }
689
690 glUseProgram(prog.program_id);
691
692 if (type == ultralight::ShaderType::Fill) {
693 glUniform1i(glGetUniformLocation(prog.program_id, "Texture1"), 0);
694 glUniform1i(glGetUniformLocation(prog.program_id, "Texture2"), 1);
695 glUniform1i(glGetUniformLocation(prog.program_id, "Texture3"), 2);
696 }
697
698 programs_[type] = prog;
699}

784void ToastGPUDriver::CreateFBOTexture(uint32_t texture_id, ultralight::RefPtr<ultralight::Bitmap> bitmap) {
785 CHECK_GL();
786
787 TextureEntry& entry = texture_map_[texture_id];
788 entry.width = bitmap->width();
789 entry.height = bitmap->height();
790
791 // Allocate a single-sampled texture
792 glGenTextures(1, &entry.tex_id);
793 glActiveTexture(GL_TEXTURE0);
794 glBindTexture(GL_TEXTURE_2D, entry.tex_id);
795
796 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
797 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
798 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
799 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
800
801 // Allocate texture in linear space
802 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, entry.width, entry.height, 0, GL_BGRA, GL_UNSIGNED_BYTE, nullptr);
803
804 if (context_->msaa_enabled()) {
805 // Allocate the multisampled texture
806 glGenTextures(1, &entry.msaa_tex_id);
807 glActiveTexture(GL_TEXTURE0);
808 glBindTexture(GL_TEXTURE_2D_MULTISAMPLE, entry.msaa_tex_id);
809 glTexImage2DMultisample(GL_TEXTURE_2D_MULTISAMPLE, 4, GL_RGBA8, entry.width, entry.height, GL_FALSE);
810 }
811
812 CHECK_GL();
813 glGenerateMipmap(GL_TEXTURE_2D);
814 CHECK_GL();
815}

817void ToastGPUDriver::CreateFBOIfNeededForActiveContext(uint32_t render_buffer_id) {
818 if (render_buffer_id == 0) {
819 return;
820 }
821
822 auto i = render_buffer_map_.find(render_buffer_id);
823 if (i == render_buffer_map_.end()) {
824 GPU_FATAL("Error, render buffer entry should exist here.");
825 return;
826 }
827
828 RenderBufferEntry& entry = i->second;
829 auto j = entry.fbo_map.find(glfwGetCurrentContext());
830 if (j != entry.fbo_map.end()) {
831 return; // Already exists
832 }
833
834 FBOEntry& fbo_entry = entry.fbo_map[glfwGetCurrentContext()];
835
836 glGenFramebuffers(1, &fbo_entry.fbo_id);
837 CHECK_GL();
838 glBindFramebuffer(GL_FRAMEBUFFER, fbo_entry.fbo_id);
839 CHECK_GL();
840
841 TextureEntry& textureEntry = texture_map_[entry.texture_id];
842
843 glBindTexture(GL_TEXTURE_2D, textureEntry.tex_id);
844 CHECK_GL();
845 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, textureEntry.tex_id, 0);
846 CHECK_GL();
847
848 GLenum drawBuffers[1] = { GL_COLOR_ATTACHMENT0 };
849 glDrawBuffers(1, drawBuffers);
850 CHECK_GL();
851
852 GLenum result = glCheckFramebufferStatus(GL_FRAMEBUFFER);
853 if (result != GL_FRAMEBUFFER_COMPLETE) {
854 GPU_FATAL("Error creating FBO: " << result);
855 }
856 CHECK_GL();
857
858 if (!context_->msaa_enabled()) {
859 return;
860 }
861
862 // Create MSAA FBO
863 glGenFramebuffers(1, &fbo_entry.msaa_fbo_id);
864 CHECK_GL();
865 glBindFramebuffer(GL_FRAMEBUFFER, fbo_entry.msaa_fbo_id);
866 CHECK_GL();
867
868 glBindTexture(GL_TEXTURE_2D_MULTISAMPLE, textureEntry.msaa_tex_id);
869 CHECK_GL();
870 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D_MULTISAMPLE, textureEntry.msaa_tex_id, 0);
871 CHECK_GL();
872
873 glDrawBuffers(1, drawBuffers);
874 CHECK_GL();
875
876 result = glCheckFramebufferStatus(GL_FRAMEBUFFER);
877 if (result != GL_FRAMEBUFFER_COMPLETE) {
878 GPU_FATAL("Error creating MSAA FBO: " << result);
879 }
880 CHECK_GL();
881}

883void ToastGPUDriver::CreateVAOIfNeededForActiveContext(uint32_t geometry_id) {
884 auto i = geometry_map_.find(geometry_id);
885 if (i == geometry_map_.end()) {

949void ToastGPUDriver::ResolveIfNeeded(uint32_t render_buffer_id) {
950 if (!context_->msaa_enabled()) {
951 return;
952 }
953
954 if (render_buffer_id == 0) {
955 return;
956 }
957
958 auto it = render_buffer_map_.find(render_buffer_id);
959 if (it == render_buffer_map_.end()) {
960 return;
961 }
962
963 RenderBufferEntry& renderBufferEntry = it->second;
964 if (!renderBufferEntry.texture_id) {
965 return;
966 }
967
968 auto i = renderBufferEntry.fbo_map.find(glfwGetCurrentContext());
969 if (i == renderBufferEntry.fbo_map.end()) {
970 return;
971 }
972
973 FBOEntry& fbo_entry = i->second;
974
975 TextureEntry& textureEntry = texture_map_[renderBufferEntry.texture_id];
976 if (fbo_entry.needs_resolve) {
977 GLint drawFboId = 0, readFboId = 0;
978 glGetIntegerv(GL_DRAW_FRAMEBUFFER_BINDING, &drawFboId);
979 glGetIntegerv(GL_READ_FRAMEBUFFER_BINDING, &readFboId);
980 CHECK_GL();
981
982 glBindFramebuffer(GL_DRAW_FRAMEBUFFER, fbo_entry.fbo_id);
983 glBindFramebuffer(GL_READ_FRAMEBUFFER, fbo_entry.msaa_fbo_id);
984 CHECK_GL();
985
986 glBlitFramebuffer(0, 0, textureEntry.width, textureEntry.height, 0, 0, textureEntry.width, textureEntry.height, GL_COLOR_BUFFER_BIT, GL_NEAREST);
987 CHECK_GL();
988
989 glBindFramebuffer(GL_DRAW_FRAMEBUFFER, drawFboId);
990 glBindFramebuffer(GL_READ_FRAMEBUFFER, readFboId);
991 CHECK_GL();
992
993 fbo_entry.needs_resolve = false;
994 }
995}
