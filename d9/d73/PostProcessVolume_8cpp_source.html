71void PostProcessVolume::Load(json_t j, bool force_create) {
72 Actor::Load(j, force_create);
73 m_priority = j.value("priority", m_priority);
74 m_halfExtents = j.value("halfExtents", m_halfExtents);
75 m_enabledVolume = j.value("enabledVolume", m_enabledVolume);
76
77 if (j.contains("volumeStack")) {
78 auto* rendererBase = renderer::IRendererBase::GetInstance();
79 auto* ppm = rendererBase ? rendererBase->GetPostProcessManager() : nullptr;
80 if (ppm) {
81 (void)ppm->LoadStackInto(j.at("volumeStack"), m_stack);
82 } else if (j.at("volumeStack").is_array()) {
83 m_stack.clear();
84 for (const auto& item : j.at("volumeStack")) {
85 if (!item.is_object() || !item.contains("type")) {
86 continue;
87 }
88 auto pass = renderer::post::CreateByType(item.at("type").get<std::string>());
89 if (!pass) {
90 continue;
91 }
92 pass->LoadState(item);
93 m_stack.push_back(std::move(pass));
94 }
95 }
96 }

109bool PostProcessVolume::IsCameraInside2D() const {
110 auto* rendererBase = renderer::IRendererBase::GetInstance();
111 auto* activeCamera = rendererBase ? rendererBase->GetActiveCamera() : nullptr;
112 if (!activeCamera) {
113 return false;
114 }
115
116 const glm::vec3 volumePos = const_cast<PostProcessVolume*>(this)->transform()->worldPosition();
117 const glm::vec3 camPos = activeCamera->transform()->worldPosition();
118
119 return glm::abs(camPos.x - volumePos.x) <= m_halfExtents.x && glm::abs(camPos.y - volumePos.y) <= m_halfExtents.y;
120}
