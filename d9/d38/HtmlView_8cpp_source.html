52void HtmlView::CreateUlView() {
53 if (m_view || m_url.empty()) {
54 return;
55 }
56
57 auto* hud = renderer::HUD::HUDLayer::Get();
58 if (!hud) {
59 TOAST_ERROR("HtmlView: HUDLayer not available");
60 return;
61 }
62
63 auto* fb = hud->GetFramebuffer();
64 if (!fb) {
65 TOAST_ERROR("HtmlView: HUDLayer framebuffer not ready");
66 return;
67 }
68
69 m_view = hud->CreateView(fb->Width(), fb->Height());
70 if (m_view) {
71 // If a console callback is set, install a per-view listener
72 if (m_consoleCb) {
73 m_viewListener = std::make_unique<HtmlViewListener>(m_consoleCb);
74 m_view->set_view_listener(m_viewListener.get());
75 }
76 m_view->LoadURL(ultralight::String(m_url.c_str()));
77 hud->SetViewSortOrder(m_view, m_sortOrder);
78 TOAST_INFO("HtmlView: loaded {}", m_url);
79 }
80}
