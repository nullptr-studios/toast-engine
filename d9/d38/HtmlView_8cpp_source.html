124void HtmlView::CreateUlView() {
125 if (m_view || m_url.empty()) {
126 return;
127 }
128
129 auto* hud = renderer::HUD::HUDLayer::Get();
130 if (!hud) {
131 TOAST_ERROR("HtmlView: HUDLayer not available");
132 return;
133 }
134
135 auto* fb = hud->GetFramebuffer();
136 if (!fb) {
137 TOAST_ERROR("HtmlView: HUDLayer framebuffer not ready");
138 return;
139 }
140
141 m_view = hud->CreateView(fb->Width(), fb->Height());
142 if (m_view) {
143 // If a console callback is set, install a per-view listener
144 if (m_consoleCb) {
145 m_viewListener = std::make_unique<HtmlViewListener>(m_consoleCb, m_view->view_listener());
146 m_view->set_view_listener(m_viewListener.get());
147 }
148 if (m_domReadyCb) {
149 m_loadListener = std::make_unique<HtmlLoadListener>(m_domReadyCb, m_view->load_listener());
150 m_view->set_load_listener(m_loadListener.get());
151 }
152 m_view->LoadURL(ultralight::String(m_url.c_str()));
153 hud->SetViewSortOrder(m_view, m_sortOrder);
154 hud->ExecuteJS(Localization::BuildApplyScript());
155 TOAST_INFO("HtmlView: loaded {}", m_url);
156 }
157}
