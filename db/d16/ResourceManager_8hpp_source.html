231template<typename R, typename... Args>
232std::shared_ptr<R> ResourceManager::LoadResource(const std::string& path, Args&&... args) {
233 static_assert(std::is_base_of_v<IResource, R>, "Must be IResource type");
234
235 // Normalize path to use forward slashes
236 std::string formattedPath = ToForwardSlashes(path);
237
238 // TOAST_INFO("Loading resource: {0}", formattedPath);
239
240 // Fast path: try to return cached
241 {
242 std::lock_guard lock(m_mtx);
243 auto it = m_cachedResources.find(formattedPath);
244 if (it != m_cachedResources.end()) {
245 // lock the base weak_ptr to get a shared_ptr<IResource>
246 if (auto baseSp = it->second) {
247 // try to cast to derived R
248 if (auto derivedSp = std::dynamic_pointer_cast<R>(baseSp)) {
249 return derivedSp;
250 }
251 // cached resource exists but not of requested derived type
252 return std::shared_ptr<R> {};
253 }
254 // else: expired weak_ptr -> fallthrough to create new
255 }
256 }
257
258 // Create the object first (owning pointer) - perfect-forward extra args (optional)
259 auto res = std::make_shared<R>(formattedPath, std::forward<Args>(args)...);
260
261 res->Load();
262
263 {
264 std::lock_guard lock(m_mtx);
265 m_cachedResources[formattedPath] = res;
266 }
267
268 // If the resource needs GPU upload, enqueue an upload job that captures the shared_ptr.
269 if (res->IsGPU()) {
270 if (m_mainThreadId == std::this_thread::get_id()) {
271 // If we're already on the main thread, load immediately
272 res->LoadMainThread();
273 } else {
274 std::lock_guard lock(m_uploadMtx);
275 m_uploadResources.emplace_back(res);
276 }
277 }
278
279 // Return the shared_ptr<R> for the caller
280 return res;
281}
