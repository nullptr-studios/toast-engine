99template<typename R, typename... Args>
100std::shared_ptr<R> ResourceManager::LoadResource(const std::string& path, Args&&... args) {
101 static_assert(std::is_base_of_v<IResource, R>, "Must be IResource type");
102
103 // Normalize path to use forward slashes
104 std::string formattedPath = ToForwardSlashes(path);
105
106 TOAST_INFO("Loading resource: {0}", formattedPath);
107
108 // Fast path: try to return cached
109 {
110 std::lock_guard lock(m_mtx);
111 auto it = m_cachedResources.find(formattedPath);
112 if (it != m_cachedResources.end()) {
113 // lock the base weak_ptr to get a shared_ptr<IResource>
114 if (auto baseSp = it->second) {
115 // try to cast to derived R
116 if (auto derivedSp = std::dynamic_pointer_cast<R>(baseSp)) {
117 return derivedSp;
118 }
119 // cached resource exists but not of requested derived type
120 return std::shared_ptr<R> {};
121 }
122 // else: expired weak_ptr -> fallthrough to create new
123 }
124 }
125
126 // Create the object first (owning pointer) - perfect-forward extra args (optional)
127 auto res = std::make_shared<R>(formattedPath, std::forward<Args>(args)...);
128
129 // Insert weak_ptr into cache BEFORE performing the expensive load so other threads see it.
130 // Important: don't std::move(res) here â€” we still need the shared_ptr locally to call Load().
131 {
132 std::lock_guard lock(m_mtx);
133 m_cachedResources[formattedPath] = res; // constructs weak_ptr<IResource> from shared_ptr<R>
134 }
135
136 // Now load without holding the resource map mutex.
137 res->Load();
138
139 // If the resource needs GPU upload, enqueue an upload job that captures the shared_ptr.
140 if (res->IsGPU()) {
141 if (m_mainThreadId == std::this_thread::get_id()) {
142 // If we're already on the main thread, load immediately
143 res->LoadMainThread();
144 } else {
145 std::lock_guard lock(m_uploadMtx);
146 m_uploadResources.emplace_back(res);
147 }
148 }
149
150 // Return the shared_ptr<R> for the caller
151 return res;
152}
