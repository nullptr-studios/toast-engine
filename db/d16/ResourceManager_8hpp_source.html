214template<typename R, typename... Args>
215std::shared_ptr<R> ResourceManager::LoadResource(const std::string& path, Args&&... args) {
216 static_assert(std::is_base_of_v<IResource, R>, "Must be IResource type");
217
218 // Normalize path to use forward slashes
219 std::string formattedPath = ToForwardSlashes(path);
220
221 TOAST_INFO("Loading resource: {0}", formattedPath);
222
223 // Fast path: try to return cached
224 {
225 std::lock_guard lock(m_mtx);
226 auto it = m_cachedResources.find(formattedPath);
227 if (it != m_cachedResources.end()) {
228 // lock the base weak_ptr to get a shared_ptr<IResource>
229 if (auto baseSp = it->second) {
230 // try to cast to derived R
231 if (auto derivedSp = std::dynamic_pointer_cast<R>(baseSp)) {
232 return derivedSp;
233 }
234 // cached resource exists but not of requested derived type
235 return std::shared_ptr<R> {};
236 }
237 // else: expired weak_ptr -> fallthrough to create new
238 }
239 }
240
241 // Create the object first (owning pointer) - perfect-forward extra args (optional)
242 auto res = std::make_shared<R>(formattedPath, std::forward<Args>(args)...);
243
244 // Insert weak_ptr into cache BEFORE performing the expensive load so other threads see it.
245 // Important: don't std::move(res) here â€” we still need the shared_ptr locally to call Load().
246 {
247 std::lock_guard lock(m_mtx);
248 m_cachedResources[formattedPath] = res; // constructs weak_ptr<IResource> from shared_ptr<R>
249 }
250
251 // Now load without holding the resource map mutex.
252 res->Load();
253
254 // If the resource needs GPU upload, enqueue an upload job that captures the shared_ptr.
255 if (res->IsGPU()) {
256 if (m_mainThreadId == std::this_thread::get_id()) {
257 // If we're already on the main thread, load immediately
258 res->LoadMainThread();
259 } else {
260 std::lock_guard lock(m_uploadMtx);
261 m_uploadResources.emplace_back(res);
262 }
263 }
264
265 // Return the shared_ptr<R> for the caller
266 return res;
267}
