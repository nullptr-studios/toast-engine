219template<typename R, typename... Args>
220std::shared_ptr<R> ResourceManager::LoadResource(const std::string& path, Args&&... args) {
221 static_assert(std::is_base_of_v<IResource, R>, "Must be IResource type");
222
223 // Normalize path to use forward slashes
224 std::string formattedPath = ToForwardSlashes(path);
225
226 TOAST_INFO("Loading resource: {0}", formattedPath);
227
228 // Fast path: try to return cached
229 {
230 std::lock_guard lock(m_mtx);
231 auto it = m_cachedResources.find(formattedPath);
232 if (it != m_cachedResources.end()) {
233 // lock the base weak_ptr to get a shared_ptr<IResource>
234 if (auto baseSp = it->second) {
235 // try to cast to derived R
236 if (auto derivedSp = std::dynamic_pointer_cast<R>(baseSp)) {
237 return derivedSp;
238 }
239 // cached resource exists but not of requested derived type
240 return std::shared_ptr<R> {};
241 }
242 // else: expired weak_ptr -> fallthrough to create new
243 }
244 }
245
246 // Create the object first (owning pointer) - perfect-forward extra args (optional)
247 auto res = std::make_shared<R>(formattedPath, std::forward<Args>(args)...);
248
249 // Insert weak_ptr into cache BEFORE performing the expensive load so other threads see it.
250 // Important: don't std::move(res) here — we still need the shared_ptr locally to call Load().
251 {
252 std::lock_guard lock(m_mtx);
253 m_cachedResources[formattedPath] = res; // constructs weak_ptr<IResource> from shared_ptr<R>
254 }
255
256 // Now load without holding the resource map mutex.
257 res->Load();
258
259 // If the resource needs GPU upload, enqueue an upload job that captures the shared_ptr.
260 if (res->IsGPU()) {
261 if (m_mainThreadId == std::this_thread::get_id()) {
262 // If we're already on the main thread, load immediately
263 res->LoadMainThread();
264 } else {
265 std::lock_guard lock(m_uploadMtx);
266 m_uploadResources.emplace_back(res);
267 }
268 }
269
270 // Return the shared_ptr<R> for the caller
271 return res;
272}
