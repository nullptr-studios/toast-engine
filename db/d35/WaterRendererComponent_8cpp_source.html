49void WaterRendererComponent::OnRender(renderer::IRenderablePass pass, const glm::mat4& precomputed_mat) noexcept {
50 if (pass != renderer::IRenderablePass::GEOMETRY || !enabled()) {
51 return;
52 }
53
54 auto mesh = GetMesh().lock();
55 if (!mesh || !m_waterShader || m_refractionTextureId == 0) {
56 return;
57 }
58
59 if (!OclussionVolume::isTransformedAABBOnPlanes(mesh->boundingBox(), GetWorldMatrix())) {
60 return;
61 }
62
63 if (!m_normalTexture || m_normalTexture->id() == 0) {
64 if (!m_normalTexture) {
65 TOAST_WARN("WaterRendererComponent: missing normal texture EDITOR/WaterNormal.jpg");
66 }
67 return;
68 }
69
70 const glm::mat4 model = GetWorldMatrix();
71 const glm::mat4 mvp = precomputed_mat * model;
72
73 m_waterShader->Use();
74 m_waterShader->Set("gWorld", model);
75 m_waterShader->Set("gMVP", mvp);
76 m_waterShader->Set("gColor", m_tint);
77 m_waterShader->Set("uRefractionStrengthX", m_refractionStrengthX);
78 m_waterShader->Set("uRefractionStrengthY", m_refractionStrengthY);
79 m_waterShader->Set("uBaseOffsetX", m_baseOffsetX);
80 m_waterShader->Set("uBaseOffsetY", m_baseOffsetY);
81 m_waterShader->Set("uNormalStrength", m_normalStrength);
82 m_waterShader->Set("uNormalTiling", m_normalTiling);
83 m_waterShader->Set("uOpacity", m_opacity);
84 m_waterShader->Set("uTintMix", m_tintMix);
85 m_waterShader->Set("uFlipY", m_flipSceneY ? 1.0f : 0.0f);
86 m_waterShader->Set("uWaveSpeed1", m_waveSpeed1);
87 m_waterShader->Set("uWaveSpeed2", m_waveSpeed2);
88 m_waterShader->Set("uDistortionMix", m_distortionMix);
89 m_waterShader->Set("uStraightDistortStrength", m_straightDistortStrength);
90 m_waterShader->Set("uStraightBlend", m_straightBlend);
91 m_waterShader->Set("uVerticalPull", m_verticalPull);
92 m_waterShader->Set("uCausticStrength", m_causticStrength);
93 m_waterShader->Set("uCausticScale", m_causticScale);
94 m_waterShader->Set("uCausticSpeed", m_causticSpeed);
95 m_waterShader->Set("uCausticSharpness", m_causticSharpness);
96 m_waterShader->Set("time", static_cast<float>(Time::uptime()));
97
98 glActiveTexture(GL_TEXTURE0);
99 glBindTexture(GL_TEXTURE_2D, m_refractionTextureId);
100 m_waterShader->SetSampler("uSceneTex", 0);
101
102 glActiveTexture(GL_TEXTURE1);
103 glBindTexture(GL_TEXTURE_2D, m_normalTexture->id());
104 m_waterShader->SetSampler("uNormalMap", 1);
105
106 glDepthMask(GL_TRUE);
107 glEnable(GL_BLEND);
108 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
109
110 mesh->Draw();
111}
