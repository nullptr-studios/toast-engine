10static bool isPointInTriangle(const vec2& p, const glm::vec2& a, const glm::vec2& b, const glm::vec2& c) {
11 vec2 ab = b - a;
12 vec2 bc = c - b;
13 vec2 ca = a - c;
14
15 vec2 ap = p - a;
16 vec2 bp = p - b;
17 vec2 cp = p - c;
18
19 float cross1 = determinant(glm::mat2{ab, ap});
20 float cross2 = determinant(glm::mat2{bc, bp});
21 float cross3 = determinant(glm::mat2{ca, cp});
22
23 // If the point is inside, all cross products will have the same sign.
24 bool has_neg = (cross1 < 0.0f) || (cross2 < 0.0f) || (cross3 < 0.0f);
25 bool has_pos = (cross1 > 0.0f) || (cross2 > 0.0f) || (cross3 > 0.0f);
26
27 return !(has_neg && has_pos);
28}

30static std::vector<uint16_t> triangulate(const std::vector<vec3>& vertices) {
31 std::vector<uint16_t> indices;
32
33 // A polygon must have at least 3 vertices
34 if (vertices.size() < 3) {
35 return indices;
36 }
37
38 // Create a working list of indices representing the remaining polygon
39 std::vector<size_t> remaining_indices(vertices.size());
40 for (size_t i = 0; i < vertices.size(); ++i) {
41 remaining_indices[i] = i;
42 }
43
44 // 1. Determine the overall winding order of the polygon.
45 // Even if you don't care about the output winding, the algorithm needs to
46 // know what "inside" vs "outside" means to find convex angles.
47 float area = 0.0f;
48 for (size_t i = 0; i < vertices.size(); ++i) {
49 size_t j = (i + 1) % vertices.size();
50 area += vertices[i].x * vertices[j].y - vertices[j].x * vertices[i].y;
51 }
52
53 // If the area is negative, the vertices are clockwise.
54 // We reverse our working indices to process everything as counter-clockwise.
55 if (area < 0.0f) {
56 std::reverse(remaining_indices.begin(), remaining_indices.end());
57 }
58
59 // 2. Ear Clipping Loop
60 // We use a safety counter to prevent infinite loops if the polygon is
61 // self-intersecting or contains degenerate (overlapping) data.
62 size_t safety_counter = vertices.size() * 2;
63
64 while (remaining_indices.size() > 3) {
65 if (--safety_counter == 0) break;
66
67 bool ear_found = false;
68 size_t count = remaining_indices.size();
69
70 for (size_t i = 0; i < count; ++i) {
71 size_t prev_idx = remaining_indices[(i + count - 1) % count];
72 size_t curr_idx = remaining_indices[i];
73 size_t next_idx = remaining_indices[(i + 1) % count];
74
75 const vec2& prev = vertices[prev_idx];
76 const vec2& curr = vertices[curr_idx];
77 const vec2& next = vertices[next_idx];
78
79 // Check if the vertex is convex (interior angle < 180).
80 // Because we forced CCW, the cross product of the edges must be positive.
81 vec2 edge1 = curr - prev;
82 vec2 edge2 = next - curr;
83 if (determinant(glm::mat2{edge1, edge2}) <= 0.00001f) {
84 continue; // Angle is concave or collinear, cannot be an ear tip
85 }
86
87 // Check if any OTHER vertex of the polygon is inside this triangle
88 bool is_ear = true;
89 for (size_t j = 0; j < count; ++j) {
90 if (j == i || j == (i + count - 1) % count || j == (i + 1) % count) {
91 continue; // Skip the vertices that make up the triangle itself
92 }
93
94 size_t test_idx = remaining_indices[j];
95 if (isPointInTriangle(vertices[test_idx], prev, curr, next)) {
96 is_ear = false;
97 break;
98 }
99 }
100
101 // If it's a valid ear, clip it!
102 if (is_ear) {
103 indices.push_back(prev_idx);
104 indices.push_back(curr_idx);
105 indices.push_back(next_idx);
106
107 // Remove the ear tip from our working list
108 remaining_indices.erase(remaining_indices.begin() + i);
109 ear_found = true;
110 break;
111 }
112 }
113
114 // If we looped through all remaining vertices and couldn't find an ear,
115 // the polygon is likely invalid (e.g., self-intersecting). Break out.
116 if (!ear_found) {
117 break;
118 }
119 }
120
121 // 3. Add the final remaining 3 vertices as the last triangle
122 if (remaining_indices.size() == 3) {
123 indices.push_back(remaining_indices[0]);
124 indices.push_back(remaining_indices[1]);
125 indices.push_back(remaining_indices[2]);
126 }
127
128 return indices;
129}
