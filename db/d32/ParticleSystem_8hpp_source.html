53

57 void LoadFromLua(const sol::table& table) {
58 if (table.size() >= 2) {
59 sol::optional<T> minOpt = table[1];
60 sol::optional<T> maxOpt = table[2];
61 if (minOpt) {
62 min = *minOpt;
63 }
64 if (maxOpt) {
65 max = *maxOpt;
66 }
67 } else if (table.size() == 1) {
68 sol::optional<T> valOpt = table[1];
69 if (valOpt) {
70 min = max = *valOpt;
71 }
72 }
73 }
74
76 void LoadFromLua(const sol::object& obj) {
77 if (obj.is<sol::table>()) {
78 LoadFromLua(obj.as<sol::table>());
79 } else if (obj.is<T>()) {
80 min = max = obj.as<T>();
81 }
82 }
