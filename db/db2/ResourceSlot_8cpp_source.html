129void editor::ResourceSlot::ProcessDrop(Entry* e) {
130 if (!e) {
131 return;
132 }
133
134 if (!CheckCorrectType(e)) {
135 // Show a modal popup to the user explaining the bad drop
136 m_typeErrorMessage = "Invalid resource type: " + e->name + "\nExpected: ";
137 // append expected type name
138 switch (m_requiredType) {
139 case resource::ResourceType::TEXTURE: m_typeErrorMessage += "Texture"; break;
140 case resource::ResourceType::MODEL: m_typeErrorMessage += "Model"; break;
141 case resource::ResourceType::AUDIO: m_typeErrorMessage += "Audio"; break;
142 case resource::ResourceType::SHADER: m_typeErrorMessage += "Shader"; break;
143 case resource::ResourceType::FONT: m_typeErrorMessage += "Font"; break;
144 default: m_typeErrorMessage += "Unknown"; break;
145 }
146
147 // Append allowed extensions if available
148 auto it = s_resourceRegistry.find(m_requiredType);
149 if (it != s_resourceRegistry.end()) {
150 const auto& exts = it->second.extensions;
151 if (!exts.empty()) {
152 m_typeErrorMessage += "\nAllowed extensions: ";
153 for (size_t i = 0; i < exts.size(); ++i) {
154 m_typeErrorMessage += exts[i];
155 if (i + 1 < exts.size()) {
156 m_typeErrorMessage += ", ";
157 }
158 }
159 }
160 }
161
162 m_showTypeErrorPopup = true;
163
164 TOAST_WARN("ResourceSlot: Dropped resource has invalid extension: {}", e->name);
165 return;
166 }
167 m_selectedEntry = *e;
168
169 // Always notify listeners with the path, regardless of loading success
170 // (we care about the path, not whether it can be loaded)
171 if (m_onDropped) {
172 m_onDropped(ToForwardSlashes(e->relativePath.string()));
173 }
174
175 std::shared_ptr<IResource> resource;
176 auto it = s_resourceRegistry.find(m_requiredType);
177 if (it != s_resourceRegistry.end() && it->second.loader) {
178 resource = it->second.loader(ToForwardSlashes(e->relativePath.string()));
179 } else {
180 TOAST_WARN("ResourceSlot: No loader registered for resource type {}", static_cast<int>(m_requiredType));
181 }
182
183 if (resource) {
184 // m_resource = resource;
185 TOAST_INFO("ResourceSlot: Bound resource: {}", e->name);
186 } else {
187 TOAST_WARN("ResourceSlot: Failed to load resource, but path stored: {}", e->name);
188 }
189}

339bool editor::ResourceSlot::CheckCorrectType(Entry* res) const {
340 EnsureRegistryInitialized();
341
342 auto it = s_resourceRegistry.find(m_requiredType);
343 if (it == s_resourceRegistry.end()) {
344 TOAST_WARN("ResourceSlot: Unknown or unregistered resource type");
345 return false;
346 }
347
348 const auto& exts = it->second.extensions;
349 if (exts.empty()) {
350 return true; // accept any
351 }
352
353 std::string ext = ToLower(res->extension);
354 auto found = std::find(exts.begin(), exts.end(), ext);
355 if (found == exts.end()) {
356 TOAST_WARN("ResourceSlot: Invalid resource type for {}: {}", static_cast<int>(m_requiredType), res->name);
357 return false;
358 }
359
360 return true;
361}
